country_event = {
	id = chaosx.weaponized_zombies.1
	title = chaosx.weaponized_zombies.1.t
	desc = chaosx.weaponized_zombies.1.d
	picture = GFX_report_event_generic_biowarfare

	is_triggered_only = yes

	immediate = {
		resolve_weaponized_zombie_profile = yes
		weaponized_zombie_refresh_raid_success_profile = yes
		set_technology = {
			zombie_disease_bomb_delivery_systems = 1
			popup = no
		}
		if = {
			limit = {
				has_country_flag = weaponized_zombies_completed
				check_variable = {
					var = num_equipment@zombie_disease_bomb_1
					value = 1
					compare = less_than
				}
			}
			clr_country_flag = weaponized_zombie_initial_stockpile_granted
		}
		ensure_weaponized_zombie_initial_stockpile = yes
		set_temp_variable = { wz_debug_completed = 0 }
		if = {
			limit = { has_country_flag = weaponized_zombies_completed }
			set_temp_variable = { wz_debug_completed = 1 }
		}
		set_temp_variable = { wz_debug_raid_low = 0 }
		if = {
			limit = { has_country_flag = weaponized_zombie_raid_success_low }
			set_temp_variable = { wz_debug_raid_low = 1 }
		}
		set_temp_variable = { wz_debug_raid_medium = 0 }
		if = {
			limit = { has_country_flag = weaponized_zombie_raid_success_medium }
			set_temp_variable = { wz_debug_raid_medium = 1 }
		}
		set_temp_variable = { wz_debug_raid_high = 0 }
		if = {
			limit = { has_country_flag = weaponized_zombie_raid_success_high }
			set_temp_variable = { wz_debug_raid_high = 1 }
		}
		set_temp_variable = { wz_debug_friendly_humans = 0 }
		if = {
			limit = { has_country_flag = weaponized_zombie_friendly_to_humans }
			set_temp_variable = { wz_debug_friendly_humans = 1 }
		}
		log = "[Chaos Redux] WZ RAID DEBUG country=[ROOT.GetTag] completed=[?wz_debug_completed|0] low=[?wz_debug_raid_low|0] medium=[?wz_debug_raid_medium|0] high=[?wz_debug_raid_high|0] friendly_humans=[?wz_debug_friendly_humans|0] bombs=[?num_equipment@zombie_disease_bomb_1|0]"
	}

	option = {
		name = chaosx.weaponized_zombies.1.a
	}
}

					}
				}
			}
			else = {
				set_country_flag = achievement_fight_fire_with_fire
			}
		}

country_event = {
	id = chaosx.weaponized_zombies.5
	title = chaosx.weaponized_zombies.5.t
	desc = chaosx.weaponized_zombies.5.d
	picture = GFX_report_event_generic_bioweapon

	trigger = {
		has_country_flag = weaponized_zombies_started
		is_special_project_being_researched = sp:weaponize_the_zombies
		NOT = { is_special_project_completed = sp:weaponize_the_zombies }
		NOT = { has_country_flag = weaponized_zombie_lab_accident_happened }
		has_country_flag = weaponized_zombie_step_resources_done
		NOT = { has_country_flag = weaponized_zombie_resource_minimal }
		OR = {
			has_country_flag = weaponized_zombie_resource_expanded
			has_country_flag = weaponized_zombie_resource_maximum
		}
		any_controlled_state = {
			biowarfare_facility > 0
		}
	}

	mean_time_to_happen = {
		years = constant:weaponized_zombies.accident_base_years

		modifier = {
			factor = constant:weaponized_zombies.accident_factor_risk_low
			check_variable = {
				var = weaponized_zombie_accident_risk
				value = constant:weaponized_zombies.accident_risk_threshold_low
				compare = greater_than_or_equals
			}
		}
		modifier = {
			factor = constant:weaponized_zombies.accident_factor_risk_medium
			check_variable = {
				var = weaponized_zombie_accident_risk
				value = constant:weaponized_zombies.accident_risk_threshold_medium
				compare = greater_than_or_equals
			}
		}
		modifier = {
			factor = constant:weaponized_zombies.accident_factor_risk_high
			check_variable = {
				var = weaponized_zombie_accident_risk
				value = constant:weaponized_zombies.accident_risk_threshold_high
				compare = greater_than_or_equals
			}
		}
		modifier = {
			factor = constant:weaponized_zombies.accident_factor_expanded
			has_country_flag = weaponized_zombie_resource_expanded
		}
		modifier = {
			factor = constant:weaponized_zombies.accident_factor_maximum
			has_country_flag = weaponized_zombie_resource_maximum
		}
		modifier = {
			factor = constant:weaponized_zombies.accident_factor_field_tested
			has_country_flag = weaponized_zombie_field_tested
		}
		modifier = {
			factor = constant:weaponized_zombies.accident_factor_domestic_test
			has_country_flag = weaponized_zombie_domestic_test
		}
		modifier = {
			factor = constant:weaponized_zombies.accident_factor_forbidden
			has_country_flag = weaponized_zombie_forbidden_branch
		}
		modifier = {
			factor = constant:weaponized_zombies.accident_factor_high_infectiousness
			check_variable = {
				var = weaponized_zombie_infectiousness
				value = 2
				compare = greater_than
			}
		}
		modifier = {
			factor = constant:weaponized_zombies.accident_factor_demonic
			has_country_flag = weaponized_zombie_nature_demonic
		}
		modifier = {
			factor = constant:weaponized_zombies.accident_factor_fail_safe_containment
			has_tech = fail_safe_containment_facilities
		}
		modifier = {
			factor = constant:weaponized_zombies.accident_factor_sealed_labs
			has_tech = sealed_containment_laboratories
		}
	}

	immediate = {
		hidden_effect = {
			set_country_flag = weaponized_zombie_lab_accident_happened
			set_country_flag = achievement_containment_was_temporary

			random_controlled_state = {
				limit = {
					biowarfare_facility > 0
				}
				save_event_target_as = weaponized_zombie_target_state
				save_event_target_as = outbreak_state
			}

			if = {
				limit = {
					check_variable = {
						var = num_equipment@zombie_disease_bomb_1
						value = 0
						compare = greater_than
					}
				}
				set_temp_variable = { weaponized_zombie_bombs_lost = num_equipment@zombie_disease_bomb_1 }
				multiply_temp_variable = { weaponized_zombie_bombs_lost = constant:weaponized_zombies.lab_accident_equipment_loss_ratio }
				round_temp_variable = weaponized_zombie_bombs_lost

				if = {
					limit = {
						check_variable = {
							var = weaponized_zombie_bombs_lost
							value = 1
							compare = less_than
						}
					}
					set_temp_variable = { weaponized_zombie_bombs_lost = 1 }
				}

				set_temp_variable = { weaponized_zombie_bombs_lost_negative = weaponized_zombie_bombs_lost }
				multiply_temp_variable = { weaponized_zombie_bombs_lost_negative = -1 }
				add_equipment_to_stockpile = {
					type = zombie_disease_bomb_1
					amount = weaponized_zombie_bombs_lost_negative
					producer = ROOT
				}
			}

			if = {
				limit = { has_event_target = outbreak_state }
				spawn_selected_dynamic_zombie_outbreak = yes
				news_event = { id = chaosx.news.3 days = 3 }
			}
		}
	}

	option = {
		name = chaosx.weaponized_zombies.5.a
		add_political_power = constant:weaponized_zombies.lab_accident_pp_loss
		add_stability = constant:weaponized_zombies.lab_accident_stability_loss
		add_war_support = constant:weaponized_zombies.lab_accident_warsupport_loss
	}
}
