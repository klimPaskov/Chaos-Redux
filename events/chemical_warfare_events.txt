country_event = {
	id = chaosx_contamination.9
	title = chaosx_contamination.9.t
	desc = chaosx_contamination.9.d
	picture = GFX_report_event_generic_sign_treaty2

	is_triggered_only = yes

	option = {
		name = chaosx_contamination.9.a
		trigger = {
			NOT = { has_country_flag = air_cleanliness_treaty_member }
			NOT = { has_country_flag = air_cleanliness_treaty_banned }
			NOT = { has_used_unconventional_weapons = yes }
			OR = {
				NOT = { has_variable = chem_warfare_condemnation }
				check_variable = {
					var = chem_warfare_condemnation
					value = constant:air_cleanliness_treaty.member_condemnation_max
					compare = less_than_or_equals
				}
			}
		}
		ai_chance = {
			base = constant:air_cleanliness_treaty_ai.join_base
			modifier = {
				factor = constant:air_cleanliness_treaty_ai.join_factor_democratic
				has_government = democratic
			}
			modifier = {
				factor = constant:air_cleanliness_treaty_ai.join_factor_minor_no_stockpile
				is_major = no
				NOT = { has_unconventional_weapon_stockpile = yes }
			}
			modifier = {
				factor = constant:air_cleanliness_treaty_ai.join_factor_has_stockpile
				has_unconventional_weapon_stockpile = yes
			}
			modifier = {
				factor = constant:air_cleanliness_treaty_ai.join_factor_used_unconventional
				has_used_unconventional_weapons = yes
			}
			modifier = {
				factor = constant:air_cleanliness_treaty_ai.join_factor_at_war_with_founder
				has_war_with = FROM
			}
			modifier = {
				factor = constant:air_cleanliness_treaty_ai.join_factor_major
				is_major = yes
			}
		}
		set_country_flag = air_cleanliness_treaty_member
		if = {
			limit = { FROM = { exists = yes } }
			add_opinion_modifier = {
				target = FROM
				modifier = air_cleanliness_treaty_member_respect
			}
		}
	}

	option = {
		name = chaosx_contamination.9.b
		ai_chance = {
			base = 1
		}
	}
}
