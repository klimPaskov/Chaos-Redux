add_namespace = chaosx.nr3

# Overview:
# - Holy Realm event ecosystem. Event 3 remains the stable fire-once entry.
# - The chain transforms Tibet or a Tibet-centered country into the Holy Realm,
#   tracks Mandala counters, logs evolution stages, opens doctrine decisions,
#   warns foreign players, and can resolve into the Final Silence world-end.

country_event = {
	id = chaosx.nr3.1
	title = chaosx.nr3.1.t
	is_triggered_only = yes
	hidden = yes

	immediate = {
		if = {
			limit = {
				TIB = { can_host_holy_realm_refuge = yes }
			}
			TIB = { country_event = { id = chaosx.nr3.2 days = 1 } }
		}
		else_if = {
			limit = {
				any_country = { can_host_holy_realm_refuge = yes }
			}
			random_country = {
				limit = { can_host_holy_realm_refuge = yes }
				country_event = { id = chaosx.nr3.2 days = 1 }
			}
		}
	}
}

	id = chaosx.nr3.2

	desc = chaosx.nr3.2.d
	picture = GFX_report_event_lhasa
	is_triggered_only = yes

	trigger = {
		can_host_holy_realm_refuge = yes
		NOT = { has_global_flag = holy_realm_exists }
	}

	option = {
		name = chaosx.nr3.2.a
		ai_chance = {
			base = 70
			modifier = {
				factor = 0.35
				check_variable = { global.chaos_meter_value < constant:holy_realm_chaos.formation_min }
			}
			modifier = {
				factor = 2
				OR = {
					tag = TIB
					original_tag = TIB
				}
			}
		}
		hidden_effect = {
			holy_realm_form_country = yes
			news_event = { id = chaosx.news.9 days = 2 }
			BHU = {
				if = {
					limit = {
						exists = yes
						NOT = { tag = ROOT }
					}
					country_event = { id = chaosx.nr3.3 days = 2 }
				}
			}
			NEP = {
				if = {
					limit = {
						exists = yes
						NOT = { tag = ROOT }
					}
					country_event = { id = chaosx.nr3.3 days = 3 }
				}
			}
			TIB = {
				if = {
					limit = {
						exists = yes
						NOT = { tag = ROOT }
					}
					country_event = { id = chaosx.nr3.3 days = 1 }
				}
			}
		}
	}

	option = {
		name = chaosx.nr3.2.b
		ai_chance = { base = 30 }
		set_country_flag = holy_realm_refused_calling
		add_stability = -0.05
	}
}
