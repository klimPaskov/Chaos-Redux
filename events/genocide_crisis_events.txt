# ============================================================================
# CHAOS REDUX - CAMPS AND GENOCIDE CRISIS EVENTS
# ============================================================================
# Overview:
# - Fired by the genocide crisis scripted effects when camp evidence is
#   discovered, condemnation thresholds are crossed, domestic crisis pressure
#   appears, refugee pressure reaches neighbors, or tribunal preparations begin.
# - These events are mechanic-support events, not random Chaos event entries.
# ============================================================================

country_event = {
	id = chaosx_genocide.1
	title = chaosx_genocide.1.t
	desc = chaosx_genocide.1.d
	picture = GFX_report_event_dead_jews
	is_triggered_only = yes

	option = {
		name = chaosx_genocide.1.a
		add_war_support = constant:genocide_political_effect.discoverer_war_support
		add_political_power = constant:genocide_political_effect.discoverer_pp
		if = {
			limit = { has_event_target = genocide_responsible_country }
			add_opinion_modifier = {
				target = event_target:genocide_responsible_country
				modifier = genocide_atrocity_condemned
			}
			set_country_flag = genocide_discovery_publicized
		}
	}
}
