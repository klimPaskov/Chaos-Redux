country_event = {
	id = chaosx.nr1.1
	title = chaosx.nr1.1.t
	desc = chaosx.nr1.1.d
	picture = GFX_report_event_communist_insurgency_start

	is_triggered_only = yes

	trigger = {
		hidden_trigger = { NOT = { is_special_chaos_country = yes } }
	}

	immediate = {
		hidden_effect = {
			if = {
				limit = { NOT = { has_global_flag = communism_spread } }
					set_global_flag = communism_spread
					every_other_country = {
						limit = { hidden_trigger = { NOT = { is_special_chaos_country = yes } } }
						country_event = { id = chaosx.nr1.1 }
					}
				}

				if = {
					limit = {
						hidden_trigger = { NOT = { is_special_chaos_country = yes } }
						NOT = { has_government = communism }
					}
					if = {
						limit = { NOT = { has_idea = communism_spread_idea } }
						add_ideas = communism_spread_idea
					}
					queue_communism_spread_defeat_grace_timeout = yes
					set_temp_variable = { base_value = 0.04 }
				set_temp_variable = { base_add = 0.01 }
				modify_value_based_on_chaos_tier = yes
					add_popularity = {
						ideology = communism
						popularity = var:modified_value
					}
					seed_initial_communist_control_state = yes
					refresh_communism_spread_pressure = yes
					queue_initial_communism_spread_maintenance = yes
				}
		}
	}

	option = {
			trigger = {
				hidden_trigger = { NOT = { is_special_chaos_country = yes } }
				NOT = { has_government = communism }
			}
		name = chaosx.nr1.1.a
		effect_tooltip = {
			add_ideas = communism_spread_idea
			add_popularity = {
				ideology = communism
				popularity = 0.04
			}
		}
	}
	option = {
			trigger = {
				hidden_trigger = { NOT = { is_special_chaos_country = yes } }
				has_government = communism
			}
		name = chaosx.nr1.1.b
	}
}

country_event = {
	id = chaosx.nr1.2
	title = chaosx.nr1.2.t
	desc = chaosx.nr1.2.d
	picture = GFX_report_event_communist_industry_sabotage

	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			if = {
				limit = { NOT = { has_event_target = communist_sabotage_state } }
				select_communist_sabotage_state = yes
			}
		}
	}

	option = {
		name = chaosx.nr1.2.a
		ai_chance = { base = 65 }
		custom_effect_tooltip = chaosx_tt_1_2a
		hidden_effect = {
			event_target:communist_sabotage_state = {
				if = {
					limit = { has_state_flag = communist_control_level_1 }
					damage_building = {
						type = industrial_complex
						damage = 0.20
					}
					add_dynamic_modifier = {
						modifier = communism_post_crackdown_scars
						days = 45
					}
				}
				else_if = {
					limit = { has_state_flag = communist_control_level_2 }
					damage_building = {
						type = industrial_complex
						damage = 0.35
					}
					damage_building = {
						type = arms_factory
						damage = 0.25
					}
				}
				else = {
					damage_building = {
						type = industrial_complex
						damage = 0.55
					}
					damage_building = {
						type = arms_factory
						damage = 0.45
					}
					add_dynamic_modifier = {
						modifier = communism_post_crackdown_scars
						days = 90
					}
				}
			}
		}
	}

	option = {
		name = chaosx.nr1.2.b
		ai_chance = {
			base = 35
			modifier = {
				add = 20
				has_war = yes
			}
			modifier = {
				factor = 0.5
				has_stability < 0.35
			}
		}
		add_command_power = -10
		add_stability = -0.01
		hidden_effect = {
			event_target:communist_sabotage_state = {
				if = {
					limit = { has_state_flag = communist_control_level_2 }
					random_list = {
						55 = { set_communism_state_control_level_1 = yes }
						45 = { }
					}
				}
				else_if = {
					limit = { has_state_flag = communist_control_level_1 }
					random_list = {
						65 = { clear_communism_state_control = yes }
						35 = { }
					}
				}
			}
			refresh_communism_spread_pressure = yes
		}
	}
}
