# Overview:
# - Handles the zombie special-project support and offensive branches.
# - Tracks the creator's zombie profile, resolves it into a final archetype,
#   and spawns custom outbreak countries for strikes, accidents, and tests.
# - Keeps the anti-zombie containment raid idempotent.

apply_zombie_decontamination_state_modifier = {
    if = {
        limit = {
            NOT = { has_dynamic_modifier = { modifier = zombie_decontaminated_state } }
        }
        add_dynamic_modifier = {
            modifier = zombie_decontaminated_state
        }
        force_update_dynamic_modifier = yes
    }
}

initialize_weaponized_zombie_tracking = {
    if = {
        limit = { NOT = { has_country_flag = weaponized_zombies_started } }
        clear_weaponized_zombie_research_step_flags = yes
    }
    if = {
        limit = { NOT = { has_variable = weaponized_zombie_strength } }
        set_variable = { weaponized_zombie_strength = 0 }
    }
    if = {
        limit = { NOT = { has_variable = weaponized_zombie_infectiousness } }
        set_variable = { weaponized_zombie_infectiousness = 0 }
    }
    if = {
        limit = { NOT = { has_variable = weaponized_zombie_speed } }
        set_variable = { weaponized_zombie_speed = 0 }
    }
    if = {
        limit = { NOT = { has_variable = weaponized_zombie_durability } }
        set_variable = { weaponized_zombie_durability = 0 }
    }
    if = {
        limit = { NOT = { has_variable = weaponized_zombie_cure_resistance } }
        set_variable = { weaponized_zombie_cure_resistance = 0 }
    }
    if = {
        limit = { NOT = { has_variable = weaponized_zombie_friendliness } }
        set_variable = { weaponized_zombie_friendliness = 0 }
    }
    if = {
        limit = { NOT = { has_variable = weaponized_zombie_test_subjects } }
        set_variable = { weaponized_zombie_test_subjects = 0 }
    }
    if = {
        limit = { NOT = { has_variable = weaponized_zombie_accident_risk } }
        set_variable = { weaponized_zombie_accident_risk = 0 }
    }
    if = {
        limit = { NOT = { has_variable = weaponized_zombie_cost_score } }
        set_variable = { weaponized_zombie_cost_score = 0 }
    }
    if = {
        limit = { NOT = { has_variable = weaponized_zombie_mutation_extremes } }
        set_variable = { weaponized_zombie_mutation_extremes = 0 }
    }
    if = {
        limit = { NOT = { has_variable = weaponized_zombie_outbreak_stage } }
        set_variable = { weaponized_zombie_outbreak_stage = 0 }
    }
    if = {
        limit = { NOT = { has_variable = weaponized_zombie_creator_cure_days } }
        set_variable = { weaponized_zombie_creator_cure_days = constant:weaponized_zombies.creator_cure_days_low }
    }
    set_country_flag = weaponized_zombies_started
}

    set_country_flag = weaponized_zombies_started
}

remove_weaponized_zombie_completion_ideas = {
	remove_weaponized_zombie_cost_ideas = yes
	if = {
		limit = { has_idea = weaponized_zombie_creator_cure_advantage }
		remove_ideas = weaponized_zombie_creator_cure_advantage
	}
}

complete_weaponized_zombie_project_from_project_output = {
	if = {
		limit = { has_country_flag = weaponized_zombie_console_skip_project_output }
		clr_country_flag = weaponized_zombie_console_skip_project_output
	}
	else = {
		complete_weaponized_zombie_project = yes
	}
}

evaluate_wendigo_world_end_readiness = {
    clear_wendigo_world_end_continent_flags = yes

    set_temp_variable = { wendigo_continent_total_states = 0 }
    set_temp_variable = { wendigo_continent_controlled_states = 0 }
    every_state = {
        limit = { is_on_continent = europe }
        add_to_temp_variable = { wendigo_continent_total_states = 1 }
        if = {
            limit = { is_controlled_by = ROOT }
            add_to_temp_variable = { wendigo_continent_controlled_states = 1 }
        }
    }
    set_temp_variable = { wendigo_continent_missing_states = wendigo_continent_total_states }
    subtract_from_temp_variable = { wendigo_continent_missing_states = wendigo_continent_controlled_states }
    set_temp_variable = { wendigo_continent_control_percent = wendigo_continent_controlled_states }
    multiply_temp_variable = { wendigo_continent_control_percent = constant:weaponized_zombies.wendigo_world_end_percent_scale }
    divide_temp_variable = { wendigo_continent_control_percent = wendigo_continent_total_states }
    if = {
        limit = {
            check_variable = {
                var = wendigo_continent_controlled_states
                value = 1
                compare = greater_than_or_equals
            }
            check_variable = {
                var = wendigo_continent_missing_states
                value = constant:weaponized_zombies.wendigo_world_end_missing_states
                compare = less_than_or_equals
            }
            check_variable = {
                var = wendigo_continent_control_percent
                value = constant:weaponized_zombies.wendigo_world_end_control_percent
                compare = greater_than_or_equals
            }
        }
        set_country_flag = wendigo_world_end_ready
        set_country_flag = wendigo_world_end_continent_europe
    }

    set_temp_variable = { wendigo_continent_total_states = 0 }
    set_temp_variable = { wendigo_continent_controlled_states = 0 }
    every_state = {
        limit = { is_on_continent = north_america }
        add_to_temp_variable = { wendigo_continent_total_states = 1 }
        if = {
            limit = { is_controlled_by = ROOT }
            add_to_temp_variable = { wendigo_continent_controlled_states = 1 }
        }
    }

weaponized_zombie_calculate_world_pressure = {
    set_temp_variable = { weaponized_zombie_disaster_pressure = 0 }

    if = {
        limit = { has_global_flag = { flag = chaos_tier value = 2 } }
        add_to_temp_variable = { weaponized_zombie_disaster_pressure = constant:weaponized_zombies.world_pressure_chaos_tier_2 }
    }
    else_if = {
        limit = { has_global_flag = { flag = chaos_tier value = 3 } }
        add_to_temp_variable = { weaponized_zombie_disaster_pressure = constant:weaponized_zombies.world_pressure_chaos_tier_3 }
    }
    else_if = {
        limit = { has_global_flag = { flag = chaos_tier value = 4 } }
        add_to_temp_variable = { weaponized_zombie_disaster_pressure = constant:weaponized_zombies.world_pressure_chaos_tier_4 }
    }
    else_if = {
        limit = { has_global_flag = { flag = chaos_tier value > 4 } }
        add_to_temp_variable = { weaponized_zombie_disaster_pressure = constant:weaponized_zombies.world_pressure_chaos_tier_5 }
    }

    if = {
        limit = {
            ZZZ = {
                has_country_flag = { flag = zombie_current_tier value = 2 }
            }
        }
        add_to_temp_variable = { weaponized_zombie_disaster_pressure = constant:weaponized_zombies.world_pressure_evolution_tier_2 }
    }
    else_if = {
        limit = {
            ZZZ = {
                has_country_flag = { flag = zombie_current_tier value = 3 }
            }
        }
        add_to_temp_variable = { weaponized_zombie_disaster_pressure = constant:weaponized_zombies.world_pressure_evolution_tier_3 }
    }

    if = {
        limit = { has_country_flag = weaponized_zombie_life_transformed }
        add_to_temp_variable = { weaponized_zombie_disaster_pressure = constant:weaponized_zombies.world_pressure_transformed }
    }
    if = {
        limit = { has_country_flag = weaponized_zombie_forbidden_branch }
        add_to_temp_variable = { weaponized_zombie_disaster_pressure = constant:weaponized_zombies.world_pressure_forbidden }
    }
    if = {
        limit = {
            check_variable = {
                var = weaponized_zombie_mutation_extremes
                value = 4
                compare = greater_than_or_equals
            }
        }
        add_to_temp_variable = { weaponized_zombie_disaster_pressure = constant:weaponized_zombies.world_pressure_mutation_extreme }
    }
}

weaponized_zombie_resolve_base_type = {
    if = {
        limit = { has_country_flag = weaponized_zombie_nature_demonic }
        weaponized_zombie_set_type_demonic = yes
    }
    else_if = {
        limit = { has_country_flag = weaponized_zombie_nature_necrotic }
        weaponized_zombie_set_type_necrotic = yes
    }
    else_if = {
        limit = { has_country_flag = weaponized_zombie_life_dead }
        weaponized_zombie_set_type_undead = yes
    }
    else_if = {
        limit = { has_country_flag = weaponized_zombie_nature_parasitic }
        weaponized_zombie_set_type_parasitic = yes
    }
    else_if = {
        limit = { has_country_flag = weaponized_zombie_nature_rage }
        weaponized_zombie_set_type_rabid = yes
    }
    else = {
        weaponized_zombie_set_type_infected = yes
    }
}

weaponized_zombie_apply_type_pressure = {
    weaponized_zombie_resolve_base_type = yes
    weaponized_zombie_calculate_world_pressure = yes

    if = {
        limit = {
            NOT = { has_country_flag = weaponized_zombie_type_wendigo }
            NOT = { has_country_flag = weaponized_zombie_type_demonic }
            OR = {
                has_country_flag = weaponized_zombie_life_transformed
                AND = {
                    has_country_flag = weaponized_zombie_life_dead
                    has_country_flag = weaponized_zombie_nature_necrotic
                }
            }
            OR = {
                has_country_flag = weaponized_zombie_forbidden_branch
                check_variable = {
                    var = weaponized_zombie_mutation_extremes
                    value = 4
                    compare = greater_than_or_equals

            }
        }
        weaponized_zombie_roll_demonic_override = yes
    }

    if = {
        limit = {
            has_country_flag = weaponized_zombie_life_transformed
            NOT = { has_country_flag = weaponized_zombie_type_wendigo }
            NOT = { has_country_flag = weaponized_zombie_type_demonic }
            NOT = { has_country_flag = weaponized_zombie_type_mutant }
        }
        weaponized_zombie_roll_mutant_override = yes
    }
}

			OR = {
				AND = {
					ROOT = { has_country_flag = weaponized_zombie_friendly_to_humans }
					is_actual_nonhuman_country = yes
				}
				AND = {
					ROOT = { NOT = { has_country_flag = weaponized_zombie_friendly_to_humans } }
					NOT = { original_tag = ZZZ }
					NOT = {
						ROOT = {
							is_in_array = {
                                array = weaponized_zombie_friendly_countries
                                value = PREV
                            }
                        }
                    }
                }
            }
        }
        if = {
            limit = { NOT = { is_subject_of = ROOT } }
            ROOT = {
                declare_war_on = {
                    target = PREV
                    type = annex_everything
                }
            }
        }
        else = {
            ROOT = {
                annex_country = {
                    target = PREV
                    transfer_troops = no
                }
            }
        }
    }
}

apply_weaponized_zombie_strike_consequences = {
	var:actor_country = {
		if = {
			limit = {
				NOT = { has_country_flag = weaponized_zombie_friendly_to_humans }
				NOT = {
					var:ROOT.target_state = {
						is_zombie_controlled_state = yes
                    }
                }
            }
            add_political_power = constant:weaponized_zombies.condemnation_pp_loss_negative
            add_stability = constant:weaponized_zombies.condemnation_stability_loss_negative
            add_war_support = constant:weaponized_zombies.condemnation_warsupport_loss_negative
            set_country_flag = used_bioweapon
            set_country_flag = used_unconventional_weapon
            if = {
                limit = { NOT = { has_country_flag = used_weaponized_zombie_weapon } }
                set_country_flag = used_weaponized_zombie_weapon
            }
            every_other_country = {
                limit = {
                    exists = yes
                    has_government = democratic
                    NOT = { tag = ROOT }
                }
                add_opinion_modifier = {
                    target = PREV
                    modifier = strategic_bioweapon_user
                }
            }
		}
		else_if = {
			limit = {
				has_country_flag = weaponized_zombie_friendly_to_humans
				var:ROOT.target_state = {
					is_zombie_controlled_state = yes
				}
            }
            set_country_flag = achievement_fight_fire_with_fire
        }
    }
}

            save_global_event_target_as = wendigo_super_event_creator
        }
    }

        limit = { has_event_target = weaponized_zombie_cure_target_creator }

            save_global_event_target_as = wendigo_super_event_creator

    }

        save_global_event_target_as = wendigo_super_event_creator
    }

    every_country = {
        limit = { is_ai = no }

    }
}
