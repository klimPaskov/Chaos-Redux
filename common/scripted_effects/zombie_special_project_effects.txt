# Overview:
# - Handles the zombie special-project support and offensive branches.
# - Tracks the creator's zombie profile, resolves it into a final archetype,
#   and spawns custom outbreak countries for strikes, accidents, and tests.
# - Keeps the anti-zombie containment raid idempotent.

apply_zombie_decontamination_state_modifier = {
	if = {
		limit = {
			NOT = { has_dynamic_modifier = { modifier = zombie_decontaminated_state } }
		}
		add_dynamic_modifier = {
			modifier = zombie_decontaminated_state
		}
		force_update_dynamic_modifier = yes
	}
}

initialize_weaponized_zombie_tracking = {
	if = {
		limit = { NOT = { has_country_flag = weaponized_zombies_started } }
		clear_weaponized_zombie_research_step_flags = yes
	}
	if = {
		limit = { NOT = { has_variable = weaponized_zombie_strength } }
		set_variable = { weaponized_zombie_strength = 0 }
	}
	if = {
		limit = { NOT = { has_variable = weaponized_zombie_infectiousness } }
		set_variable = { weaponized_zombie_infectiousness = 0 }
	}
	if = {

	}
	if = {

		set_variable = { weaponized_zombie_durability = 0 }
	}
	if = {

		set_variable = { weaponized_zombie_cure_resistance = 0 }
	}
	if = {
		limit = { NOT = { has_variable = weaponized_zombie_friendliness } }
		set_variable = { weaponized_zombie_friendliness = 0 }
	}
	if = {
		limit = { NOT = { has_variable = weaponized_zombie_test_subjects } }
		set_variable = { weaponized_zombie_test_subjects = 0 }
	}
	if = {
		limit = { NOT = { has_variable = weaponized_zombie_accident_risk } }
		set_variable = { weaponized_zombie_accident_risk = 0 }
	}
	if = {
		limit = { NOT = { has_variable = weaponized_zombie_cost_score } }
		set_variable = { weaponized_zombie_cost_score = 0 }
	}
	if = {
		limit = { NOT = { has_variable = weaponized_zombie_mutation_extremes } }
		set_variable = { weaponized_zombie_mutation_extremes = 0 }
	}
	if = {
		limit = { NOT = { has_variable = weaponized_zombie_outbreak_stage } }
		set_variable = { weaponized_zombie_outbreak_stage = 0 }
	}
	if = {
		limit = { NOT = { has_variable = weaponized_zombie_creator_cure_days } }
		set_variable = { weaponized_zombie_creator_cure_days = constant:weaponized_zombies.creator_cure_days_low }
	}

}

			limit = { is_controlled_by = ROOT }

weaponized_zombie_calculate_world_pressure = {
	set_temp_variable = { weaponized_zombie_disaster_pressure = 0 }

	if = {
		limit = { has_global_flag = { flag = chaos_tier value = 2 } }
		add_to_temp_variable = { weaponized_zombie_disaster_pressure = constant:weaponized_zombies.world_pressure_chaos_tier_2 }
	}
	else_if = {
		limit = { has_global_flag = { flag = chaos_tier value = 3 } }
		add_to_temp_variable = { weaponized_zombie_disaster_pressure = constant:weaponized_zombies.world_pressure_chaos_tier_3 }
	}
	else_if = {
		limit = { has_global_flag = { flag = chaos_tier value = 4 } }
		add_to_temp_variable = { weaponized_zombie_disaster_pressure = constant:weaponized_zombies.world_pressure_chaos_tier_4 }
	}
	else_if = {
		limit = { has_global_flag = { flag = chaos_tier value > 4 } }
		add_to_temp_variable = { weaponized_zombie_disaster_pressure = constant:weaponized_zombies.world_pressure_chaos_tier_5 }
	}

	if = {
		limit = {
			ZZZ = {
				has_country_flag = { flag = zombie_current_tier value = 2 }
			}
		}
		add_to_temp_variable = { weaponized_zombie_disaster_pressure = constant:weaponized_zombies.world_pressure_evolution_tier_2 }
	}
	else_if = {
		limit = {
			ZZZ = {
				has_country_flag = { flag = zombie_current_tier value = 3 }
			}
		}
		add_to_temp_variable = { weaponized_zombie_disaster_pressure = constant:weaponized_zombies.world_pressure_evolution_tier_3 }
	}

	if = {
		limit = { has_country_flag = weaponized_zombie_life_transformed }
		add_to_temp_variable = { weaponized_zombie_disaster_pressure = constant:weaponized_zombies.world_pressure_transformed }
	}
	if = {
		limit = { has_country_flag = weaponized_zombie_forbidden_branch }
		add_to_temp_variable = { weaponized_zombie_disaster_pressure = constant:weaponized_zombies.world_pressure_forbidden }
	}
	if = {
		limit = {
			check_variable = {
				var = weaponized_zombie_mutation_extremes
				value = 4
				compare = greater_than_or_equals
			}
		}
		add_to_temp_variable = { weaponized_zombie_disaster_pressure = constant:weaponized_zombies.world_pressure_mutation_extreme }
	}
}

weaponized_zombie_resolve_base_type = {
	if = {
		limit = { has_country_flag = weaponized_zombie_nature_demonic }
		weaponized_zombie_set_type_demonic = yes
	}
	else_if = {
		limit = { has_country_flag = weaponized_zombie_nature_necrotic }
		weaponized_zombie_set_type_necrotic = yes
	}
	else_if = {
		limit = { has_country_flag = weaponized_zombie_life_dead }
		weaponized_zombie_set_type_undead = yes
	}
	else_if = {
		limit = { has_country_flag = weaponized_zombie_nature_parasitic }
		weaponized_zombie_set_type_parasitic = yes
	}
	else_if = {
		limit = { has_country_flag = weaponized_zombie_nature_rage }
		weaponized_zombie_set_type_rabid = yes
	}
	else = {
		weaponized_zombie_set_type_infected = yes
	}
}

	if = {
		limit = {
			check_variable = {

			}
		}

resolve_weaponized_zombie_profile = {
	initialize_weaponized_zombie_tracking = yes
	clear_weaponized_zombie_resolution_flags = yes
	set_variable = { weaponized_zombie_cost_score = 0 }
	set_variable = { weaponized_zombie_outbreak_stage = 0 }

	add_to_variable = { weaponized_zombie_cost_score = weaponized_zombie_strength }
	add_to_variable = { weaponized_zombie_cost_score = weaponized_zombie_infectiousness }
	add_to_variable = { weaponized_zombie_cost_score = weaponized_zombie_speed }
	add_to_variable = { weaponized_zombie_cost_score = weaponized_zombie_durability }
	add_to_variable = { weaponized_zombie_cost_score = weaponized_zombie_cure_resistance }
	add_to_variable = { weaponized_zombie_cost_score = weaponized_zombie_friendliness }

	if = {
		limit = { has_country_flag = weaponized_zombie_nature_parasitic }
		add_to_variable = { weaponized_zombie_cost_score = 1 }
	}
	if = {
		limit = { has_country_flag = weaponized_zombie_nature_demonic }
		add_to_variable = { weaponized_zombie_cost_score = 3 }
	}
	if = {
		limit = { has_country_flag = weaponized_zombie_forbidden_branch }
		add_to_variable = { weaponized_zombie_cost_score = 2 }
	}
	if = {
		limit = { has_country_flag = weaponized_zombie_field_tested }
		add_to_variable = { weaponized_zombie_cost_score = 1 }
	}

	set_temp_variable = { weaponized_zombie_stage_score = weaponized_zombie_strength }
	add_to_temp_variable = { weaponized_zombie_stage_score = weaponized_zombie_infectiousness }
	add_to_temp_variable = { weaponized_zombie_stage_score = weaponized_zombie_speed }
	add_to_temp_variable = { weaponized_zombie_stage_score = weaponized_zombie_durability }

	if = {
		limit = { check_variable = { var = weaponized_zombie_stage_score value > 10 } }
		set_variable = { weaponized_zombie_outbreak_stage = 3 }
	}
	else_if = {
		limit = { check_variable = { var = weaponized_zombie_stage_score value > 8 } }
		set_variable = { weaponized_zombie_outbreak_stage = 2 }
	}
	else_if = {
		limit = { check_variable = { var = weaponized_zombie_stage_score value > 6 } }
		set_variable = { weaponized_zombie_outbreak_stage = 1 }
	}

	weaponized_zombie_apply_type_pressure = yes

	if = {
		limit = { has_country_flag = weaponized_zombie_type_wendigo }
		set_country_flag = weaponized_zombie_archetype_wendigo
	}
	else_if = {
		limit = {
			check_variable = { var = weaponized_zombie_friendliness value > 2 }
			OR = {
				has_country_flag = weaponized_zombie_life_fragmented
				has_country_flag = weaponized_zombie_life_alive
				has_country_flag = weaponized_zombie_life_transformed
			}
			NOT = { has_country_flag = weaponized_zombie_nature_rage }
			NOT = { has_country_flag = weaponized_zombie_nature_demonic }
			check_variable = { var = weaponized_zombie_strength value < 3 }
			check_variable = { var = weaponized_zombie_speed value < 3 }
			check_variable = { var = weaponized_zombie_durability value < 3 }
		}
		set_country_flag = weaponized_zombie_archetype_semi_sapient
	}
	else_if = {
		limit = {
			check_variable = { var = weaponized_zombie_friendliness value > 1 }
			OR = {
				has_country_flag = weaponized_zombie_life_fragmented
				has_country_flag = weaponized_zombie_life_transformed
			}
			OR = {
				has_country_flag = weaponized_zombie_nature_neurobiological
				has_country_flag = weaponized_zombie_nature_parasitic
			}
			check_variable = { var = weaponized_zombie_speed value < 3 }
			check_variable = { var = weaponized_zombie_strength value < 3 }
		}
		set_country_flag = weaponized_zombie_archetype_purifier
	}
	else_if = {
		limit = {
			check_variable = { var = weaponized_zombie_friendliness value > 2 }
			OR = {
				has_country_flag = weaponized_zombie_life_fragmented
				has_country_flag = weaponized_zombie_life_alive
				has_country_flag = weaponized_zombie_life_transformed
			}
			OR = {
				has_country_flag = weaponized_zombie_nature_neurobiological
				has_country_flag = weaponized_zombie_nature_parasitic
			}
		}
		set_country_flag = weaponized_zombie_archetype_controlled_loyalty
	}
	else_if = {
		limit = {
			check_variable = { var = weaponized_zombie_friendliness value > 1 }
			OR = {
				has_country_flag = weaponized_zombie_nature_parasitic
				has_country_flag = weaponized_zombie_life_fragmented
			}
		}
		set_country_flag = weaponized_zombie_archetype_parasitic_directed
	}
	else_if = {
		limit = {
			check_variable = { var = weaponized_zombie_friendliness value > 1 }
			OR = {
				has_country_flag = weaponized_zombie_nature_rage
				has_country_flag = weaponized_zombie_nature_demonic
			}
		}
		set_country_flag = weaponized_zombie_archetype_creator_betrayer
	}
	else_if = {
		limit = {
			OR = {
				has_country_flag = weaponized_zombie_nature_necrotic
				has_country_flag = weaponized_zombie_nature_demonic
			}
			has_country_flag = weaponized_zombie_life_dead
			check_variable = { var = weaponized_zombie_durability value > 1 }
			check_variable = { var = weaponized_zombie_speed value < 3 }
		}
		set_country_flag = weaponized_zombie_archetype_necrotic_siege
	}
	else_if = {
		limit = {
			has_country_flag = weaponized_zombie_life_dead
			NOT = { has_country_flag = weaponized_zombie_nature_parasitic }
			check_variable = { var = weaponized_zombie_mutation_extremes value < 4 }
		}
		set_country_flag = weaponized_zombie_archetype_canonical_convergence
	}
	else = {
		set_country_flag = weaponized_zombie_archetype_rabid_horde
	}

	if = {
		limit = {
			NOT = { has_country_flag = weaponized_zombie_archetype_canonical_convergence }
		}
		set_country_flag = weaponized_zombie_attack_other_zombies
	}
	if = {
		limit = {
			OR = {
				has_country_flag = weaponized_zombie_archetype_controlled_loyalty
				has_country_flag = weaponized_zombie_archetype_creator_betrayer
				has_country_flag = weaponized_zombie_archetype_parasitic_directed
			}
		}
		set_country_flag = weaponized_zombie_friendly_to_creator
	}
	if = {
		limit = {
			OR = {
				has_country_flag = weaponized_zombie_archetype_purifier
				has_country_flag = weaponized_zombie_archetype_semi_sapient
			}
		}
		set_country_flag = weaponized_zombie_friendly_to_humans
		set_country_flag = weaponized_zombie_targets_zombies_only
	}

	if = {
		limit = { check_variable = { var = weaponized_zombie_cost_score value < constant:weaponized_zombies.profile_cost_tier_low } }
		set_variable = { weaponized_zombie_creator_cure_days = constant:weaponized_zombies.creator_cure_days_low }
	}
	else_if = {
		limit = { check_variable = { var = weaponized_zombie_cost_score value < constant:weaponized_zombies.profile_cost_tier_medium } }
		set_variable = { weaponized_zombie_creator_cure_days = constant:weaponized_zombies.creator_cure_days_medium }
	}
	else_if = {
		limit = { check_variable = { var = weaponized_zombie_cost_score value < constant:weaponized_zombies.profile_cost_tier_high } }
		set_variable = { weaponized_zombie_creator_cure_days = constant:weaponized_zombies.creator_cure_days_high }
	}
	else = {
		set_variable = { weaponized_zombie_creator_cure_days = constant:weaponized_zombies.creator_cure_days_extreme }
	}

	if = {
		limit = {
			check_variable = {
				var = weaponized_zombie_strength
				value = constant:weaponized_zombies.profile_hardened_strength_threshold
				compare = greater_than_or_equals
			}
			check_variable = {
				var = weaponized_zombie_durability
				value = constant:weaponized_zombies.profile_hardened_durability_threshold
				compare = greater_than_or_equals
			}
		}
		set_country_flag = weaponized_zombie_profile_hardened
	}
