# Communism spread state-control effects.
# COUNTRY SCOPE unless a block explicitly documents STATE SCOPE.
# The system is self-scheduled from chaosx.nr1.20 instead of using another global monthly country sweep.

queue_communism_spread_maintenance = {
	if = {
		limit = {
			has_global_flag = communism_spread
			hidden_trigger = { NOT = { is_special_chaos_country = yes } }
			has_idea = communism_spread_idea
			NOT = { has_country_flag = communism_spread_maintenance_queued }
			NOT = { has_government = communism }
		}
		set_country_flag = communism_spread_maintenance_queued
		country_event = {
			id = chaosx.nr1.20
			days = constant:communism_spread_timing.maintenance_min_days
			random_days = 30
		}
	}
}

clear_communism_state_control_modifier = {
	# STATE SCOPE
	if = {
		limit = { has_dynamic_modifier = communism_control_level_1 }
		remove_dynamic_modifier = { modifier = communism_control_level_1 }
	}
	if = {
		limit = { has_dynamic_modifier = communism_control_level_2 }
		remove_dynamic_modifier = { modifier = communism_control_level_2 }
	}
	if = {
		limit = { has_dynamic_modifier = communism_control_level_3 }
		remove_dynamic_modifier = { modifier = communism_control_level_3 }
	}
}

clear_communism_state_control = {
	# STATE SCOPE
	if = {
		limit = { has_state_flag = communist_added_demilitarized_zone }
		set_demilitarized_zone = no
	}
	clr_state_flag = communist_control_level_1
	clr_state_flag = communist_control_level_2
	clr_state_flag = communist_control_level_3
	clr_state_flag = communist_controlled_state
	clr_state_flag = communist_added_demilitarized_zone
	clr_state_flag = communist_worker_ritual_state
	clr_state_flag = communist_world_revolution_whisper_state
	clr_state_flag = communist_surprise_breakaway_state
	clear_communism_state_control_modifier = yes
}

set_communism_state_control_level_1 = {
	# STATE SCOPE
	clear_communism_state_control = yes
	set_state_flag = communist_controlled_state
	set_state_flag = communist_control_level_1
	add_dynamic_modifier = { modifier = communism_control_level_1 }
}

set_communism_state_control_level_2 = {
	# STATE SCOPE
	clear_communism_state_control = yes
	set_state_flag = communist_controlled_state
	set_state_flag = communist_control_level_2
	add_dynamic_modifier = { modifier = communism_control_level_2 }
}

set_communism_state_control_level_3 = {
	# STATE SCOPE
	clear_communism_state_control = yes
	set_state_flag = communist_controlled_state
	set_state_flag = communist_control_level_3
	if = {
		limit = { NOT = { is_demilitarized_zone = yes } }
		set_state_flag = communist_added_demilitarized_zone
	}
	set_demilitarized_zone = yes
	add_dynamic_modifier = { modifier = communism_control_level_3 }
}

refresh_communism_spread_counts = {
	set_variable = { communism_controlled_states = 0 }
	set_variable = { communism_control_level_1_states = 0 }
	set_variable = { communism_control_level_2_states = 0 }
	set_variable = { communism_control_level_3_states = 0 }
	set_variable = { communism_emergency_control_threshold = num_owned_states }
	multiply_variable = { communism_emergency_control_threshold = 0.5 }
	# Trigger syntax has no inclusive greater-than check, so 50% must pass a > comparison.
	subtract_from_variable = { communism_emergency_control_threshold = 0.1 }

	every_owned_state = {
		limit = { has_state_flag = communist_controlled_state }
		ROOT = { add_to_variable = { communism_controlled_states = 1 } }
		if = {
			limit = { has_state_flag = communist_control_level_1 }
			ROOT = { add_to_variable = { communism_control_level_1_states = 1 } }
		}
		else_if = {
			limit = { has_state_flag = communist_control_level_2 }
			ROOT = { add_to_variable = { communism_control_level_2_states = 1 } }
		}
		else_if = {
			limit = { has_state_flag = communist_control_level_3 }
			ROOT = { add_to_variable = { communism_control_level_3_states = 1 } }
		}
	}
}

refresh_communism_spread_pressure = {
	refresh_communism_spread_counts = yes
	set_variable = { communism_control_daily_drift = communism_controlled_states }
	multiply_variable = { communism_control_daily_drift = constant:communism_spread_drift.per_controlled_state }
	clamp_variable = {
		var = communism_control_daily_drift
		max = constant:communism_spread_drift.max_from_states
	}

	if = {
		limit = {
			check_variable = { communism_controlled_states > 0 }
			NOT = { has_dynamic_modifier = communism_state_control_pressure }
		}
		add_dynamic_modifier = { modifier = communism_state_control_pressure }
	}
	else_if = {
		limit = {
			check_variable = { communism_controlled_states = 0 }
			has_dynamic_modifier = communism_state_control_pressure
		}
		remove_dynamic_modifier = { modifier = communism_state_control_pressure }
	}
	refresh_communism_spread_dashboard_values = yes
	force_update_dynamic_modifier = yes
}

can_communism_spread_take_new_state = {
	set_temp_variable = { communism_spread_can_take_state = 0 }
	if = {
		limit = {
			communism > constant:communism_spread_state_control.min_support_for_state_spread
			NOT = { has_country_flag = communism_state_spread_cooldown }
			any_owned_state = {
				impassable = no
				NOT = { is_capital = yes }
				NOT = { has_state_flag = communist_controlled_state }
			}
		}
		refresh_communism_spread_counts = yes
		set_temp_variable = { communism_spread_soft_ceiling = party_popularity@communism }
		multiply_temp_variable = { communism_spread_soft_ceiling = constant:communism_spread_state_control.soft_ceiling_ratio }
		multiply_temp_variable = { communism_spread_soft_ceiling = num_owned_states }
		round_temp_variable = communism_spread_soft_ceiling
		if = {
			limit = { check_variable = { communism_spread_soft_ceiling < 1 } }
			set_temp_variable = { communism_spread_soft_ceiling = 1 }
		}
		if = {
			limit = { check_variable = { communism_controlled_states < communism_spread_soft_ceiling } }
			set_temp_variable = { communism_spread_can_take_state = 1 }
		}
	}
}

select_new_communist_control_state = {
	# Priority tiers keep spread strategic while preserving a valid target pool for small countries.
	if = {
		limit = {
			any_owned_state = {
				impassable = no
				NOT = { is_capital = yes }
				NOT = { has_state_flag = communist_controlled_state }
				any_neighbor_state = { has_state_flag = communist_controlled_state }
				is_communist_spread_priority_state = yes
			}
		}
		random_owned_state = {
			limit = {
				impassable = no
				NOT = { is_capital = yes }
				NOT = { has_state_flag = communist_controlled_state }
				any_neighbor_state = { has_state_flag = communist_controlled_state }
				is_communist_spread_priority_state = yes
			}
			set_communism_state_control_level_1 = yes
			save_event_target_as = communist_target_state
		}
	}
	else_if = {
		limit = {
			any_owned_state = {
				impassable = no
				NOT = { is_capital = yes }
				NOT = { has_state_flag = communist_controlled_state }
				any_neighbor_state = { has_state_flag = communist_controlled_state }
			}
		}
		random_owned_state = {
			limit = {
				impassable = no
				NOT = { is_capital = yes }
				NOT = { has_state_flag = communist_controlled_state }
				any_neighbor_state = { has_state_flag = communist_controlled_state }
			}
			set_communism_state_control_level_1 = yes
			save_event_target_as = communist_target_state
		}
	}
	else_if = {
		limit = {
			any_owned_state = {
				impassable = no
				NOT = { is_capital = yes }
				NOT = { has_state_flag = communist_controlled_state }
				is_communist_spread_priority_state = yes
			}
		}
		random_owned_state = {
			limit = {
				impassable = no
				NOT = { is_capital = yes }
				NOT = { has_state_flag = communist_controlled_state }
				is_communist_spread_priority_state = yes
			}
			set_communism_state_control_level_1 = yes
			save_event_target_as = communist_target_state
		}
	}
	else = {
		random_owned_state = {
			limit = {
				impassable = no
				NOT = { is_capital = yes }
				NOT = { has_state_flag = communist_controlled_state }
			}
			set_communism_state_control_level_1 = yes
			save_event_target_as = communist_target_state
		}
	}
	queue_communism_state_spread_cooldown = yes
	refresh_communism_spread_pressure = yes
}

seed_initial_communist_control_state = {
	# COUNTRY SCOPE. The root event must immediately create a visible state-control problem.
	refresh_communism_spread_counts = yes
	if = {
		limit = {
			check_variable = { communism_controlled_states = 0 }
			any_owned_state = {
				impassable = no
				NOT = { has_state_flag = communist_controlled_state }
			}
		}
		if = {
			limit = {
				any_owned_state = {
					impassable = no
					NOT = { is_capital = yes }
					NOT = { has_state_flag = communist_controlled_state }
				}
			}
			select_new_communist_control_state = yes
		}
		else = {
			random_owned_state = {
				limit = {
					impassable = no
					NOT = { has_state_flag = communist_controlled_state }
				}
				set_communism_state_control_level_1 = yes
				save_event_target_as = communist_target_state
			}
			queue_communism_state_spread_cooldown = yes
			refresh_communism_spread_pressure = yes
		}
		country_event = { id = chaosx.nr1.4 days = 1 }
	}
}

try_communism_state_spread = {
	can_communism_spread_take_new_state = yes
	if = {
		limit = { check_variable = { communism_spread_can_take_state = 1 } }
		set_temp_variable = { communism_state_spread_chance = 5 }
		if = {
			limit = { communism > constant:communism_spread_state_control.medium_support }
			add_to_temp_variable = { communism_state_spread_chance = 10 }
		}
		if = {
			limit = { communism > constant:communism_spread_state_control.high_support }
			add_to_temp_variable = { communism_state_spread_chance = 15 }
		}
		if = {
			limit = { communism > constant:communism_spread_state_control.very_high_support }
			add_to_temp_variable = { communism_state_spread_chance = 15 }
		}
		if = {
			limit = { has_stability < constant:communism_spread_state_control.low_stability_threshold }
			add_to_temp_variable = { communism_state_spread_chance = 15 }
		}
		if = {
			limit = { has_war = yes }
			add_to_temp_variable = { communism_state_spread_chance = 10 }
		}
		set_temp_variable = { communism_state_no_spread_chance = 100 }
		subtract_from_temp_variable = { communism_state_no_spread_chance = communism_state_spread_chance }
		random_list = {
			var:communism_state_spread_chance = {
				select_new_communist_control_state = yes
				country_event = { id = chaosx.nr1.4 days = 1 }
			}
			var:communism_state_no_spread_chance = { }
		}
	}
}

try_communism_state_escalation = {
	if = {
		limit = {
			NOT = { has_country_flag = communism_state_escalation_cooldown }
			any_owned_state = {
				has_state_flag = communist_controlled_state
				NOT = { has_state_flag = communist_control_level_3 }
			}
		}
		set_temp_variable = { communism_state_escalation_chance = 15 }
		if = {
			limit = { communism > constant:communism_spread_state_control.high_support }
			add_to_temp_variable = { communism_state_escalation_chance = 15 }
		}
		if = {
			limit = { has_stability < constant:communism_spread_state_control.low_stability_threshold }
			add_to_temp_variable = { communism_state_escalation_chance = 10 }
		}
		if = {
			limit = { has_war = yes }
			add_to_temp_variable = { communism_state_escalation_chance = 10 }
		}
		set_temp_variable = { communism_state_no_escalation_chance = 100 }
		subtract_from_temp_variable = { communism_state_no_escalation_chance = communism_state_escalation_chance }
		random_list = {
			var:communism_state_escalation_chance = {
				if = {
					limit = {
						any_owned_state = {
							has_state_flag = communist_control_level_2
						}
					}
					random_owned_state = {
						limit = { has_state_flag = communist_control_level_2 }
						set_communism_state_control_level_3 = yes
						save_event_target_as = communist_target_state
					}
				}
				else = {
					random_owned_state = {
						limit = { has_state_flag = communist_control_level_1 }
						set_communism_state_control_level_2 = yes
						save_event_target_as = communist_target_state
					}
				}
				queue_communism_state_escalation_cooldown = yes
				refresh_communism_spread_pressure = yes
				country_event = { id = chaosx.nr1.5 days = 1 }
			}
			var:communism_state_no_escalation_chance = { }
		}
	}
}

	if = {
		limit = {
			any_owned_state = {
				has_state_flag = communist_controlled_state
				OR = {
					arms_factory > 0
					industrial_complex > 0
					infrastructure > 1
				}
			}
		}
		random_owned_state = {
			limit = {
				has_state_flag = communist_controlled_state
				OR = {
					arms_factory > 0
					industrial_complex > 0
					infrastructure > 1
				}
			}
			save_event_target_as = communist_sabotage_state
		}
	}
	else = {
		random_owned_state = {
			limit = { has_state_flag = communist_controlled_state }
			save_event_target_as = communist_sabotage_state
		}
	}
}

try_communism_sabotage_event = {
	if = {
		limit = {
			NOT = { has_country_flag = communism_sabotage_cooldown }
			any_owned_state = { has_state_flag = communist_controlled_state }
		}
		set_temp_variable = { communism_sabotage_chance = 12 }
		if = {
			limit = { communism > constant:communism_spread_state_control.high_support }
			add_to_temp_variable = { communism_sabotage_chance = 10 }
		}
		if = {
			limit = { check_variable = { communism_control_level_2_states > 0 } }
			add_to_temp_variable = { communism_sabotage_chance = 8 }
		}
		if = {
			limit = { check_variable = { communism_control_level_3_states > 0 } }
			add_to_temp_variable = { communism_sabotage_chance = 12 }
		}
		set_temp_variable = { communism_no_sabotage_chance = 100 }
		subtract_from_temp_variable = { communism_no_sabotage_chance = communism_sabotage_chance }
		random_list = {
			var:communism_sabotage_chance = {
				select_communist_sabotage_state = yes
				queue_communism_sabotage_cooldown = yes
				country_event = { id = chaosx.nr1.2 days = 1 }
			}
			var:communism_no_sabotage_chance = { }
		}
	}
}

mark_communist_uprising_conversion_states = {
	every_owned_state = {
		limit = {
			NOT = { is_capital = yes }
			has_state_flag = communist_control_level_1
			NOT = { has_state_flag = communist_uprising_convert }
		}

calculate_communist_rebel_state_unit_count = {
	# STATE SCOPE. Output: temp variable communist_rebel_units_to_spawn.
	set_temp_variable = { communist_rebel_units_to_spawn = constant:communism_spread_rebels.level_1_units }
	if = {
		limit = { has_state_flag = communist_control_level_2 }
		set_temp_variable = { communist_rebel_units_to_spawn = constant:communism_spread_rebels.level_2_units }
	}
	else_if = {
		limit = { has_state_flag = communist_control_level_3 }
		set_temp_variable = { communist_rebel_units_to_spawn = constant:communism_spread_rebels.level_3_units }
	}
	if = { limit = { industrial_complex > 1 } add_to_temp_variable = { communist_rebel_units_to_spawn = constant:communism_spread_rebels.industrial_bonus_units } }
	if = { limit = { arms_factory > 0 } add_to_temp_variable = { communist_rebel_units_to_spawn = constant:communism_spread_rebels.arms_factory_bonus_units } }
	if = { limit = { state_population_k > 1000 } add_to_temp_variable = { communist_rebel_units_to_spawn = 1 } }
	if = { limit = { is_communist_spread_priority_state = yes } add_to_temp_variable = { communist_rebel_units_to_spawn = 1 } }
	if = { limit = { ROOT = { is_major = yes } } add_to_temp_variable = { communist_rebel_units_to_spawn = constant:communism_spread_rebels.major_country_bonus_units } }
	if = { limit = { ROOT = { communism > constant:communism_spread_state_control.high_support } } add_to_temp_variable = { communist_rebel_units_to_spawn = 1 } }
	if = { limit = { ROOT = { has_war = yes } } add_to_temp_variable = { communist_rebel_units_to_spawn = 1 } }
	if = { limit = { ROOT = { has_manpower > 100000 } } add_to_temp_variable = { communist_rebel_units_to_spawn = 1 } }
	if = { limit = { ROOT = { num_divisions > 60 } } add_to_temp_variable = { communist_rebel_units_to_spawn = 1 } }
	if = { limit = { has_state_flag = communist_worker_ritual_state } add_to_temp_variable = { communist_rebel_units_to_spawn = 1 } }
	if = { limit = { has_state_flag = communist_world_revolution_whisper_state } add_to_temp_variable = { communist_rebel_units_to_spawn = 1 } }
	if = { limit = { has_state_flag = communist_surprise_breakaway_state } add_to_temp_variable = { communist_rebel_units_to_spawn = 1 } }
	round_temp_variable = communist_rebel_units_to_spawn
}

spawn_communist_rebel_units_in_state = {
	# STATE SCOPE. Called while the dynamic rebel country is PREV.PREV.
	calculate_communist_rebel_state_unit_count = yes
	if = {
		limit = { has_state_flag = communist_control_level_3 }
		spawn_dynamic_communist_rebel_unit_from_state = yes
	}
	set_temp_variable = { communist_rebel_units_spawned = 0 }
	while_loop_effect = {
		limit = { check_variable = { communist_rebel_units_spawned < communist_rebel_units_to_spawn } }
		add_to_temp_variable = { communist_rebel_units_spawned = 1 }
		spawn_dynamic_communist_rebel_unit_from_state = yes
	}
}

create_communist_emergency_uprising = {
	save_event_target_as = communist_original_country
	mark_communist_uprising_conversion_states = yes
	if = {
		limit = {
			any_owned_state = { has_state_flag = communist_uprising_convert }
		}
		create_dynamic_country = {
			original_tag = ROOT
			set_politics = {
				ruling_party = communism
				elections_allowed = no
			}
			create_country_leader = {
				name = "Revolutionary Committee"
				picture = "GFX_portrait_communist_rebels"
				ideology = marxism
			}
			set_country_flag = communist_rebel_state
			set_country_flag = communist_state_control_rebel
			create_communist_rebel_templates = yes
			set_temp_variable = { communist_rebel_manpower_gain = constant:communism_spread_rebels.base_manpower }
			ROOT = {
				refresh_communism_spread_counts = yes
				set_temp_variable = { communist_rebel_manpower_from_states = communism_controlled_states }
				multiply_temp_variable = { communist_rebel_manpower_from_states = constant:communism_spread_rebels.manpower_per_converted_state }
				set_temp_variable = { communist_rebel_manpower_from_support = party_popularity@communism }
				multiply_temp_variable = { communist_rebel_manpower_from_support = constant:communism_spread_rebels.manpower_support_scale }
			}
			add_to_temp_variable = { communist_rebel_manpower_gain = communist_rebel_manpower_from_states }
			add_to_temp_variable = { communist_rebel_manpower_gain = communist_rebel_manpower_from_support }
			round_temp_variable = communist_rebel_manpower_gain
			add_manpower = communist_rebel_manpower_gain
			ROOT = {
				every_owned_state = {
					limit = { has_state_flag = communist_uprising_convert }
					transfer_state_to = PREV.PREV
					spawn_communist_rebel_units_in_state = yes
					clear_communism_state_control = yes
				}
			}
			declare_war_on = {
				target = event_target:communist_original_country
				type = annex_everything
			}
		}
		set_country_flag = fighting_communist_rebels
		add_stability = -0.06
		add_war_support = 0.05
	}
	else = {
		resolve_communist_emergency_without_uprising = yes
	}
	clear_communist_uprising_conversion_flags = yes
	refresh_communism_spread_pressure = yes
}

maybe_fire_revolutionary_war_warning = {
	refresh_communism_spread_dashboard_values = yes
	if = {
		limit = {
			NOT = { has_country_flag = communism_revolutionary_war_cooldown }
			OR = {
				check_variable = { communism_revolutionary_threat > 99 }
				check_variable = { communism_control_level_3_states > 1 }
				AND = {
					check_variable = { communism_controlled_states > 0 }
					check_variable = { num_owned_states > 0 }
				}
			}
		}
		set_temp_variable = { communist_control_ratio_check = communism_controlled_states }
		divide_temp_variable = { communist_control_ratio_check = num_owned_states }
		if = {
			limit = {
				OR = {
					check_variable = { communism_revolutionary_threat > 99 }
					check_variable = { communist_control_ratio_check > 0.49 }
					check_variable = { communism_control_level_3_states > 1 }
				}
			}
			queue_communism_revolutionary_war_cooldown = yes
			country_event = { id = chaosx.nr1.8 days = 1 }
		}
	}
}

try_communism_evolution_progression = {
	refresh_communism_spread_counts = yes
	apply_global_communism_evolution_flags_to_country = yes
	set_temp_variable = { communism_evolution_unstable_threshold = constant:communism_spread_evolution.unstable_min_controlled_states }
	subtract_from_temp_variable = { communism_evolution_unstable_threshold = 1 }
	set_temp_variable = { communism_evolution_ritual_threshold = constant:communism_spread_evolution.dark_rituals_min_level_2_states }
	subtract_from_temp_variable = { communism_evolution_ritual_threshold = 1 }
	set_temp_variable = { communism_evolution_whisper_threshold = constant:communism_spread_evolution.world_revolution_whispers_min_level_3_states }
	subtract_from_temp_variable = { communism_evolution_whisper_threshold = 1 }

		limit = {
			NOT = { has_global_flag = communism_evolution_progression_cooldown }
			has_communism_evolution_tier_1_unlocked = yes
			NOT = { has_global_flag = communism_global_evolution_unstable_activity }
			check_variable = { communism_controlled_states > communism_evolution_unstable_threshold }
		}
		queue_communism_evolution_progression_cooldown = yes
		random_list = {
			constant:communism_spread_evolution.progression_roll_chance = { record_communism_spread_evolution_stage_1 = yes }
			constant:communism_spread_evolution.progression_no_change_chance = { }
		}
	}
	if = {
		limit = {
			NOT = { has_global_flag = communism_evolution_progression_cooldown }
			has_communism_evolution_tier_2_unlocked = yes
			has_global_flag = communism_global_evolution_unstable_activity
			NOT = { has_global_flag = communism_global_evolution_dark_worker_rituals }
			check_variable = { communism_control_level_2_states > communism_evolution_ritual_threshold }
		}
		queue_communism_evolution_progression_cooldown = yes
		random_list = {
			constant:communism_spread_evolution.progression_roll_chance = { record_communism_spread_evolution_stage_2 = yes }
			constant:communism_spread_evolution.progression_no_change_chance = { }
		}
	}
	if = {
		limit = {
			NOT = { has_global_flag = communism_evolution_progression_cooldown }
			has_communism_evolution_tier_3_unlocked = yes
			has_global_flag = communism_global_evolution_dark_worker_rituals
			NOT = { has_global_flag = communism_global_evolution_world_revolution_whispers }
			check_variable = { communism_control_level_3_states > communism_evolution_whisper_threshold }
		}
		queue_communism_evolution_progression_cooldown = yes
		random_list = {
			constant:communism_spread_evolution.progression_roll_chance = { record_communism_spread_evolution_stage_3 = yes }
			constant:communism_spread_evolution.progression_no_change_chance = { }
		}
	}
}

try_communism_evolution_incident = {
	if = {
		limit = {
			NOT = { has_country_flag = communism_evolution_incident_cooldown }
			has_communism_evolution_tier_1_unlocked = yes
			has_country_flag = communism_evolution_unstable_activity
			any_owned_state = { has_state_flag = communist_controlled_state }
		}
		set_temp_variable = { communism_surprise_revolt_chance = constant:communism_spread_evolution.surprise_revolt_chance }
		set_temp_variable = { communism_worker_ritual_chance = 0 }
		set_temp_variable = { communism_world_revolution_whisper_chance = 0 }
		if = {
			limit = {
				has_communism_evolution_tier_2_unlocked = yes
				has_country_flag = communism_evolution_dark_worker_rituals
			}
			set_temp_variable = { communism_worker_ritual_chance = constant:communism_spread_evolution.worker_ritual_chance }
		}
		if = {
			limit = {
				has_communism_evolution_tier_3_unlocked = yes
				has_country_flag = communism_evolution_world_revolution_whispers
			}
			set_temp_variable = { communism_world_revolution_whisper_chance = constant:communism_spread_evolution.world_revolution_whisper_chance }
		}
		set_temp_variable = { communism_no_evolution_incident_chance = 100 }
		subtract_from_temp_variable = { communism_no_evolution_incident_chance = communism_surprise_revolt_chance }
		subtract_from_temp_variable = { communism_no_evolution_incident_chance = communism_worker_ritual_chance }
		subtract_from_temp_variable = { communism_no_evolution_incident_chance = communism_world_revolution_whisper_chance }
		random_list = {
			var:communism_surprise_revolt_chance = { apply_communist_surprise_revolt = yes }
			var:communism_worker_ritual_chance = { apply_dark_worker_ritual = yes }
			var:communism_world_revolution_whisper_chance = { apply_world_revolution_whisper = yes }
			var:communism_no_evolution_incident_chance = { }
		}
	}
}

spawn_world_revolution_units_from_communist_state = {
	# STATE SCOPE. Called before transfer to REV; state and country strength set count, control level biases quality.
	calculate_communist_rebel_state_unit_count = yes
	add_to_temp_variable = { communist_rebel_units_to_spawn = 1 }
	if = {
		limit = { has_state_flag = communist_control_level_3 }
		add_to_temp_variable = { communist_rebel_units_to_spawn = 1 }
	}
	set_temp_variable = { communist_rebel_units_spawned = 0 }
	while_loop_effect = {
		limit = { check_variable = { communist_rebel_units_spawned < communist_rebel_units_to_spawn } }
		add_to_temp_variable = { communist_rebel_units_spawned = 1 }
		spawn_rev_communist_rebel_unit_from_state = yes
	}
}

run_communism_spread_maintenance = {
	if = {
		limit = {
			has_global_flag = communism_spread
			hidden_trigger = { NOT = { is_special_chaos_country = yes } }
			has_idea = communism_spread_idea
			NOT = { has_government = communism }
		}
		refresh_communism_spread_pressure = yes
		try_communism_evolution_progression = yes
		try_communism_state_spread = yes
		try_communism_state_escalation = yes
		try_communism_sabotage_event = yes
		try_communism_evolution_incident = yes
		maybe_fire_revolutionary_war_warning = yes
		try_defeat_country_communism_spread_system = yes
		if = {
			limit = { has_idea = communism_spread_idea }
			queue_communism_spread_maintenance = yes
		}
	}
	else = {
		clear_communism_spread_system = yes
	}
}

try_defeat_country_communism_spread_system = {
	refresh_communism_spread_counts = yes
	if = {
			limit = {
				hidden_trigger = { NOT = { is_special_chaos_country = yes } }
				has_idea = communism_spread_idea
				NOT = { has_country_flag = fighting_communist_rebels }
				NOT = { has_country_flag = communism_spread_defeat_grace }
				check_variable = { communism_controlled_states = 0 }
				communism < constant:communism_spread_state_control.min_support_for_state_spread
			}
		country_event = { id = chaosx.nr1.17 days = 1 }
		clear_communism_spread_system = yes
	}
}

try_defeat_communism_spread_system = {
	if = {
		limit = {
			has_global_flag = communism_spread
			NOT = {
				any_country = {
					hidden_trigger = { NOT = { is_special_chaos_country = yes } }
					has_active_communism_spread_country_state = yes
				}
			}
			NOT = {
				any_country = {
					exists = yes
					has_country_flag = communist_rebel_state
				}
			}
		}
		clr_global_flag = communism_spread
		clr_global_flag = communism_spread_world_revolution_unlocked
		clr_global_flag = communism_global_evolution_unstable_activity
		clr_global_flag = communism_global_evolution_dark_worker_rituals
		clr_global_flag = communism_global_evolution_world_revolution_whispers
		clr_global_flag = communism_evolution_progression_cooldown
		set_temp_variable = { event_id = 91 }
		set_temp_variable = { new_weight = 0 }
		set_event_weight = yes
	}
}

clear_communism_spread_system = {
	if = {
		limit = { has_idea = communism_spread_idea }
		remove_ideas = communism_spread_idea
	}
	if = {
		limit = { has_dynamic_modifier = communism_state_control_pressure }
		remove_dynamic_modifier = { modifier = communism_state_control_pressure }
	}
	every_owned_state = {
		limit = { has_state_flag = communist_controlled_state }
		clear_communism_state_control = yes
	}
	clr_country_flag = communism_spread_maintenance_queued
	clr_country_flag = communism_state_spread_cooldown
	clr_country_flag = communism_state_escalation_cooldown
	clr_country_flag = communism_sabotage_cooldown
	clr_country_flag = communism_revolutionary_war_cooldown
	clr_country_flag = communism_evolution_incident_cooldown
	clr_country_flag = communism_spread_defeat_grace
	clr_country_flag = communism_evolution_unstable_activity
	clr_country_flag = communism_evolution_dark_worker_rituals
	clr_country_flag = communism_evolution_world_revolution_whispers
	clear_variable = communism_controlled_states
	clear_variable = communism_control_level_1_states
	clear_variable = communism_control_level_2_states
	clear_variable = communism_control_level_3_states
	clear_variable = communism_control_daily_drift
	clear_variable = communism_control_daily_drift_percent
	clear_variable = communism_revolutionary_threat
	clear_variable = communism_revolutionary_threat_bar
	try_defeat_communism_spread_system = yes
}
