# Communism spread state-control effects.
# COUNTRY SCOPE unless a block explicitly documents STATE SCOPE.
# The system is self-scheduled from chaosx.nr1.20 instead of using another global monthly country sweep.

@COMMUNISM_LOCAL_SUPPRESSION_DAYS = 90

queue_communism_spread_maintenance = {
	if = {
		limit = {
			has_global_flag = communism_spread
			hidden_trigger = { NOT = { is_special_chaos_country = yes } }
			has_idea = communism_spread_idea
			NOT = { has_country_flag = communism_spread_maintenance_queued }
			NOT = { has_government = communism }
			OR = {
				has_communism_spread_minimum_country_size = yes
				has_country_flag = fighting_communist_rebels
			}
		}
		set_country_flag = communism_spread_maintenance_queued
		country_event = {
			id = chaosx.nr1.20
			days = constant:communism_spread_timing.maintenance_min_days
			random_days = 30
		}
	}
}

queue_initial_communism_spread_maintenance = {
	if = {
		limit = {
			has_global_flag = communism_spread
			hidden_trigger = { NOT = { is_special_chaos_country = yes } }
			has_idea = communism_spread_idea
			NOT = { has_country_flag = communism_spread_maintenance_queued }
			NOT = { has_government = communism }
			has_communism_spread_minimum_country_size = yes
		}
		set_country_flag = communism_spread_maintenance_queued
		country_event = {
			id = chaosx.nr1.20
			days = constant:communism_spread_timing.first_maintenance_min_days
			random_days = 20
		}
	}
}

queue_communism_spread_defeat_grace_timeout = {
	set_country_flag = communism_spread_defeat_grace
	country_event = {
		id = chaosx.nr1.26
		days = constant:communism_spread_timing.defeat_grace_days
	}
}

adopt_communism_spread_controlled_state_front = {
	# COUNTRY SCOPE from on_state_control_changed. FROM.FROM is the state that changed controller.
	if = {
		limit = {
			has_global_flag = communism_spread
			hidden_trigger = { NOT = { is_special_chaos_country = yes } }
			NOT = { has_government = communism }
			FROM.FROM = {
				impassable = no
				is_controlled_by = ROOT
				has_state_flag = communist_controlled_state
			}
		}
		if = {
			limit = { NOT = { has_idea = communism_spread_idea } }
			add_ideas = communism_spread_idea
			queue_communism_spread_defeat_grace_timeout = yes
		}
		if = {
			limit = { communism < constant:communism_spread_state_control.min_support_for_state_spread }
			add_popularity = {
				ideology = communism
				popularity = constant:communism_spread_state_control.min_support_for_state_spread
			}
		}
		apply_global_communism_evolution_flags_to_country = yes
		refresh_communism_spread_pressure = yes
		queue_communism_spread_maintenance = yes
	}
}

set_communism_spread_initial_support_value = {
	get_current_chaos_tier_value = yes
	set_variable = { communism_spread_initial_support_effect = constant:communism_spread_initial_support.calm_world }
	if = {
		limit = {
			check_variable = {
				var = current_chaos_tier_value
				value = constant:chaos_meter_tier_id.tier_final
				compare = greater_than_or_equals
			}
		}
		set_variable = { communism_spread_initial_support_effect = constant:communism_spread_initial_support.world_collapse }
	}
	else_if = {
		limit = {
			check_variable = {
				var = current_chaos_tier_value
				value = constant:chaos_meter_tier_id.tier_4
				compare = greater_than_or_equals
			}
		}
		set_variable = { communism_spread_initial_support_effect = constant:communism_spread_initial_support.totalen_chaos }
	}
	else_if = {
		limit = {
			check_variable = {
				var = current_chaos_tier_value
				value = constant:chaos_meter_tier_id.tier_3
				compare = greater_than_or_equals
			}
		}
		set_variable = { communism_spread_initial_support_effect = constant:communism_spread_initial_support.chaos_tier }
	}
	else_if = {
		limit = {
			check_variable = {
				var = current_chaos_tier_value
				value = constant:chaos_meter_tier_id.tier_2
				compare = greater_than_or_equals
			}
		}
		set_variable = { communism_spread_initial_support_effect = constant:communism_spread_initial_support.rising_chaos }
	}
	else_if = {
		limit = {
			check_variable = {
				var = current_chaos_tier_value
				value = constant:chaos_meter_tier_id.tier_1
				compare = greater_than_or_equals
			}
		}
		set_variable = { communism_spread_initial_support_effect = constant:communism_spread_initial_support.gathering_storm }
	}
}

prepare_communism_local_intervention_costs = {
	# STATE SCOPE
	set_variable = { communist_intervention_cost_months = communist_control_duration_months }
	clamp_variable = {
		var = communist_intervention_cost_months
		max = constant:communism_spread_intervention.max_duration_cost_months
	}

	if = {
		limit = { has_state_flag = communist_control_level_1 }
		set_variable = { communist_intervention_command_power_cost = constant:communism_spread_intervention.level_1_command_power }
		set_variable = { communist_intervention_manpower_cost = constant:communism_spread_intervention.level_1_manpower }
		set_variable = { communist_intervention_equipment_cost = constant:communism_spread_intervention.level_1_equipment }
	}
	else_if = {
		limit = {
			has_state_flag = communist_controlled_state
			NOT = { has_state_flag = communist_control_level_2 }
			NOT = { has_state_flag = communist_control_level_3 }
		}
		set_variable = { communist_intervention_command_power_cost = constant:communism_spread_intervention.level_1_command_power }
		set_variable = { communist_intervention_manpower_cost = constant:communism_spread_intervention.level_1_manpower }
		set_variable = { communist_intervention_equipment_cost = constant:communism_spread_intervention.level_1_equipment }
	}
	else_if = {
		limit = { has_state_flag = communist_control_level_2 }
		set_variable = { communist_intervention_command_power_cost = constant:communism_spread_intervention.level_2_command_power }
		set_variable = { communist_intervention_manpower_cost = constant:communism_spread_intervention.level_2_manpower }
		set_variable = { communist_intervention_equipment_cost = constant:communism_spread_intervention.level_2_equipment }
	}
	else_if = {
		limit = { has_state_flag = communist_control_level_3 }
		set_variable = { communist_intervention_command_power_cost = constant:communism_spread_intervention.level_3_command_power }
		set_variable = { communist_intervention_manpower_cost = constant:communism_spread_intervention.level_3_manpower }
		set_variable = { communist_intervention_equipment_cost = constant:communism_spread_intervention.level_3_equipment }
	}

	# Local intervention cost is tied to the current control stage.
	# Duration still affects state penalties, but not the resource payment shown in the decision UI.

	round_variable = communist_intervention_command_power_cost
	round_variable = communist_intervention_manpower_cost
	round_variable = communist_intervention_equipment_cost

	set_variable = { communist_intervention_command_power_cost_negative = communist_intervention_command_power_cost }
	multiply_variable = { communist_intervention_command_power_cost_negative = -1 }
	set_variable = { communist_intervention_manpower_cost_negative = communist_intervention_manpower_cost }
	multiply_variable = { communist_intervention_manpower_cost_negative = -1 }
	set_variable = { communist_intervention_equipment_cost_negative = communist_intervention_equipment_cost }
	multiply_variable = { communist_intervention_equipment_cost_negative = -1 }
}

refresh_communism_state_control_modifier = {
	# STATE SCOPE
	if = {
		limit = { has_state_flag = communist_control_level_1 }
		set_variable = { communist_control_penalty_positive = constant:communism_spread_state_penalties.level_1_base_penalty }
		set_temp_variable = { communist_control_duration_penalty = communist_control_duration_months }
		multiply_temp_variable = { communist_control_duration_penalty = constant:communism_spread_state_penalties.level_1_penalty_per_month }
		add_to_variable = { communist_control_penalty_positive = communist_control_duration_penalty }
		clamp_variable = {
			var = communist_control_penalty_positive
			max = constant:communism_spread_state_penalties.level_1_max_penalty
		}
	}
	else_if = {
		limit = { has_state_flag = communist_control_level_2 }
		set_variable = { communist_control_penalty_positive = constant:communism_spread_state_penalties.level_2_base_penalty }
		set_temp_variable = { communist_control_duration_penalty = communist_control_duration_months }
		multiply_temp_variable = { communist_control_duration_penalty = constant:communism_spread_state_penalties.level_2_penalty_per_month }
		add_to_variable = { communist_control_penalty_positive = communist_control_duration_penalty }
		clamp_variable = {
			var = communist_control_penalty_positive
			max = constant:communism_spread_state_penalties.level_2_max_penalty
		}
	}
	else_if = {
		limit = { has_state_flag = communist_control_level_3 }
		set_variable = { communist_control_penalty_positive = constant:communism_spread_state_penalties.level_3_base_penalty }
		set_temp_variable = { communist_control_duration_penalty = communist_control_duration_months }
		multiply_temp_variable = { communist_control_duration_penalty = constant:communism_spread_state_penalties.level_3_penalty_per_month }
		add_to_variable = { communist_control_penalty_positive = communist_control_duration_penalty }
		clamp_variable = {
			var = communist_control_penalty_positive
			max = constant:communism_spread_state_penalties.level_3_max_penalty
		}
	}

	if = {
		limit = { has_state_flag = communist_local_suppression_active }
		multiply_variable = { communist_control_penalty_positive = constant:communism_spread_state_penalties.suppression_penalty_factor }
	}

	set_variable = { communist_control_factory_penalty = communist_control_penalty_positive }
	multiply_variable = { communist_control_factory_penalty = -1 }
	set_variable = { communist_control_manpower_penalty = communist_control_penalty_positive }
	multiply_variable = { communist_control_manpower_penalty = -1 }
	set_variable = { communist_control_resource_penalty = communist_control_penalty_positive }
	multiply_variable = { communist_control_resource_penalty = constant:communism_spread_state_penalties.resources_factor }
	multiply_variable = { communist_control_resource_penalty = -1 }
	set_variable = { communist_control_supply_penalty = communist_control_penalty_positive }
	multiply_variable = { communist_control_supply_penalty = constant:communism_spread_state_penalties.supplies_factor }
	multiply_variable = { communist_control_supply_penalty = -1 }
	set_variable = { communist_control_building_speed_penalty = communist_control_penalty_positive }
	multiply_variable = { communist_control_building_speed_penalty = -1 }
	set_variable = { communist_control_resistance_target = communist_control_penalty_positive }
	multiply_variable = { communist_control_resistance_target = constant:communism_spread_state_penalties.resistance_factor }

	prepare_communism_local_intervention_costs = yes

	if = {
		limit = {
			has_state_flag = communist_control_level_1
			NOT = { has_dynamic_modifier = communism_control_level_1 }
		}
		add_dynamic_modifier = { modifier = communism_control_level_1 }
	}
	else_if = {
		limit = {
			has_state_flag = communist_control_level_2
			NOT = { has_dynamic_modifier = communism_control_level_2 }
		}
		add_dynamic_modifier = { modifier = communism_control_level_2 }
	}
	else_if = {
		limit = {
			has_state_flag = communist_control_level_3
			NOT = { has_dynamic_modifier = communism_control_level_3 }
		}
		add_dynamic_modifier = { modifier = communism_control_level_3 }
	}
	force_update_dynamic_modifier = yes
}

refresh_communism_spread_counts = {
	set_variable = { communism_controlled_states = 0 }
	set_variable = { communism_control_level_1_states = 0 }
	set_variable = { communism_control_level_2_states = 0 }
	set_variable = { communism_control_level_3_states = 0 }
	set_variable = { communism_emergency_control_threshold = num_of_controlled_states }
	multiply_variable = { communism_emergency_control_threshold = 0.5 }
	# Trigger syntax has no inclusive greater-than check, so 50% must pass a > comparison.
	subtract_from_variable = { communism_emergency_control_threshold = 0.1 }

	every_controlled_state = {
		limit = { has_state_flag = communist_controlled_state }
		ROOT = { add_to_variable = { communism_controlled_states = 1 } }
		if = {
			limit = { has_state_flag = communist_control_level_1 }
			ROOT = { add_to_variable = { communism_control_level_1_states = 1 } }
		}
		else_if = {
			limit = { has_state_flag = communist_control_level_2 }
			ROOT = { add_to_variable = { communism_control_level_2_states = 1 } }
		}
		else_if = {
			limit = { has_state_flag = communist_control_level_3 }
			ROOT = { add_to_variable = { communism_control_level_3_states = 1 } }
		}
	}
}

resolve_communism_local_intervention = {
	# COUNTRY SCOPE, with FROM as the targeted state.
	hidden_effect = {
		FROM = { save_event_target_as = communist_target_state }
		if = {
			limit = {
				FROM = {
					OR = {
						has_state_flag = communist_control_level_1
						AND = {
							has_state_flag = communist_controlled_state
							NOT = { has_state_flag = communist_control_level_2 }
							NOT = { has_state_flag = communist_control_level_3 }
						}
					}
				}
			}
			random_list = {
				constant:communism_spread_intervention.level_1_success = {
					FROM = {
						clear_communism_state_control = yes
						add_dynamic_modifier = {
							modifier = communism_post_crackdown_scars
							days = 120
						}
					}
					country_event = { id = chaosx.nr1.6 }
				}
				constant:communism_spread_intervention.level_1_partial = {
					FROM = {
						add_communism_intervention_progress_partial = yes
						try_finish_communism_level_1_intervention = yes
					}
					add_stability = -0.005
					country_event = { id = chaosx.nr1.6 }
				}
				constant:communism_spread_intervention.level_1_bad_failure = {
					FROM = { set_communism_state_control_level_2 = yes }
					add_popularity = {
						ideology = communism
						popularity = constant:communism_spread_intervention.failed_support_gain
					}
					country_event = { id = chaosx.nr1.7 }
				}
			}
		}
		else_if = {
			limit = { FROM = { has_state_flag = communist_control_level_2 } }
			random_list = {
				constant:communism_spread_intervention.level_2_success = {
					FROM = {
						add_communism_intervention_progress_success = yes
						try_finish_communism_level_2_intervention = yes
					}
					country_event = { id = chaosx.nr1.6 }
				}
				constant:communism_spread_intervention.level_2_partial = {
					FROM = {
						add_communism_intervention_progress_partial = yes
						try_finish_communism_level_2_intervention = yes
					}
					add_stability = -0.01
					country_event = { id = chaosx.nr1.6 }
				}
				constant:communism_spread_intervention.level_2_bad_failure = {
					FROM = { set_communism_state_control_level_3 = yes }
					add_popularity = {
						ideology = communism
						popularity = constant:communism_spread_intervention.failed_support_gain
					}
					country_event = { id = chaosx.nr1.7 }
				}
			}
		}
		else_if = {
			limit = { FROM = { has_state_flag = communist_control_level_3 } }
			random_list = {
				constant:communism_spread_intervention.level_3_suppression = {
					FROM = { apply_communism_local_suppression = yes }
					add_stability = -0.015
					country_event = { id = chaosx.nr1.6 }
				}
				constant:communism_spread_intervention.level_3_bad_failure = {
					FROM = { advance_communism_state_control_duration = yes }
					add_popularity = {
						ideology = communism
						popularity = constant:communism_spread_intervention.failed_support_gain
					}
					add_stability = -0.02
					country_event = { id = chaosx.nr1.7 }
				}
			}
		}
		refresh_communism_spread_pressure = yes
	}

}

try_communism_state_spread = {
	can_communism_spread_take_new_state = yes
	if = {
		limit = { check_variable = { communism_spread_can_take_state = 1 } }
		set_temp_variable = { communism_state_spread_chance = constant:communism_spread_state_control.base_spread_chance }
		set_temp_variable = { non_ai_crisis_pressure_chance_bonus = 0 }
		set_temp_variable = { non_ai_crisis_pressure_cooldown_reduction = 0 }
		set_temp_variable = { communism_state_spread_control_pressure = communism_controlled_states }
		multiply_temp_variable = { communism_state_spread_control_pressure = constant:communism_spread_state_control.controlled_state_spread_chance }
		set_temp_variable = { communism_state_spread_control_pressure_cap = constant:communism_spread_state_control.max_controlled_state_spread_chance }
		if = {
			limit = { num_of_controlled_states > constant:communism_spread_state_control.medium_country_min_states_check }
			add_to_temp_variable = { communism_state_spread_control_pressure_cap = constant:communism_spread_state_control.medium_country_controlled_state_pressure_cap_bonus }
		}
		if = {
			limit = { num_of_controlled_states > constant:communism_spread_state_control.large_country_min_states_check }
			add_to_temp_variable = { communism_state_spread_control_pressure_cap = constant:communism_spread_state_control.large_country_controlled_state_pressure_cap_bonus }
		}
		if = {
			limit = { num_of_controlled_states > constant:communism_spread_state_control.huge_country_min_states_check }
			add_to_temp_variable = { communism_state_spread_control_pressure_cap = constant:communism_spread_state_control.huge_country_controlled_state_pressure_cap_bonus }
		}
		clamp_temp_variable = {
			var = communism_state_spread_control_pressure
			max = communism_state_spread_control_pressure_cap
		}
		add_to_temp_variable = { communism_state_spread_chance = communism_state_spread_control_pressure }
		if = {
			limit = { num_of_controlled_states > 0 }
			set_temp_variable = { communism_state_spread_control_ratio_pressure = communism_controlled_states }
			divide_temp_variable = { communism_state_spread_control_ratio_pressure = num_of_controlled_states }
			multiply_temp_variable = { communism_state_spread_control_ratio_pressure = constant:communism_spread_state_control.controlled_state_ratio_spread_scale }
			clamp_temp_variable = {
				var = communism_state_spread_control_ratio_pressure
				max = constant:communism_spread_state_control.max_controlled_state_ratio_spread_chance
			}
			add_to_temp_variable = { communism_state_spread_chance = communism_state_spread_control_ratio_pressure }
		}
		if = {
			limit = { num_of_controlled_states > constant:communism_spread_state_control.medium_country_min_states_check }
			add_to_temp_variable = { communism_state_spread_chance = constant:communism_spread_state_control.medium_country_spread_chance }
		}
		if = {
			limit = { num_of_controlled_states > constant:communism_spread_state_control.large_country_min_states_check }
			add_to_temp_variable = { communism_state_spread_chance = constant:communism_spread_state_control.large_country_spread_chance }
		}
		if = {
			limit = { num_of_controlled_states > constant:communism_spread_state_control.huge_country_min_states_check }
			add_to_temp_variable = { communism_state_spread_chance = constant:communism_spread_state_control.huge_country_spread_chance }
		}
		if = {
			limit = { communism > constant:communism_spread_state_control.medium_support }
			add_to_temp_variable = { communism_state_spread_chance = constant:communism_spread_state_control.medium_support_spread_chance }
		}
		if = {
			limit = { communism > constant:communism_spread_state_control.high_support }
			add_to_temp_variable = { communism_state_spread_chance = constant:communism_spread_state_control.high_support_spread_chance }
		}
		if = {
			limit = { communism > constant:communism_spread_state_control.very_high_support }
			add_to_temp_variable = { communism_state_spread_chance = constant:communism_spread_state_control.very_high_support_spread_chance }
		}
		if = {
			limit = { has_stability < constant:communism_spread_state_control.low_stability_threshold }
			add_to_temp_variable = { communism_state_spread_chance = constant:communism_spread_state_control.low_stability_spread_chance }
		}
		if = {
			limit = { has_stability < constant:communism_spread_state_control.very_low_stability_threshold }
			add_to_temp_variable = { communism_state_spread_chance = constant:communism_spread_state_control.very_low_stability_spread_chance }
		}
		if = {
			limit = { has_war_support < constant:communism_spread_state_control.war_support_low_threshold }
			add_to_temp_variable = { communism_state_spread_chance = constant:communism_spread_state_control.low_war_support_spread_chance }
		}
		if = {
			limit = { has_war = yes }
			add_to_temp_variable = { communism_state_spread_chance = constant:communism_spread_state_control.at_war_spread_chance }
		}
		if = {
			limit = { is_non_ai_fighting_severe_communism_spread_crisis = yes }
			chaosx_apply_non_ai_crisis_pressure_severe_factor = yes
		}
		else_if = {
			limit = { is_non_ai_fighting_communism_spread_crisis = yes }
			chaosx_apply_non_ai_crisis_pressure_factor = yes
		}
		add_to_temp_variable = { communism_state_spread_chance = non_ai_crisis_pressure_chance_bonus }
		clamp_temp_variable = {
			var = communism_state_spread_chance
			max = constant:communism_spread_state_control.max_state_spread_chance
		}
		set_temp_variable = { communism_state_no_spread_chance = 100 }
		subtract_from_temp_variable = { communism_state_no_spread_chance = communism_state_spread_chance }
		random_list = {
			var:communism_state_spread_chance = {
				if = {
					limit = {
						any_controlled_state = {
							impassable = no
							NOT = { is_capital = yes }
							NOT = { has_state_flag = communist_controlled_state }
							NOT = { any_neighbor_state = { has_state_flag = communist_controlled_state } }
						}
					}
					set_temp_variable = { communism_state_remote_spread_chance = constant:communism_spread_state_control.remote_spread_chance }
					if = {
						limit = { num_of_controlled_states > constant:communism_spread_state_control.medium_country_min_states_check }
						add_to_temp_variable = { communism_state_remote_spread_chance = constant:communism_spread_state_control.medium_country_remote_spread_bonus }
					}
					if = {
						limit = { num_of_controlled_states > constant:communism_spread_state_control.large_country_min_states_check }
						add_to_temp_variable = { communism_state_remote_spread_chance = constant:communism_spread_state_control.large_country_remote_spread_bonus }
					}
					if = {
						limit = { num_of_controlled_states > constant:communism_spread_state_control.huge_country_min_states_check }
						add_to_temp_variable = { communism_state_remote_spread_chance = constant:communism_spread_state_control.huge_country_remote_spread_bonus }
					}
					if = {
						limit = { communism > constant:communism_spread_state_control.high_support }
						add_to_temp_variable = { communism_state_remote_spread_chance = constant:communism_spread_state_control.remote_spread_high_support_bonus }
					}
					if = {
						limit = { has_stability < constant:communism_spread_state_control.low_stability_threshold }
						add_to_temp_variable = { communism_state_remote_spread_chance = constant:communism_spread_state_control.remote_spread_low_stability_bonus }
					}
					if = {
						limit = { has_war = yes }
						add_to_temp_variable = { communism_state_remote_spread_chance = constant:communism_spread_state_control.remote_spread_at_war_bonus }
					}
					if = {
						limit = { is_non_ai_fighting_communism_spread_crisis = yes }
						set_temp_variable = { non_ai_crisis_pressure_chance_bonus = 0 }
						set_temp_variable = { non_ai_crisis_pressure_cooldown_reduction = 0 }
						chaosx_apply_non_ai_crisis_pressure_mild_factor = yes
						add_to_temp_variable = { communism_state_remote_spread_chance = non_ai_crisis_pressure_chance_bonus }
					}
					clamp_temp_variable = {
						var = communism_state_remote_spread_chance
						max = constant:communism_spread_state_control.max_remote_spread_chance
					}
					set_temp_variable = { communism_state_neighbor_spread_chance = 100 }
					subtract_from_temp_variable = { communism_state_neighbor_spread_chance = communism_state_remote_spread_chance }
					random_list = {
						var:communism_state_neighbor_spread_chance = { select_new_communist_control_state = yes }
						var:communism_state_remote_spread_chance = { select_remote_communist_control_state = yes }
					}
				}
				else = {
					select_new_communist_control_state = yes
				}
				country_event = { id = chaosx.nr1.4 days = 1 }
			}
			var:communism_state_no_spread_chance = { }
		}
	}
}

			limit = { communism > constant:communism_spread_state_control.medium_support }
			add_to_temp_variable = { communism_sabotage_chance = constant:communism_spread_strikes.medium_support_chance }
		}
		if = {
			limit = { communism > constant:communism_spread_state_control.high_support }
			add_to_temp_variable = { communism_sabotage_chance = constant:communism_spread_strikes.high_support_chance }
		}

			limit = { check_variable = { communism_control_level_2_states > 0 } }
			add_to_temp_variable = { communism_sabotage_chance = constant:communism_spread_strikes.level_2_state_chance }

			limit = { check_variable = { communism_control_level_3_states > 0 } }
			add_to_temp_variable = { communism_sabotage_chance = constant:communism_spread_strikes.level_3_state_chance }

mark_communist_uprising_conversion_states = {
	clear_communist_uprising_conversion_flags = yes
	refresh_communism_spread_counts = yes
	set_temp_variable = { communist_uprising_max_convert_states = num_of_controlled_states }
	subtract_from_temp_variable = { communist_uprising_max_convert_states = 1 }
	set_temp_variable = { communist_uprising_convert_budget = constant:communism_spread_emergency.uprising_base_convert_states }
	if = {
		limit = { check_variable = { communism_revolutionary_threat > constant:communism_spread_emergency.uprising_threat_medium } }
		add_to_temp_variable = { communist_uprising_convert_budget = 1 }
	}
	if = {
		limit = { check_variable = { communism_revolutionary_threat > constant:communism_spread_emergency.uprising_threat_high } }
		add_to_temp_variable = { communist_uprising_convert_budget = 1 }
	}
	if = {
		limit = { check_variable = { communism_revolutionary_threat > constant:communism_spread_emergency.uprising_threat_extreme } }
		add_to_temp_variable = { communist_uprising_convert_budget = 1 }
	}
	if = {
		limit = { communism > constant:communism_spread_state_control.high_support }
		add_to_temp_variable = { communist_uprising_convert_budget = constant:communism_spread_emergency.uprising_support_high_bonus }
	}
	if = {
		limit = { communism > constant:communism_spread_state_control.very_high_support }
		add_to_temp_variable = { communist_uprising_convert_budget = constant:communism_spread_emergency.uprising_support_extreme_bonus }
	}
	if = {
		limit = { check_variable = { communism_control_level_3_states > 0 } }
		add_to_temp_variable = { communist_uprising_convert_budget = constant:communism_spread_emergency.uprising_level_3_bonus }
	}
	if = {
		limit = { check_variable = { communism_control_level_3_states > 2 } }
		add_to_temp_variable = { communist_uprising_convert_budget = constant:communism_spread_emergency.uprising_many_level_3_bonus }
	}
	if = {
		limit = { check_variable = { communist_uprising_convert_budget > communist_uprising_max_convert_states } }
		set_temp_variable = { communist_uprising_convert_budget = communist_uprising_max_convert_states }
	}
	if = {
		limit = { check_variable = { communist_uprising_convert_budget < 1 } }
		set_temp_variable = { communist_uprising_convert_budget = 0 }
	}
	while_loop_effect = {
		limit = {
			check_variable = { communist_uprising_convert_budget > 0 }
			any_controlled_state = {
				NOT = { is_capital = yes }
				has_state_flag = communist_control_level_3
				NOT = { has_state_flag = communist_uprising_convert }
			}
		}
		random_controlled_state = {
			limit = {
				NOT = { is_capital = yes }
				has_state_flag = communist_control_level_3
				NOT = { has_state_flag = communist_uprising_convert }
			}
			set_state_flag = communist_uprising_convert
		}
		subtract_from_temp_variable = { communist_uprising_convert_budget = 1 }
	}
	while_loop_effect = {
		limit = {
			check_variable = { communist_uprising_convert_budget > 0 }
			any_controlled_state = {
				NOT = { is_capital = yes }
				has_state_flag = communist_control_level_2
				NOT = { has_state_flag = communist_uprising_convert }
			}
		}
		random_controlled_state = {
			limit = {
				NOT = { is_capital = yes }
				has_state_flag = communist_control_level_2
				NOT = { has_state_flag = communist_uprising_convert }
			}
			set_state_flag = communist_uprising_convert
		}
		subtract_from_temp_variable = { communist_uprising_convert_budget = 1 }
	}
	while_loop_effect = {
		limit = {
			check_variable = { communist_uprising_convert_budget > 0 }
			any_controlled_state = {
				NOT = { is_capital = yes }
				OR = {
					has_state_flag = communist_control_level_1
					AND = {
						has_state_flag = communist_controlled_state
						NOT = { has_state_flag = communist_control_level_2 }
						NOT = { has_state_flag = communist_control_level_3 }
					}
				}
				NOT = { has_state_flag = communist_uprising_convert }
			}
		}
		random_controlled_state = {
			limit = {
				NOT = { is_capital = yes }
				OR = {
					has_state_flag = communist_control_level_1
					AND = {
						has_state_flag = communist_controlled_state
						NOT = { has_state_flag = communist_control_level_2 }
						NOT = { has_state_flag = communist_control_level_3 }
					}
				}
				NOT = { has_state_flag = communist_uprising_convert }
			}
			set_state_flag = communist_uprising_convert
		}
		subtract_from_temp_variable = { communist_uprising_convert_budget = 1 }
	}
}

	# STATE SCOPE. Called while the dynamic rebel country is PREV.PREV.
	calculate_communist_rebel_state_unit_count = yes
	if = {
		limit = { has_state_flag = communist_control_level_3 }
		spawn_dynamic_communist_rebel_unit_from_state = yes
	}
	set_temp_variable = { communist_rebel_units_spawned = 0 }
	while_loop_effect = {
		limit = { check_variable = { communist_rebel_units_spawned < communist_rebel_units_to_spawn } }
		add_to_temp_variable = { communist_rebel_units_spawned = 1 }
		spawn_dynamic_communist_rebel_unit_from_state = yes
	}
}

try_communism_evolution_progression = {
	refresh_communism_spread_counts = yes
	apply_global_communism_evolution_flags_to_country = yes
	set_temp_variable = { communism_evolution_unstable_threshold = constant:communism_spread_evolution.unstable_min_controlled_states }
	subtract_from_temp_variable = { communism_evolution_unstable_threshold = 1 }
	set_temp_variable = { communism_evolution_ritual_threshold = constant:communism_spread_evolution.dark_rituals_min_level_2_states }
	subtract_from_temp_variable = { communism_evolution_ritual_threshold = 1 }
	set_temp_variable = { communism_evolution_whisper_threshold = constant:communism_spread_evolution.world_revolution_whispers_min_level_3_states }
	subtract_from_temp_variable = { communism_evolution_whisper_threshold = 1 }
	if = {
		limit = {
			NOT = { has_global_flag = communism_evolution_progression_cooldown }
			has_communism_evolution_tier_1_unlocked = yes
			NOT = { has_global_flag = communism_global_evolution_unstable_activity }
			check_variable = { communism_controlled_states > communism_evolution_unstable_threshold }
		}
		queue_communism_evolution_progression_cooldown = yes
		random_list = {
			constant:communism_spread_evolution.progression_roll_chance = { record_communism_spread_evolution_stage_1 = yes }
			constant:communism_spread_evolution.progression_no_change_chance = { }
		}
	}
	if = {
		limit = {
			NOT = { has_global_flag = communism_evolution_progression_cooldown }
			has_communism_evolution_tier_2_unlocked = yes
			has_global_flag = communism_global_evolution_unstable_activity
			NOT = { has_global_flag = communism_global_evolution_dark_worker_rituals }
			check_variable = { communism_control_level_2_states > communism_evolution_ritual_threshold }
		}
		queue_communism_evolution_progression_cooldown = yes
		random_list = {
			constant:communism_spread_evolution.progression_roll_chance = { record_communism_spread_evolution_stage_2 = yes }
			constant:communism_spread_evolution.progression_no_change_chance = { }
		}
	}
	if = {
		limit = {
			NOT = { has_global_flag = communism_evolution_progression_cooldown }
			has_communism_evolution_tier_3_unlocked = yes
			has_global_flag = communism_global_evolution_dark_worker_rituals
			NOT = { has_global_flag = communism_global_evolution_world_revolution_whispers }
			check_variable = { communism_control_level_3_states > communism_evolution_whisper_threshold }
		}
		queue_communism_evolution_progression_cooldown = yes
		random_list = {
			constant:communism_spread_evolution.progression_roll_chance = { record_communism_spread_evolution_stage_3 = yes }
			constant:communism_spread_evolution.progression_no_change_chance = { }
		}
	}
}

spawn_world_revolution_units_from_communist_state = {
	# STATE SCOPE. Called before transfer to REV; state and country strength set count, control level biases quality.
	calculate_communist_rebel_state_unit_count = yes
	add_to_temp_variable = { communist_rebel_units_to_spawn = 1 }
	if = {
		limit = { has_state_flag = communist_control_level_3 }
		add_to_temp_variable = { communist_rebel_units_to_spawn = 1 }
	}
	set_temp_variable = { communist_rebel_units_spawned = 0 }
	while_loop_effect = {
		limit = { check_variable = { communist_rebel_units_spawned < communist_rebel_units_to_spawn } }
		add_to_temp_variable = { communist_rebel_units_spawned = 1 }
		spawn_rev_communist_rebel_unit_from_state = yes
	}
}

run_communism_spread_maintenance = {
	if = {
		limit = {
			has_global_flag = communism_spread
			hidden_trigger = { NOT = { is_special_chaos_country = yes } }
			has_idea = communism_spread_idea
			NOT = { has_government = communism }
			OR = {
				has_communism_spread_minimum_country_size = yes
				has_country_flag = fighting_communist_rebels
			}
		}
		every_controlled_state = {
			limit = { has_state_flag = communist_controlled_state }
			advance_communism_state_control_duration = yes
		}
		refresh_communism_spread_pressure = yes
		try_communism_evolution_progression = yes
		try_communism_state_spread = yes
		try_communism_state_escalation = yes
		try_communism_sabotage_event = yes
		try_communism_evolution_incident = yes
		maybe_fire_revolutionary_war_warning = yes
		try_defeat_country_communism_spread_system = yes
		if = {
			limit = { has_idea = communism_spread_idea }
			queue_communism_spread_maintenance = yes
		}
	}
	else = {
		clear_communism_spread_system = yes
	}
}

try_defeat_country_communism_spread_system = {
	refresh_communism_spread_counts = yes
	if = {
			limit = {
				hidden_trigger = { NOT = { is_special_chaos_country = yes } }
				has_idea = communism_spread_idea
				NOT = { has_country_flag = fighting_communist_rebels }
				NOT = { has_country_flag = communism_spread_defeat_grace }
				check_variable = { communism_controlled_states = 0 }
				communism < constant:communism_spread_state_control.min_support_for_state_spread
			}
		country_event = { id = chaosx.nr1.17 days = 1 }
		clear_communism_spread_system = yes
	}
}

try_defeat_communism_spread_system = {
	if = {
		limit = {
			has_global_flag = communism_spread
			NOT = {
				any_country = {
					hidden_trigger = { NOT = { is_special_chaos_country = yes } }
					has_active_communism_spread_country_state = yes
				}
			}
			NOT = {
				any_country = {
					exists = yes
					has_country_flag = communist_rebel_state
				}
			}
		}
		clr_global_flag = communism_spread
		clr_global_flag = communism_spread_world_revolution_unlocked
		clr_global_flag = communism_global_evolution_unstable_activity
		clr_global_flag = communism_global_evolution_dark_worker_rituals
		clr_global_flag = communism_global_evolution_world_revolution_whispers
		clr_global_flag = communism_evolution_progression_cooldown
		set_temp_variable = { event_id = 91 }
		set_temp_variable = { new_weight = 0 }
		set_event_weight = yes
	}
}
