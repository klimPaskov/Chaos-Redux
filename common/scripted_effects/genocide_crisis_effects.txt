# CHAOS REDUX - CAMPS AND GENOCIDE CRISIS EFFECTS
# ============================================================================
# Overview:
# - Builds and tracks concentration camps, extermination camps, and gulag
#   networks as state infrastructure with a stored responsible country.
# - Registers monthly civilian deaths through the shared Chaos Meter deaths
#   pipeline while keeping external condemnation hidden until enemy discovery.
# - Handles discovery, evidence destruction, foreign response thresholds,
#   historical starting sites, and country-specific AI paths.
# ============================================================================

	if = { limit = { NOT = { has_variable = genocide_foreign_pressure } } set_variable = { genocide_foreign_pressure = 0 } }
	if = { limit = { NOT = { has_variable = genocide_coverup_effort } } set_variable = { genocide_coverup_effort = 0 } }

genocide_activate_crisis = {
	genocide_initialize_country_variables = yes
	set_country_flag = genocide_crisis_active
}

genocide_add_radical_hardliner_support = {
	genocide_activate_crisis = yes
	add_timed_idea = {
		idea = radical_hardliner_support
		days = @GENOCIDE_HARDLINER_DAYS
	}
	if = {
		limit = { has_government = fascism }
		add_popularity = {
			ideology = fascism
			popularity = constant:genocide_political_effect.hardliner_fascism
		}
	}
	if = {
		limit = { has_government = communism }
		add_popularity = {
			ideology = communism
			popularity = constant:genocide_political_effect.hardliner_communism
		}
	}
	add_popularity = {
		ideology = democratic
		popularity = constant:genocide_political_effect.hardliner_democratic
	}
	add_popularity = {
		ideology = neutrality
		popularity = constant:genocide_political_effect.hardliner_neutrality
	}
	add_stability = constant:genocide_political_effect.hardliner_stability
}

genocide_credit_state_deaths_to_responsible = {
	if = {
		limit = {
			has_variable = genocide_responsible_country
			has_variable = genocide_last_state_deaths
			check_variable = { genocide_last_state_deaths > 0 }
		}
		var:genocide_responsible_country = {
			genocide_initialize_country_variables = yes
			add_to_variable = { genocide_deaths = PREV.genocide_last_state_deaths }
			add_to_variable = { hidden_atrocity_score = constant:genocide_crisis_value.hidden_atrocity_minor }
		}
	}
}

genocide_track_state_for_monthly_pulse = {
	if = {
		limit = {
			NOT = {
				is_in_array = {
					array = global.genocide_active_camp_states
					value = THIS
				}
			}
		}
		add_to_array = {
			array = global.genocide_active_camp_states
			value = THIS
		}
	}
}

genocide_prepare_death_multiplier_from_responsible = {
	set_temp_variable = { chaos_state_deaths_mult = 1 }
	if = {
		limit = { has_variable = genocide_responsible_country }
		var:genocide_responsible_country = {
			if = {
				limit = { has_variable = genocide_escalation }
				set_temp_variable = { genocide_death_escalation_mult = genocide_escalation }
				multiply_temp_variable = { genocide_death_escalation_mult = constant:genocide_death_multiplier.escalation_factor }
				add_to_temp_variable = { chaos_state_deaths_mult = genocide_death_escalation_mult }
			}
			if = {
				limit = { has_idea = radical_hardliner_support }
				add_to_temp_variable = { chaos_state_deaths_mult = constant:genocide_death_multiplier.hardliner_bonus }
			}
			if = {
				limit = { has_idea = genocide_rail_deportation_priority }
				add_to_temp_variable = { chaos_state_deaths_mult = constant:genocide_death_multiplier.rail_priority_bonus }
			}
			if = {
				limit = { has_country_flag = genocide_extermination_policy_intensified }
				add_to_temp_variable = { chaos_state_deaths_mult = constant:genocide_death_multiplier.policy_intensification_bonus }
			}
			if = {
				limit = { has_idea = germany_auschwitz_experiments_restricted }
				add_to_temp_variable = { chaos_state_deaths_mult = constant:genocide_death_multiplier.mengele_restricted_bonus }
			}
			if = {
				limit = { has_idea = germany_auschwitz_experiments }
				add_to_temp_variable = { chaos_state_deaths_mult = constant:genocide_death_multiplier.mengele_full_bonus }
			}
			if = {
				limit = {
					has_variable = mengele_permission_level
					check_variable = { mengele_permission_level > constant:germany_mengele_permission.full }
				}
				add_to_temp_variable = { chaos_state_deaths_mult = constant:genocide_death_multiplier.mengele_bypass_bonus }
			}
		}
	}
	if = {
		limit = { has_state_flag = genocide_auschwitz_experiment_site }
		add_to_temp_variable = { chaos_state_deaths_mult = constant:genocide_death_multiplier.auschwitz_bonus }
	}
	if = {
		limit = { has_state_flag = genocide_restricted_chemical_site_active }
		add_to_temp_variable = { chaos_state_deaths_mult = constant:genocide_death_multiplier.chemical_site_bonus }
	}
	clamp_temp_variable = {
		var = chaos_state_deaths_mult
		min = 1
		max = constant:genocide_death_multiplier.max
	}
}

genocide_refresh_concentration_camp_network_modifier = {
	if = {
		limit = { concentration_camp > 0 }
		set_temp_variable = { genocide_camp_production = constant:genocide_concentration_modifier.production_base }
		set_temp_variable = { genocide_camp_resources = constant:genocide_concentration_modifier.resources_base }
		set_temp_variable = { genocide_camp_resistance = constant:genocide_concentration_modifier.resistance_base }
		set_temp_variable = { genocide_camp_compliance = constant:genocide_concentration_modifier.compliance_base }
		if = {
			limit = { has_variable = genocide_responsible_country }
			var:genocide_responsible_country = {
				if = {
					limit = { has_variable = genocide_concentration_sites }
					set_temp_variable = { genocide_camp_network_value = genocide_concentration_sites }
					multiply_temp_variable = { genocide_camp_network_value = constant:genocide_concentration_modifier.production_network }
					add_to_temp_variable = { genocide_camp_production = genocide_camp_network_value }
					set_temp_variable = { genocide_camp_network_resources = genocide_concentration_sites }
					multiply_temp_variable = { genocide_camp_network_resources = constant:genocide_concentration_modifier.resources_network }
					add_to_temp_variable = { genocide_camp_resources = genocide_camp_network_resources }
				}
				if = {
					limit = {
						has_government = fascism
						check_variable = { party_popularity@fascism > 0.45 }
					}
					add_to_temp_variable = { genocide_camp_production = constant:genocide_concentration_modifier.production_fascist }
				}
				if = {
					limit = { has_variable = genocide_escalation }
					set_temp_variable = { genocide_camp_escalation_value = genocide_escalation }
					multiply_temp_variable = { genocide_camp_escalation_value = constant:genocide_concentration_modifier.production_escalation }
					add_to_temp_variable = { genocide_camp_production = genocide_camp_escalation_value }
					set_temp_variable = { genocide_camp_escalation_resistance = genocide_escalation }
					multiply_temp_variable = { genocide_camp_escalation_resistance = constant:genocide_concentration_modifier.resistance_escalation }
					add_to_temp_variable = { genocide_camp_resistance = genocide_camp_escalation_resistance }
					set_temp_variable = { genocide_camp_escalation_compliance = genocide_escalation }
					multiply_temp_variable = { genocide_camp_escalation_compliance = constant:genocide_concentration_modifier.compliance_escalation }
					add_to_temp_variable = { genocide_camp_compliance = genocide_camp_escalation_compliance }
				}
				if = {
					limit = { has_idea = radical_hardliner_support }
					add_to_temp_variable = { genocide_camp_production = constant:genocide_concentration_modifier.production_hardliner }
					add_to_temp_variable = { genocide_camp_resources = constant:genocide_concentration_modifier.resources_hardliner }
					add_to_temp_variable = { genocide_camp_resistance = constant:genocide_concentration_modifier.resistance_hardliner }
				}
				if = {
					limit = { has_idea = genocide_rail_deportation_priority }
					add_to_temp_variable = { genocide_camp_production = constant:genocide_concentration_modifier.production_rail_priority }
					add_to_temp_variable = { genocide_camp_resources = constant:genocide_concentration_modifier.resources_rail_priority }
				}
				if = {
					limit = { has_idea = germany_auschwitz_experiments }
					add_to_temp_variable = { genocide_camp_production = constant:genocide_concentration_modifier.production_auschwitz }
				}
				if = {
					limit = { has_variable = genocide_camp_administration_strength }
					set_temp_variable = { genocide_camp_admin_value = genocide_camp_administration_strength }
					multiply_temp_variable = { genocide_camp_admin_value = constant:genocide_concentration_modifier.production_admin }
					add_to_temp_variable = { genocide_camp_production = genocide_camp_admin_value }
				}
			}
		}
		clamp_temp_variable = {
			var = genocide_camp_production
			min = 0
			max = constant:genocide_concentration_modifier.production_max
		}
		clamp_temp_variable = {
			var = genocide_camp_resources
			min = 0
			max = constant:genocide_concentration_modifier.resources_max
		}
		clamp_temp_variable = {
			var = genocide_camp_resistance
			min = 0
			max = constant:genocide_concentration_modifier.resistance_max
		}
		clamp_temp_variable = {
			var = genocide_camp_compliance
			min = constant:genocide_concentration_modifier.compliance_min
			max = 0
		}
		set_variable = { genocide_concentration_camp_production = genocide_camp_production }
		set_variable = { genocide_concentration_camp_resources = genocide_camp_resources }
		set_variable = { genocide_concentration_camp_resistance = genocide_camp_resistance }
		set_variable = { genocide_concentration_camp_compliance = genocide_camp_compliance }
		add_dynamic_modifier = {
			modifier = genocide_concentration_camp_network_state
		}
		force_update_dynamic_modifier = yes
	}
}

genocide_calculate_discovery_condemnation_from_prev_state = {
	set_temp_variable = { genocide_condemnation_add = constant:genocide_condemnation.concentration_first }
	if = {
		limit = { PREV = { extermination_camp > 0 } }
		set_temp_variable = { genocide_condemnation_add = constant:genocide_condemnation.extermination_first }
	}
	else_if = {
		limit = { PREV = { gulag_labor_camp_network > 0 } }
		set_temp_variable = { genocide_condemnation_add = constant:genocide_condemnation.gulag_first }
	}
	if = {
		limit = { check_variable = { genocide_discovered_sites > 0 } }
		set_temp_variable = { genocide_condemnation_add = constant:genocide_condemnation.concentration_repeat }
		if = {
			limit = { PREV = { extermination_camp > 0 } }
			set_temp_variable = { genocide_condemnation_add = constant:genocide_condemnation.extermination_repeat }
		}
		else_if = {
			limit = { PREV = { gulag_labor_camp_network > 0 } }
			set_temp_variable = { genocide_condemnation_add = constant:genocide_condemnation.gulag_repeat }
		}
		set_temp_variable = { genocide_repeat_divisor = genocide_discovered_sites }
		add_to_temp_variable = { genocide_repeat_divisor = 1 }
		divide_temp_variable = { genocide_condemnation_add = genocide_repeat_divisor }
		clamp_temp_variable = {
			var = genocide_condemnation_add
			min = constant:genocide_condemnation.repeat_floor
			max = constant:genocide_condemnation.repeat_max
		}
	}
	if = {
		limit = { PREV = { has_state_flag = genocide_evidence_destroyed } }
		multiply_temp_variable = { genocide_condemnation_add = constant:genocide_condemnation.destroyed_evidence_factor }
	}
	if = {
		limit = { PREV = { has_state_flag = genocide_evidence_destroy_failed } }
		multiply_temp_variable = { genocide_condemnation_add = constant:genocide_condemnation.failed_coverup_factor }
	}
	if = {
		limit = {
			PREV = {
				OR = {
					has_state_flag = genocide_auschwitz_experiment_site
					has_state_flag = genocide_ss_laboratory_site
				}
			}
		}
		add_to_temp_variable = { genocide_condemnation_add = constant:genocide_condemnation.experiment_site_bonus }
		if = {
			limit = { has_variable = mengele_autonomy }
			set_temp_variable = { genocide_mengele_autonomy_condemnation = mengele_autonomy }
			multiply_temp_variable = { genocide_mengele_autonomy_condemnation = constant:genocide_condemnation.experiment_autonomy_condemnation_factor }
			add_to_temp_variable = { genocide_condemnation_add = genocide_mengele_autonomy_condemnation }
		}
		if = {
			limit = { has_variable = mengele_permission_level }
			set_temp_variable = { genocide_mengele_permission_condemnation = mengele_permission_level }
			multiply_temp_variable = { genocide_mengele_permission_condemnation = constant:genocide_condemnation.experiment_permission_condemnation_factor }
			add_to_temp_variable = { genocide_condemnation_add = genocide_mengele_permission_condemnation }
		}
	}
	if = {
		limit = { PREV = { has_state_flag = genocide_restricted_chemical_site } }
		add_to_temp_variable = { genocide_condemnation_add = constant:genocide_condemnation.restricted_chemical_site_bonus }
	}
}
