initialize_triggerable_scenarios_settings = {
	if = {
		limit = { NOT = { has_variable = triggerable_scenarios_selected } }
		set_variable = { triggerable_scenarios_selected = constant:triggerable_scenario_id.zombie_apocalypse }
	}
	if = {
		limit = { NOT = { has_variable = triggerable_scenarios_intensity } }
		set_variable = { triggerable_scenarios_intensity = constant:triggerable_scenario_intensity.medium }
	}
	if = {
		limit = { NOT = { has_variable = triggerable_scenarios_zombie_type } }
		set_variable = { triggerable_scenarios_zombie_type = constant:triggerable_scenario_zombie_type.diverse }
	}
	if = {
		limit = { NOT = { has_variable = triggerable_scenarios_clone_type } }
		set_variable = { triggerable_scenarios_clone_type = constant:triggerable_scenario_clone_type.standard }
	}
	if = {
		limit = { NOT = { has_variable = triggerable_scenarios_sort_mode } }
		set_variable = { triggerable_scenarios_sort_mode = constant:triggerable_scenario_sort_mode.id }
	}
	triggerable_scenarios_initialize_registry = yes
	triggerable_scenarios_rebuild_view = yes
}

triggerable_scenarios_initialize_registry = {
	clear_array = global.triggerable_scenario_all_ids
	clear_array = global.triggerable_scenario_all_name_sort_values
	add_to_array = {
		array = global.triggerable_scenario_all_ids
		value = constant:triggerable_scenario_id.zombie_apocalypse
	}
	add_to_array = {
		array = global.triggerable_scenario_all_name_sort_values
		value = constant:triggerable_scenario_sort_value.zombie_name
	}
	add_to_array = {
		array = global.triggerable_scenario_all_ids
		value = constant:triggerable_scenario_id.mengele_clones
	}
	add_to_array = {
		array = global.triggerable_scenario_all_name_sort_values
		value = constant:triggerable_scenario_sort_value.mengele_clones_name
	}
}

triggerable_scenarios_rebuild_view = {
	clear_array = global.triggerable_scenario_view_ids
	clear_array = global.triggerable_scenario_view_name_sort_values
	clear_array = temp_triggerable_scenario_sort_ids
	clear_array = temp_triggerable_scenario_sort_name_values

	for_loop_effect = {
		start = 0
		end = global.triggerable_scenario_all_ids^num
		value = triggerable_scenario_registry_index
		add_to_array = {
			array = temp_triggerable_scenario_sort_ids
			value = global.triggerable_scenario_all_ids^triggerable_scenario_registry_index
		}
		add_to_array = {
			array = temp_triggerable_scenario_sort_name_values
			value = global.triggerable_scenario_all_name_sort_values^triggerable_scenario_registry_index
		}
	}

	set_temp_variable = { triggerable_scenario_source_count = temp_triggerable_scenario_sort_ids^num }
	set_temp_variable = { triggerable_scenario_sorted_count = 0 }
	while_loop_effect = {
		limit = { check_variable = { triggerable_scenario_sorted_count < triggerable_scenario_source_count } }

		set_temp_variable = { triggerable_scenario_best_index = -1 }
		set_temp_variable = { triggerable_scenario_scan_index = 0 }
		while_loop_effect = {
			limit = { check_variable = { triggerable_scenario_scan_index < triggerable_scenario_source_count } }
			if = {
				limit = {
					check_variable = {
						var = temp_triggerable_scenario_sort_ids^triggerable_scenario_scan_index
						value = 0
						compare = greater_than
					}
				}

				set_temp_variable = { triggerable_scenario_candidate_metric = temp_triggerable_scenario_sort_ids^triggerable_scenario_scan_index }
				if = {
					limit = { check_variable = { triggerable_scenarios_sort_mode = constant:triggerable_scenario_sort_mode.name } }
					set_temp_variable = { triggerable_scenario_candidate_metric = temp_triggerable_scenario_sort_name_values^triggerable_scenario_scan_index }
				}

				if = {
					limit = { check_variable = { triggerable_scenario_best_index < 0 } }
					set_temp_variable = { triggerable_scenario_best_index = triggerable_scenario_scan_index }
					set_temp_variable = { triggerable_scenario_best_metric = triggerable_scenario_candidate_metric }
					set_temp_variable = { triggerable_scenario_best_id = temp_triggerable_scenario_sort_ids^triggerable_scenario_scan_index }
				}
				else_if = {
					limit = {
						has_country_flag = triggerable_scenarios_sort_descending
						check_variable = { triggerable_scenario_candidate_metric > triggerable_scenario_best_metric }
					}
					set_temp_variable = { triggerable_scenario_best_index = triggerable_scenario_scan_index }
					set_temp_variable = { triggerable_scenario_best_metric = triggerable_scenario_candidate_metric }
					set_temp_variable = { triggerable_scenario_best_id = temp_triggerable_scenario_sort_ids^triggerable_scenario_scan_index }
				}
				else_if = {
					limit = {
						NOT = { has_country_flag = triggerable_scenarios_sort_descending }
						check_variable = { triggerable_scenario_candidate_metric < triggerable_scenario_best_metric }
					}
					set_temp_variable = { triggerable_scenario_best_index = triggerable_scenario_scan_index }
					set_temp_variable = { triggerable_scenario_best_metric = triggerable_scenario_candidate_metric }
					set_temp_variable = { triggerable_scenario_best_id = temp_triggerable_scenario_sort_ids^triggerable_scenario_scan_index }
				}
				else_if = {
					limit = { check_variable = { triggerable_scenario_candidate_metric = triggerable_scenario_best_metric } }
					if = {
						limit = {
							has_country_flag = triggerable_scenarios_sort_descending
							check_variable = {
								var = temp_triggerable_scenario_sort_ids^triggerable_scenario_scan_index
								value = triggerable_scenario_best_id
								compare = greater_than
							}
						}
						set_temp_variable = { triggerable_scenario_best_index = triggerable_scenario_scan_index }
						set_temp_variable = { triggerable_scenario_best_metric = triggerable_scenario_candidate_metric }
						set_temp_variable = { triggerable_scenario_best_id = temp_triggerable_scenario_sort_ids^triggerable_scenario_scan_index }
					}
					else_if = {
						limit = {
							NOT = { has_country_flag = triggerable_scenarios_sort_descending }
							check_variable = {
								var = temp_triggerable_scenario_sort_ids^triggerable_scenario_scan_index
								value = triggerable_scenario_best_id
								compare = less_than
							}
						}
						set_temp_variable = { triggerable_scenario_best_index = triggerable_scenario_scan_index }
						set_temp_variable = { triggerable_scenario_best_metric = triggerable_scenario_candidate_metric }
						set_temp_variable = { triggerable_scenario_best_id = temp_triggerable_scenario_sort_ids^triggerable_scenario_scan_index }
					}
				}
			}
			add_to_temp_variable = { triggerable_scenario_scan_index = 1 }
		}

		if = {
			limit = { check_variable = { triggerable_scenario_best_index > -1 } }
			add_to_array = {
				array = global.triggerable_scenario_view_ids
				value = temp_triggerable_scenario_sort_ids^triggerable_scenario_best_index
			}
			add_to_array = {
				array = global.triggerable_scenario_view_name_sort_values
				value = temp_triggerable_scenario_sort_name_values^triggerable_scenario_best_index
			}
			set_variable = { temp_triggerable_scenario_sort_ids^triggerable_scenario_best_index = -1 }
			add_to_temp_variable = { triggerable_scenario_sorted_count = 1 }
		}
		else = {
			set_temp_variable = { triggerable_scenario_sorted_count = triggerable_scenario_source_count }
		}
	}
}
