# ============================================================================
# CHAOS REDUX - TRIGGERABLE SCENARIO EFFECTS
# ============================================================================
# Manual sandbox/challenge scenario framework for the Chaos Redux scenario
# window. Scenario launches bypass ordinary event pacing, while explicit
# world-end scenarios still set the same terminal flags as their live branches.

initialize_triggerable_scenarios_settings = {
	if = {
		limit = { NOT = { has_variable = triggerable_scenarios_selected } }
		set_variable = { triggerable_scenarios_selected = constant:triggerable_scenario_id.zombie_apocalypse }
	}
	if = {
		limit = { NOT = { has_variable = triggerable_scenarios_intensity } }
		set_variable = { triggerable_scenarios_intensity = constant:triggerable_scenario_intensity.medium }
	}
	if = {
		limit = { NOT = { has_variable = triggerable_scenarios_zombie_type } }
		set_variable = { triggerable_scenarios_zombie_type = constant:triggerable_scenario_zombie_type.diverse }
	}
	if = {
		limit = { NOT = { has_variable = triggerable_scenarios_clone_type } }
		set_variable = { triggerable_scenarios_clone_type = constant:triggerable_scenario_clone_type.standard }
	}
	if = {

		set_variable = { triggerable_scenarios_soviet_collapse_type = constant:triggerable_scenario_soviet_collapse_type.normal_republics }

		limit = { NOT = { has_variable = triggerable_scenarios_final_silence_type } }
		set_variable = { triggerable_scenarios_final_silence_type = constant:triggerable_scenario_final_silence_type.nuclear }

		set_variable = { triggerable_scenarios_fury_type = constant:fury_scenario_type.pact }

	}
	triggerable_scenarios_initialize_registry = yes
	triggerable_scenarios_rebuild_view = yes
}

triggerable_scenario_sort_previous = {
	initialize_triggerable_scenarios_settings = yes
	if = {
		limit = { check_variable = { triggerable_scenarios_sort_mode = constant:triggerable_scenario_sort_mode.id } }
		set_variable = { triggerable_scenarios_sort_mode = constant:triggerable_scenario_sort_mode.name }

	}
	triggerable_scenarios_rebuild_view = yes
}

triggerable_scenario_toggle_sort_order = {

triggerable_scenario_type_previous = {
	initialize_triggerable_scenarios_settings = yes
	if = {
		limit = { check_variable = { triggerable_scenarios_selected = constant:triggerable_scenario_id.zombie_apocalypse } }
		subtract_from_variable = { triggerable_scenarios_zombie_type = 1 }
		if = {
			limit = { check_variable = { triggerable_scenarios_zombie_type < constant:triggerable_scenario_zombie_type.diverse } }
			set_variable = { triggerable_scenarios_zombie_type = constant:triggerable_scenario_zombie_type.connected }
		}
	}
	else_if = {
		limit = { check_variable = { triggerable_scenarios_selected = constant:triggerable_scenario_id.mengele_clones } }
		subtract_from_variable = { triggerable_scenarios_clone_type = 1 }
		if = {
			limit = { check_variable = { triggerable_scenarios_clone_type < constant:triggerable_scenario_clone_type.standard } }
			set_variable = { triggerable_scenarios_clone_type = constant:triggerable_scenario_clone_type.aryan }
		}
	}
	else_if = {
		limit = { check_variable = { triggerable_scenarios_selected = constant:triggerable_scenario_id.soviet_collapse } }
		subtract_from_variable = { triggerable_scenarios_soviet_collapse_type = 1 }
		if = {
			limit = { check_variable = { triggerable_scenarios_soviet_collapse_type < constant:triggerable_scenario_soviet_collapse_type.normal_republics } }
			set_variable = { triggerable_scenarios_soviet_collapse_type = constant:triggerable_scenario_soviet_collapse_type.chaos_republics }
		}
	}
	else_if = {
		limit = { check_variable = { triggerable_scenarios_selected = constant:triggerable_scenario_id.world_in_fury } }
		subtract_from_variable = { triggerable_scenarios_fury_type = 1 }
		if = {
			limit = { check_variable = { triggerable_scenarios_fury_type < constant:fury_scenario_type.pact } }
			set_variable = { triggerable_scenarios_fury_type = constant:fury_scenario_type.hostile }
		}
	}
	else = {
		subtract_from_variable = { triggerable_scenarios_final_silence_type = 1 }
		if = {
			limit = { check_variable = { triggerable_scenarios_final_silence_type < constant:triggerable_scenario_final_silence_type.nuclear } }
			set_variable = { triggerable_scenarios_final_silence_type = constant:triggerable_scenario_final_silence_type.thermonuclear }
		}
	}
}

trigger_selected_chaosx_scenario = {
	initialize_triggerable_scenarios_settings = yes
	set_temp_variable = { triggerable_scenario_launch_id = triggerable_scenarios_selected }
	if = {
		limit = { has_variable = event_timer_days }
		set_variable = { triggerable_scenario_preserved_event_timer_days = event_timer_days }
		set_country_flag = triggerable_scenario_preserved_event_timer
	}
	if = {
		limit = {
			check_variable = {
				var = triggerable_scenario_launch_id
				value = constant:triggerable_scenario_id.zombie_apocalypse
				compare = equals
			}
		}
		trigger_zombie_apocalypse_scenario = yes
	}
	else_if = {
		limit = {
			check_variable = {
				var = triggerable_scenario_launch_id
				value = constant:triggerable_scenario_id.mengele_clones
				compare = equals
			}
		}
		trigger_mengele_clone_army_scenario = yes
	}
	else_if = {
		limit = {
			check_variable = {
				var = triggerable_scenario_launch_id
				value = constant:triggerable_scenario_id.soviet_collapse
				compare = equals
			}
		}
		trigger_soviet_collapse_scenario = yes
	}
	else_if = {
		limit = {
			check_variable = {
				var = triggerable_scenario_launch_id
				value = constant:triggerable_scenario_id.final_silence
				compare = equals
			}
		}
		trigger_final_silence_scenario = yes
	}
	else_if = {
		limit = {
			check_variable = {
				var = triggerable_scenario_launch_id
				value = constant:triggerable_scenario_id.world_in_fury
				compare = equals
			}
		}
		trigger_world_in_fury_scenario = yes
	}
	if = {
		limit = { has_country_flag = triggerable_scenario_preserved_event_timer }
		set_variable = { event_timer_days = triggerable_scenario_preserved_event_timer_days }
		clear_variable = triggerable_scenario_preserved_event_timer_days
		clr_country_flag = triggerable_scenario_preserved_event_timer
	}
}

triggerable_scenario_resolve_zombie_neighbor_relations = {
	save_event_target_as = triggerable_zombie_relation_actor
	every_neighbor_country = {
		limit = {
			is_triggerable_scenario_zombie_country = yes
		}
		save_event_target_as = triggerable_zombie_relation_neighbor
		if = {
			limit = {
				triggerable_scenario_zombie_same_type_as_actor = yes
			}
			event_target:triggerable_zombie_relation_actor = {
				set_variable = { triggerable_zombie_relation_divisions = num_divisions }
			}
			set_variable = { triggerable_zombie_relation_divisions = num_divisions }
			if = {
				limit = {
					check_variable = {
						var = triggerable_zombie_relation_divisions
						value = event_target:triggerable_zombie_relation_actor.triggerable_zombie_relation_divisions
						compare = greater_than
					}
				}
				annex_country = {
					target = event_target:triggerable_zombie_relation_actor
					transfer_troops = yes
				}
			}
			else = {
				event_target:triggerable_zombie_relation_actor = {
					annex_country = {
						target = event_target:triggerable_zombie_relation_neighbor
						transfer_troops = yes
					}
				}
			}
		}
		else_if = {
			limit = {
				event_target:triggerable_zombie_relation_actor = {
					NOT = { has_war_with = event_target:triggerable_zombie_relation_neighbor }
					can_declare_war_on = event_target:triggerable_zombie_relation_neighbor
				}
			}
			event_target:triggerable_zombie_relation_actor = {
				declare_war_on = {
					target = event_target:triggerable_zombie_relation_neighbor
					type = annex_everything
				}
			}
		}
	}
}
