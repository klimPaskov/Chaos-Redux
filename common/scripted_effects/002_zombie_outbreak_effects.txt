# Calculate total zombie strength for threat checks shared across decisions and on_actions.
calculate_total_zombie_strength = {
    set_temp_variable = { total_zombie_divisions = 0 }
    set_temp_variable = { total_zombie_controlled_states = 0 }

    if = {
        limit = { country_exists = ZZZ }
        ZZZ = {
            set_temp_variable = { zombie_strength_divisions = num_divisions }
            PREV = { add_to_temp_variable = { var = total_zombie_divisions value = zombie_strength_divisions } }
            set_temp_variable = { zombie_strength_states = num_of_controlled_states }
            PREV = { add_to_temp_variable = { var = total_zombie_controlled_states value = zombie_strength_states } }
        }
    }

    every_country = {
        limit = {
            exists = yes
            has_country_flag = zombie_outbreak_dynamic_country
        }
        set_temp_variable = { zombie_strength_divisions = num_divisions }
        PREV = { add_to_temp_variable = { var = total_zombie_divisions value = zombie_strength_divisions } }
        set_temp_variable = { zombie_strength_states = num_of_controlled_states }
        PREV = { add_to_temp_variable = { var = total_zombie_controlled_states value = zombie_strength_states } }
    }
}

zombie_on_state_captured_by_outbreak = {
    set_temp_variable = { zombie_state_decay_current_day = global.num_days }

    if = {
        limit = { NOT = { has_variable = zombie_state_population_decay_months_done } }
        set_variable = { zombie_state_population_decay_months_done = 0 }
    }

    set_variable = { zombie_state_next_population_decay_day = zombie_state_decay_current_day }
    add_to_variable = {
        var = zombie_state_next_population_decay_day
        value = constant:zombie_state_decay.monthly_population_interval_days
    }

    set_variable = { zombie_state_next_degradation_day = zombie_state_decay_current_day }
    add_to_variable = {
        var = zombie_state_next_degradation_day
        value = constant:zombie_state_decay.degradation_interval_days
    }
}

}

zombie_set_state_population_decay_profile = {
	set_temp_variable = { chaos_state_deaths_percent = constant:zombie_state_decay.monthly_population_percent_base }

	if = {
		limit = { ROOT = { has_country_flag = { flag = zombie_current_tier value = 3 } } }
		multiply_temp_variable = { chaos_state_deaths_percent = constant:zombie_state_decay.monthly_population_stage_factor_3 }
	}
	else_if = {
		limit = { ROOT = { has_country_flag = { flag = zombie_current_tier value = 2 } } }
		multiply_temp_variable = { chaos_state_deaths_percent = constant:zombie_state_decay.monthly_population_stage_factor_2 }
	}
	else_if = {
		limit = { ROOT = { has_country_flag = { flag = zombie_current_tier value = 1 } } }
		multiply_temp_variable = { chaos_state_deaths_percent = constant:zombie_state_decay.monthly_population_stage_factor_1 }
	}
	else = {
		multiply_temp_variable = { chaos_state_deaths_percent = constant:zombie_state_decay.monthly_population_stage_factor_0 }
	}
}

zombie_apply_state_population_decay = {
    if = {
        limit = { check_variable = { state_population_k > 0 } }
        zombie_set_state_population_decay_profile = yes
        set_temp_variable = { chaos_state_deaths_cap_factor = constant:zombie_state_decay.monthly_population_cap_factor }
        set_temp_variable = { chaos_state_deaths_random_min = constant:zombie_state_decay.monthly_population_random_min }
        set_temp_variable = { chaos_state_deaths_random_span = constant:zombie_state_decay.monthly_population_random_span }
        set_temp_variable = { chaos_deaths_reason = constant:chaos_meter_deaths_reason.zombie_outbreak_decay }
        chaos_meter_register_state_civilian_deaths_percent = yes
        add_to_variable = { zombie_state_population_decay_months_done = 1 }
    }
}

zombie_process_controlled_state_decay = {
    every_controlled_state = {
        zombie_schedule_state_decay_if_missing = yes
        set_temp_variable = { zombie_state_decay_current_day = global.num_days }

        if = {
            limit = {
                check_variable = {
                    var = zombie_state_population_decay_months_done
                    value = constant:zombie_state_decay.monthly_population_max_months
                    compare = less_than
                }
                check_variable = {
                    var = zombie_state_next_population_decay_day
                    value = zombie_state_decay_current_day
                    compare = less_than_or_equals
                }
            }
            zombie_apply_state_population_decay = yes
            add_to_variable = {
                var = zombie_state_next_population_decay_day
                value = constant:zombie_state_decay.monthly_population_interval_days
            }
        }

        if = {
            limit = {
                check_variable = {
                    var = zombie_state_next_degradation_day
                    value = zombie_state_decay_current_day
                    compare = less_than_or_equals
                }
            }
            zombie_apply_state_degradation = yes
            add_to_variable = {
                var = zombie_state_next_degradation_day
                value = constant:zombie_state_decay.degradation_interval_days
            }
        }
    }
}

join_anti_zombie_league_effect = {
    every_country = {
        limit = {
            has_war_with = ROOT
            NOT = { original_tag = ZZZ }
            NOT = { has_country_flag = zombie_outbreak_dynamic_country }
            NOT = { has_country_flag = weaponized_zombie_outbreak_country }
        }
        ROOT = {
            white_peace = PREV
        }
    }

    if = {
        limit = { is_in_faction = yes }
        leave_faction = yes
    }

	prepare_anti_zombie_league_member = yes
	clr_country_flag = azl_left_due_distant_threat
	clr_country_flag = azl_distant_threat_reviewed
	clr_country_flag = azl_left_after_main_outbreak_gone
	clr_country_flag = azl_can_join_later

    if = {
        limit = {
            has_event_target = azl_faction_leader
            event_target:azl_faction_leader = {
                has_country_flag = azl_leader
                is_faction_leader = yes
            }
        }
        event_target:azl_faction_leader = {
            add_to_faction = ROOT
        }
    }
    else = {
        random_country = {
            limit = {
                has_country_flag = azl_leader
                is_faction_leader = yes
            }
            add_to_faction = PREV
        }
    }

    if = {
        limit = { is_in_anti_zombie_league_faction = yes }
        grant_anti_zombie_league_membership_benefits = yes
    }
    else = {
        remove_anti_zombie_league_membership_package = yes
    }
}

calculate_zombie_continent_presence = {
    set_temp_variable = { zombie_continent_europe = 0 }
    set_temp_variable = { zombie_continent_africa = 0 }
    set_temp_variable = { zombie_continent_asia = 0 }
    set_temp_variable = { zombie_controls_continent = 0 }

    if = {
        limit = { country_exists = ZZZ }
        ZZZ = {
            every_controlled_state = {
                if = {
                    limit = { is_on_continent = europe }
                    ROOT = { add_to_temp_variable = { zombie_continent_europe = 1 } }
                }
                if = {
                    limit = { is_on_continent = africa }
                    ROOT = { add_to_temp_variable = { zombie_continent_africa = 1 } }
                }
                if = {
                    limit = { is_on_continent = asia }
                    ROOT = { add_to_temp_variable = { zombie_continent_asia = 1 } }
                }
            }
        }
    }

    every_country = {
        limit = {
            exists = yes
            has_country_flag = zombie_outbreak_dynamic_country
        }
        every_controlled_state = {
            if = {
                limit = { is_on_continent = europe }
                ROOT = { add_to_temp_variable = { zombie_continent_europe = 1 } }
            }
            if = {
                limit = { is_on_continent = africa }
                ROOT = { add_to_temp_variable = { zombie_continent_africa = 1 } }
            }
            if = {
                limit = { is_on_continent = asia }
                ROOT = { add_to_temp_variable = { zombie_continent_asia = 1 } }
            }
        }
    }

    if = {
        limit = {
            OR = {
                check_variable = { zombie_continent_europe > constant:anti_zombie_league.emergency_europe_states }
                check_variable = { zombie_continent_africa > constant:anti_zombie_league.emergency_africa_states }
                check_variable = { zombie_continent_asia > constant:anti_zombie_league.emergency_asia_states }
            }
        }
        set_temp_variable = { zombie_controls_continent = 1 }
    }
}

apply_zombie_evacuation_transfer = {
    calculate_zombie_evacuation_request = yes

    if = {
        limit = {
            has_event_target = refugee_host_country
            check_variable = {
                var = zombie_evacuation_population_to_move
                value = 0
                compare = greater_than
            }
            check_variable = {
                var = zombie_evacuation_population_pool
                value = 0
                compare = greater_than
            }
        }

        set_temp_variable = { zombie_evacuation_removed_population = 0 }

        every_controlled_state = {
            set_temp_variable = { state_evacuation_population = state_population_k }
            multiply_temp_variable = { state_evacuation_population = 1000 }
            multiply_temp_variable = { state_evacuation_population = PREV.zombie_evacuation_population_to_move }
            divide_temp_variable = { state_evacuation_population = PREV.zombie_evacuation_population_pool }
            round_temp_variable = state_evacuation_population

            if = {
                limit = {
                    check_variable = {
                        var = state_evacuation_population
                        value = 0
                        compare = greater_than
                    }
                }
                multiply_temp_variable = { state_evacuation_population = -1 }
                add_manpower = state_evacuation_population
                multiply_temp_variable = { state_evacuation_population = -1 }
                PREV = {
                    add_to_temp_variable = {
                        var = zombie_evacuation_removed_population
                        value = state_evacuation_population
                    }
                }
            }
        }

        if = {
            limit = {
                check_variable = {
                    var = zombie_evacuation_removed_population
                    value = 0
                    compare = greater_than
                }
            }
            event_target:refugee_host_country = {
                capital_scope = {
                    add_manpower = zombie_evacuation_removed_population
                }
            }
        }
    }

    clr_country_flag = zombie_evacuation_request_pending
    set_country_flag = {
        flag = zombie_evacuation_cooldown
        days = @ZOMBIE_EVACUATION_RETRY_COOLDOWN_DAYS

            add_to_temp_variable = { zombie_dynamic_outbreak_state_weight = constant:zombie_outbreak.state_pick_high_population_bonus }

            limit = { state_population_k > constant:zombie_outbreak.state_population_extreme_risk_threshold }
            add_to_temp_variable = { zombie_dynamic_outbreak_state_weight = constant:zombie_outbreak.state_pick_extreme_population_bonus }

            limit = { infrastructure < constant:zombie_outbreak.low_infrastructure_threshold }

        }

            limit = { infrastructure < constant:zombie_outbreak.very_low_infrastructure_threshold }
            add_to_temp_variable = { zombie_dynamic_outbreak_state_weight = constant:zombie_outbreak.state_pick_very_low_infrastructure_bonus }
        }
        if = {
            limit = {
                industrial_complex < constant:zombie_outbreak.low_industry_threshold
                arms_factory < constant:zombie_outbreak.low_industry_threshold
            }
            add_to_temp_variable = { zombie_dynamic_outbreak_state_weight = constant:zombie_outbreak.state_pick_low_industry_bonus }
        }
        if = {
            limit = {
                industrial_complex < constant:zombie_outbreak.very_low_industry_threshold
                arms_factory < constant:zombie_outbreak.very_low_industry_threshold
            }
            add_to_temp_variable = { zombie_dynamic_outbreak_state_weight = constant:zombie_outbreak.state_pick_very_low_industry_bonus }
        }

        round_temp_variable = zombie_dynamic_outbreak_state_weight

        if = {
            limit = {
                check_variable = {
                    var = zombie_dynamic_outbreak_state_weight
                    value = 1
                    compare = less_than
                }
            }
            set_temp_variable = { zombie_dynamic_outbreak_state_weight = 1 }
        }
        if = {
            limit = {
                check_variable = {
                    var = zombie_dynamic_outbreak_state_weight
                    value = constant:zombie_outbreak.state_pick_weight_cap
                    compare = greater_than
                }
            }
            set_temp_variable = { zombie_dynamic_outbreak_state_weight = constant:zombie_outbreak.state_pick_weight_cap }
        }

        for_loop_effect = {
            start = 0
            end = zombie_dynamic_outbreak_state_weight
            value = zombie_dynamic_outbreak_state_pool_index

            add_to_array = {
                array = global.dynamic_zombie_outbreak_state_pool
                value = THIS
            }
        }
    }

        limit = {
            check_variable = {
                var = global.dynamic_zombie_outbreak_state_pool^num
                value = 0
                compare = greater_than
            }
        }
        random_scope_in_array = {
            array = global.dynamic_zombie_outbreak_state_pool

attempt_dynamic_zombie_outbreak = {
    if = {
        limit = {
            has_global_flag = has_zombie_outbreak
            NOT = { has_global_flag = zombie_system_disabled }
            NOT = { has_global_flag = zombie_dynamic_outbreak_global_cooldown }
            any_country = {
                can_roll_dynamic_zombie_outbreak = yes
            }
        }
        set_temp_variable = { zombie_dynamic_outbreak_roll_yes = constant:zombie_outbreak.weekly_roll_days }
        set_temp_variable = { zombie_dynamic_outbreak_roll_no = global.best_dynamic_zombie_outbreak_days }
        subtract_from_temp_variable = { zombie_dynamic_outbreak_roll_no = zombie_dynamic_outbreak_roll_yes }

        if = {
            limit = {
                check_variable = {
                    var = zombie_dynamic_outbreak_roll_no
                    value = 1
                    compare = less_than
                }
            }
            set_temp_variable = { zombie_dynamic_outbreak_roll_no = 1 }
        }

        random_list = {
            zombie_dynamic_outbreak_roll_no = { }
            zombie_dynamic_outbreak_roll_yes = { fire_weighted_dynamic_zombie_outbreak_candidate = yes }
        }
    }
}

# Calculate zombie outbreak chance for a country (used for outbreak UI and weekly target selection)
calculate_zombie_outbreak_chance = {
    set_variable = { zombie_outbreak_days = mtth:zombie_outbreak_chance }

    if = {
        limit = { can_roll_dynamic_zombie_outbreak = yes }
        # Convert to months for display (approximate)
        set_variable = { zombie_outbreak_months = zombie_outbreak_days }
        divide_variable = { zombie_outbreak_months = 30 } # Average days per month
        round_variable = zombie_outbreak_months

        # Ensure minimum of 1 month for display
        if = {
            limit = { check_variable = { zombie_outbreak_months < 1 } }
            set_variable = { zombie_outbreak_months = 1 }
        }
    }
    else = {
        # ZZZ country or countries with no states get no risk
        set_zombie_outbreak_no_risk = yes
    }
}
# Update zombie outbreak chance for all countries (weekly effect)
update_all_zombie_outbreak_chances = {
    set_variable = { global.best_dynamic_zombie_outbreak_days = constant:zombie_outbreak.no_risk_days }

    every_country = {
        limit = {
            exists = yes
            NOT = { original_tag = ZZZ }
        }
        calculate_zombie_outbreak_chance = yes

        if = {
            limit = {
                can_roll_dynamic_zombie_outbreak = yes
                check_variable = {
                    var = zombie_outbreak_days
                    value = global.best_dynamic_zombie_outbreak_days
                    compare = less_than
                }
            }
            set_variable = { global.best_dynamic_zombie_outbreak_days = zombie_outbreak_days }
        }
    }
}

            limit = { check_variable = { global.target_level > 11 } }

            multiply_variable = { cure_research_days_1 = constant:anti_zombie_league.cure_days_factor_high_investment }
            multiply_variable = { cure_research_days_2 = constant:anti_zombie_league.cure_days_factor_high_investment }

    }

# Dynamic cure research value calculation effects
# Simple system: cure value = research slots + biowarfare facilities

# Calculate the cure research value based on research slots and facilities
calculate_cure_research_value = {
    # Start with research slots
    set_temp_variable = { cure_value = amount_research_slots }

    # Count biowarfare facilities
    set_temp_variable = { facility_count = 0 }
    every_owned_state = {
        if = {
            limit = { biowarfare_facility > 0 }
            ROOT = { add_to_temp_variable = { facility_count = 1 } }
        }
    }

    # Add facilities to cure value
    add_to_temp_variable = { cure_value = facility_count }

    # Store the result
    set_variable = { cure_research_value = cure_value }
}

# Update all countries' cure research values
update_all_cure_research_values = {
    every_country = {
        limit = {
            NOT = { original_tag = ZZZ }
            NOT = { has_country_flag = zombie_outbreak_dynamic_country }
        }
        calculate_cure_research_value = yes
    }
}
