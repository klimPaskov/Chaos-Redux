sanitize_events_log_history_actor_entries = {
    if = {
        limit = {
            check_variable = {
                var = global.events_log_history_event_id_entries^num
                value = global.events_log_history_actor_entries^num
                compare = greater_than
            }
        }
        set_temp_variable = { missing_actor_entries = global.events_log_history_event_id_entries^num }
        subtract_from_temp_variable = { missing_actor_entries = global.events_log_history_actor_entries^num }

        while_loop_effect = {
            limit = { check_variable = { missing_actor_entries > 0 } }
            add_to_array = {
                array = global.events_log_history_actor_entries
                value = 0
            }
            add_to_array = {
                array = global.events_log_history_has_actor_entries
                value = 0
            }
            subtract_from_temp_variable = { missing_actor_entries = 1 }
        }
    }

    if = {
        limit = {
            check_variable = {
                var = global.events_log_history_event_id_entries^num
                value = global.events_log_history_has_actor_entries^num
                compare = greater_than
            }
        }
        set_temp_variable = { missing_has_actor_entries = global.events_log_history_event_id_entries^num }
        subtract_from_temp_variable = { missing_has_actor_entries = global.events_log_history_has_actor_entries^num }

        while_loop_effect = {
            limit = { check_variable = { missing_has_actor_entries > 0 } }
            add_to_array = {
                array = global.events_log_history_has_actor_entries
                value = 0
            }
            subtract_from_temp_variable = { missing_has_actor_entries = 1 }
        }
    }

    set_temp_variable = { events_log_sanitize_index = 0 }
    while_loop_effect = {
        limit = {
            check_variable = {
                var = events_log_sanitize_index
                value = global.events_log_history_event_id_entries^num
                compare = less_than
            }
        }

        if = {
            limit = {
                check_variable = {
                    var = global.events_log_history_has_actor_entries^events_log_sanitize_index
                    value = 1
                    compare = less_than
                }
            }
            set_variable = { global.events_log_history_actor_entries^events_log_sanitize_index = 0 }
            set_variable = { global.events_log_history_has_actor_entries^events_log_sanitize_index = 0 }
        }

        if = {
            limit = {
                check_variable = {
                    var = global.events_log_history_has_actor_entries^events_log_sanitize_index
                    value = 0
                    compare = greater_than
                }
                check_variable = {
                    var = global.events_log_history_actor_entries^events_log_sanitize_index
                    value = 1
                    compare = less_than
                }
            }
            set_variable = { global.events_log_history_actor_entries^events_log_sanitize_index = 0 }
            set_variable = { global.events_log_history_has_actor_entries^events_log_sanitize_index = 0 }
        }

        add_to_temp_variable = { events_log_sanitize_index = 1 }
    }
}

record_events_log_history_entry = {
    if = {
        limit = { NOT = { has_variable = global.events_log_history_sequence } }
        set_variable = { global.events_log_history_sequence = 0 }
    }

    sanitize_events_log_history_actor_entries = yes

    set_temp_variable = { events_log_entry_event_id = global.last_fired_event_id }
    set_temp_variable = { event_id = events_log_entry_event_id }
    get_event_type = yes

    set_temp_variable = { total_unfired_events = global.unfired_major_events_count }
    add_to_temp_variable = { total_unfired_events = global.unfired_fire_once_events_count }
    add_to_temp_variable = { total_unfired_events = global.unfired_repeatable_events_count }

    add_to_variable = { global.events_log_history_sequence = 1 }

    add_to_array = {
        array = global.events_log_history_sequence_entries
        value = global.events_log_history_sequence
        index = 0
    }
    add_to_array = {
        array = global.events_log_history_date_entries
        value = global.date
        index = 0
    }
    add_to_array = {
        array = global.events_log_history_event_id_entries
        value = event_id
        index = 0
    }
    add_to_array = {
        array = global.events_log_history_event_type_entries
        value = event_type
        index = 0
    }
    events_log_set_default_actor_for_current_event = yes
    add_to_array = {
        array = global.events_log_history_actor_entries
        value = events_log_default_actor
        index = 0
    }
    add_to_array = {
        array = global.events_log_history_has_actor_entries
        value = events_log_default_has_actor
        index = 0
    }
    add_to_array = {
        array = global.events_log_history_minor_since_major_entries
        value = global.minor_events_since_major
        index = 0
    }
    add_to_array = {
        array = global.events_log_history_unique_left_entries
        value = total_unfired_events
        index = 0
    }
    add_to_array = {
        array = global.events_log_history_total_fired_entries
        value = global.total_events_fired
        index = 0
    }

    every_country = {
        limit = {
            is_ai = no
            has_country_flag = events_log_window_open
        }
        if = {
            limit = { has_country_flag = events_log_history_tab }
            rebuild_events_log_history_view = yes
        }
        if = {
            limit = { has_country_flag = events_log_events_tab }
            rebuild_events_log_events_view = yes
        }
        if = {
            limit = { has_country_flag = events_log_event_details_open }
            events_log_rebuild_open_event_details_view = yes
        }
    }
}

record_events_log_evolution_entry = {
    if = {
        limit = { NOT = { has_variable = global.events_log_evolution_sequence } }
        set_variable = { global.events_log_evolution_sequence = 0 }
    }

    if = {
        limit = { NOT = { has_variable = events_log_evolution_type } }
        set_temp_variable = { events_log_evolution_type = 0 }
    }
    if = {
        limit = { NOT = { has_variable = events_log_evolution_stage } }
        set_temp_variable = { events_log_evolution_stage = 1 }
    }
    set_temp_variable = { events_log_evolution_event_id_value = 0 }
    if = {
        limit = { NOT = { has_variable = events_log_evolution_tier } }
        set_temp_variable = { events_log_evolution_tier = 0 }
    }
    if = {
        limit = { NOT = { has_variable = events_log_evolution_has_actor } }
        set_temp_variable = { events_log_evolution_has_actor = 0 }
    }

    set_temp_variable = { events_log_evolution_event_type_value = constant:event_system_event_type.unknown }
    if = {
        limit = { has_variable = events_log_evolution_event_id }
        set_temp_variable = { events_log_evolution_event_id_value = events_log_evolution_event_id }
        set_temp_variable = { event_id = events_log_evolution_event_id }
        get_event_type = yes
        set_temp_variable = { events_log_evolution_event_type_value = event_type }
    }
    else_if = {
        limit = { has_variable = events_log_evolution_event_type }
        set_temp_variable = { events_log_evolution_event_type_value = events_log_evolution_event_type }
    }

    sanitize_events_log_evolution_actor_entries = yes

    add_to_variable = { global.events_log_evolution_sequence = 1 }

    add_to_array = {
        array = global.events_log_evolution_sequence_entries
        value = global.events_log_evolution_sequence
        index = 0
    }
    add_to_array = {
        array = global.events_log_evolution_date_entries
        value = global.date
        index = 0
    }
    add_to_array = {
        array = global.events_log_evolution_type_entries
        value = events_log_evolution_type
        index = 0
    }
    add_to_array = {
        array = global.events_log_evolution_event_id_entries
        value = events_log_evolution_event_id_value
        index = 0
    }
    add_to_array = {
        array = global.events_log_evolution_event_type_entries
        value = events_log_evolution_event_type_value
        index = 0
    }
    add_to_array = {
        array = global.events_log_evolution_tier_entries
        value = events_log_evolution_tier
        index = 0
    }
    add_to_array = {
        array = global.events_log_evolution_stage_entries
        value = events_log_evolution_stage
        index = 0
    }

    if = {
        limit = {
            has_event_target = events_log_evolution_actor
            check_variable = {
                var = events_log_evolution_has_actor
                value = 0
                compare = greater_than
            }
        }
        event_target:events_log_evolution_actor = {
            add_to_array = {
                array = global.events_log_evolution_actor_entries
                value = THIS
                index = 0
            }
        }
        add_to_array = {
            array = global.events_log_evolution_has_actor_entries
            value = 1
            index = 0
        }
    }
    else = {
        add_to_array = {
            array = global.events_log_evolution_actor_entries
            value = 0
            index = 0
        }
        add_to_array = {
            array = global.events_log_evolution_has_actor_entries
            value = 0
            index = 0
        }
    }

    set_temp_variable = { events_log_evolution_has_actor = 0 }

    every_country = {
        limit = {
            is_ai = no
            has_country_flag = events_log_window_open
            has_country_flag = events_log_evolutions_tab
        }
        rebuild_events_log_evolution_view = yes
    }
}

events_log_history_append_rows_by_actor_sort = {
    clear_array = temp_events_log_history_unique_actors

    # Build unique actor list from rows with actors.
    set_temp_variable = { events_log_actor_collect_index = 0 }
    while_loop_effect = {
        limit = { check_variable = { events_log_actor_collect_index < events_log_source_count } }
        set_temp_variable = { events_log_source_index = events_log_actor_collect_index }
        if = {
            limit = {
                check_variable = {
                    var = global.events_log_history_has_actor_entries^events_log_source_index
                    value = 0
                    compare = greater_than
                }
                NOT = {
                    is_in_array = {
                        temp_events_log_history_unique_actors = global.events_log_history_actor_entries^events_log_source_index
                    }
                }
            }
            add_to_array = {
                array = temp_events_log_history_unique_actors
                value = global.events_log_history_actor_entries^events_log_source_index
            }
        }
        add_to_temp_variable = { events_log_actor_collect_index = 1 }
    }

    # Deterministic actor-id sort passes.
    while_loop_effect = {
        limit = { check_variable = { temp_events_log_history_unique_actors^num > 0 } }

        if = {
            limit = { has_country_flag = events_log_sort_descending }
            find_highest_in_array = {
                array = temp_events_log_history_unique_actors
                value = events_log_actor_sort_target
            }
        }
        else = {
            find_lowest_in_array = {
                array = temp_events_log_history_unique_actors
                value = events_log_actor_sort_target
            }
        }

        if = {
            limit = { has_country_flag = events_log_sort_descending }
            set_temp_variable = { events_log_actor_member_index = 0 }
            while_loop_effect = {
                limit = { check_variable = { events_log_actor_member_index < events_log_source_count } }
                set_temp_variable = { events_log_source_index = events_log_actor_member_index }
                if = {
                    limit = {
                        check_variable = {
                            var = global.events_log_history_has_actor_entries^events_log_source_index
                            value = 0
                            compare = greater_than
                        }
                        check_variable = {
                            var = global.events_log_history_actor_entries^events_log_source_index
                            value = events_log_actor_sort_target
                            compare = equals
                        }
                    }
                    events_log_row_matches_active_filter = yes
                    if = {
                        limit = { check_variable = { events_log_row_matches_filter > 0 } }
                        events_log_append_view_row_from_source = yes
                    }
                }
                add_to_temp_variable = { events_log_actor_member_index = 1 }
            }
        }
        else = {
            set_temp_variable = { events_log_actor_member_index = events_log_source_count }
            while_loop_effect = {
                limit = { check_variable = { events_log_actor_member_index > 0 } }
                subtract_from_temp_variable = { events_log_actor_member_index = 1 }
                set_temp_variable = { events_log_source_index = events_log_actor_member_index }
                if = {
                    limit = {
                        check_variable = {
                            var = global.events_log_history_has_actor_entries^events_log_source_index
                            value = 0
                            compare = greater_than
                        }
                        check_variable = {
                            var = global.events_log_history_actor_entries^events_log_source_index
                            value = events_log_actor_sort_target
                            compare = equals
                        }
                    }
                    events_log_row_matches_active_filter = yes
                    if = {
                        limit = { check_variable = { events_log_row_matches_filter > 0 } }
                        events_log_append_view_row_from_source = yes
                    }
                }
            }
        }

        remove_from_array = {
            array = temp_events_log_history_unique_actors
            value = events_log_actor_sort_target
        }
    }

    # Non-actor rows are always appended after actor rows.
    if = {
        limit = { has_country_flag = events_log_sort_descending }
        set_temp_variable = { events_log_source_index = 0 }
        while_loop_effect = {
            limit = { check_variable = { events_log_source_index < events_log_source_count } }
            if = {
                limit = {
                    check_variable = {
                        var = global.events_log_history_has_actor_entries^events_log_source_index
                        value = 1
                        compare = less_than
                    }
                }
                events_log_row_matches_active_filter = yes
                if = {
                    limit = { check_variable = { events_log_row_matches_filter > 0 } }
                    events_log_append_view_row_from_source = yes
                }
            }
            add_to_temp_variable = { events_log_source_index = 1 }
        }
    }
    else = {
        set_temp_variable = { events_log_source_index = events_log_source_count }
        while_loop_effect = {
            limit = { check_variable = { events_log_source_index > 0 } }
            subtract_from_temp_variable = { events_log_source_index = 1 }
            if = {
                limit = {
                    check_variable = {
                        var = global.events_log_history_has_actor_entries^events_log_source_index
                        value = 1
                        compare = less_than
                    }
                }
                events_log_row_matches_active_filter = yes
                if = {
                    limit = { check_variable = { events_log_row_matches_filter > 0 } }
                    events_log_append_view_row_from_source = yes
                }
            }
        }
    }
}

rebuild_events_log_history_view = {
    sanitize_events_log_history_actor_entries = yes

    if = {
        limit = { NOT = { has_variable = settings_events_log_sort_mode } }
        set_variable = { settings_events_log_sort_mode = 0 }
    }
    if = {
        limit = { check_variable = { settings_events_log_sort_mode > 2 } }
        set_variable = { settings_events_log_sort_mode = 0 }
    }
    if = {
        limit = { check_variable = { settings_events_log_sort_mode < 0 } }
        set_variable = { settings_events_log_sort_mode = 0 }
    }
    if = {
        limit = { NOT = { has_variable = settings_events_log_filter_type } }
        set_variable = { settings_events_log_filter_type = 0 }
    }
    if = {
        limit = { check_variable = { settings_events_log_filter_type > 3 } }
        set_variable = { settings_events_log_filter_type = 0 }
    }
    if = {
        limit = { check_variable = { settings_events_log_filter_type < 0 } }
        set_variable = { settings_events_log_filter_type = 0 }
    }

    clear_array = global.events_log_view_sequence_entries
    clear_array = global.events_log_view_date_entries
    clear_array = global.events_log_view_event_id_entries
    clear_array = global.events_log_view_event_type_entries
    clear_array = global.events_log_view_actor_entries
    clear_array = global.events_log_view_has_actor_entries
    clear_array = global.events_log_view_cluster_id_entries
    clear_array = global.events_log_view_cluster_sequence_entries
    clear_array = global.events_log_view_cluster_type_entries

    set_temp_variable = { events_log_source_count = global.events_log_history_event_id_entries^num }

    if = {
        limit = { check_variable = { events_log_source_count > 0 } }
        if = {
            limit = { check_variable = { settings_events_log_sort_mode = 0 } }
            if = {
                limit = { has_country_flag = events_log_sort_descending }
                for_loop_effect = {
                    start = 0
                    end = events_log_source_count
                    value = i
                    set_temp_variable = { events_log_source_index = i }
                    events_log_row_matches_active_filter = yes
                    if = {
                        limit = { check_variable = { events_log_row_matches_filter > 0 } }
                        events_log_append_view_row_from_source = yes
                    }
                }
            }
            else = {
                set_temp_variable = { events_log_asc_index = events_log_source_count }
                while_loop_effect = {
                    limit = { check_variable = { events_log_asc_index > 0 } }
                    subtract_from_temp_variable = { events_log_asc_index = 1 }
                    set_temp_variable = { events_log_source_index = events_log_asc_index }
                    events_log_row_matches_active_filter = yes
                    if = {
                        limit = { check_variable = { events_log_row_matches_filter > 0 } }
                        events_log_append_view_row_from_source = yes
                    }
                }
            }
        }
        else_if = {
            limit = { check_variable = { settings_events_log_sort_mode = 1 } }
            if = {
                limit = { has_country_flag = events_log_sort_descending }
                set_temp_variable = { events_log_sort_id = 2000 }
                while_loop_effect = {
                    limit = { check_variable = { events_log_sort_id > 0 } }
                    subtract_from_temp_variable = { events_log_sort_id = 1 }
                    for_loop_effect = {
                        start = 0
                        end = events_log_source_count
                        value = i
                        set_temp_variable = { events_log_source_index = i }
                        if = {
                            limit = {
                                check_variable = {
                                    var = global.events_log_history_event_id_entries^events_log_source_index
                                    value = events_log_sort_id
                                    compare = equals
                                }
                            }
                            events_log_row_matches_active_filter = yes
                            if = {
                                limit = { check_variable = { events_log_row_matches_filter > 0 } }
                                events_log_append_view_row_from_source = yes
                            }
                        }
                    }
                }
            }
            else = {
                for_loop_effect = {
                    start = 1
                    end = 2001
                    value = events_log_sort_id
                    for_loop_effect = {
                        start = 0
                        end = events_log_source_count
                        value = i
                        set_temp_variable = { events_log_source_index = i }
                        if = {
                            limit = {
                                check_variable = {
                                    var = global.events_log_history_event_id_entries^events_log_source_index
                                    value = events_log_sort_id
                                    compare = equals
                                }
                            }
                            events_log_row_matches_active_filter = yes
                            if = {
                                limit = { check_variable = { events_log_row_matches_filter > 0 } }
                                events_log_append_view_row_from_source = yes
                            }
                        }
                    }
                }
            }
        }
        else = {
            events_log_history_append_rows_by_actor_sort = yes
        }
    }

    events_log_append_cluster_history_rows = yes
}

                            has_country_flag = events_log_events_sort_descending
                            check_variable = {
                                var = temp_events_log_events_source_event_id_entries^events_log_events_scan_index
                                value = events_log_events_best_event_id
                                compare = greater_than
                            }
                        }
                        set_temp_variable = { events_log_events_best_index = events_log_events_scan_index }
                        set_temp_variable = { events_log_events_best_metric = events_log_events_candidate_metric }
                        set_temp_variable = { events_log_events_best_event_id = temp_events_log_events_source_event_id_entries^events_log_events_scan_index }
                    }
                    else_if = {
                        limit = {
                            NOT = { has_country_flag = events_log_events_sort_descending }
                            check_variable = {
                                var = temp_events_log_events_source_event_id_entries^events_log_events_scan_index
                                value = events_log_events_best_event_id
                                compare = less_than
                            }
                        }
                        set_temp_variable = { events_log_events_best_index = events_log_events_scan_index }
                        set_temp_variable = { events_log_events_best_metric = events_log_events_candidate_metric }
                        set_temp_variable = { events_log_events_best_event_id = temp_events_log_events_source_event_id_entries^events_log_events_scan_index }
                    }
                }
            }
            add_to_temp_variable = { events_log_events_scan_index = 1 }
        }

        if = {
            limit = { check_variable = { events_log_events_best_index > -1 } }
            add_to_array = {
                array = global.events_log_view_event_id_entries
                value = temp_events_log_events_source_event_id_entries^events_log_events_best_index
            }
            add_to_array = {
                array = global.events_log_view_event_type_entries
                value = temp_events_log_events_source_event_type_entries^events_log_events_best_index
            }
            add_to_array = {
                array = global.events_log_events_view_fired_entries
                value = temp_events_log_events_source_fired_entries^events_log_events_best_index
            }
            add_to_array = {
                array = global.events_log_events_view_enabled_entries
                value = temp_events_log_events_source_enabled_entries^events_log_events_best_index
            }
            add_to_array = {
                array = global.events_log_events_view_unique_entries
                value = temp_events_log_events_source_unique_entries^events_log_events_best_index
            }
            add_to_array = {
                array = global.events_log_events_view_weight_entries
                value = temp_events_log_events_source_weight_entries^events_log_events_best_index
            }

            set_variable = { temp_events_log_events_source_event_id_entries^events_log_events_best_index = -1 }
            add_to_temp_variable = { events_log_events_sorted_count = 1 }
        }
        else = {
            set_temp_variable = { events_log_events_sorted_count = events_log_events_source_count }
        }
    }
}
