tensions_rising_prepare_random_event_fire = {
	get_tensions_rising_stage = yes
	if = {
		limit = {
			check_variable = { tensions_rising_stage < constant:tensions_rising_event_log.stage_1 }
			NOT = { can_tensions_rising_baseline_fire = yes }
		}
		set_temp_variable = { event_single_fire_allowed = 0 }
	}
}

evaluate_tensions_rising_random_candidate = {
	get_tensions_rising_stage = yes
	if = {
		limit = {
			check_variable = { tensions_rising_stage < constant:tensions_rising_event_log.stage_1 }
			NOT = { can_tensions_rising_baseline_fire = yes }
		}
		set_temp_variable = { random_event_selection_candidate_is_valid = 0 }
	}
}

get_tensions_rising_stage = {
	set_temp_variable = { tensions_rising_stage = 0 }
	if = {
		limit = {
			OR = {
				has_global_flag = { flag = chaos_tier value = 4 }
				has_global_flag = { flag = chaos_tier value = 5 }
			}
		}
		set_temp_variable = { tensions_rising_stage = constant:tensions_rising_event_log.stage_4 }
	}
	else_if = {
		limit = {
			has_global_flag = { flag = chaos_tier value = 3 }
		}
		set_temp_variable = { tensions_rising_stage = constant:tensions_rising_event_log.stage_3 }
	}
	else_if = {
		limit = {
			has_global_flag = { flag = chaos_tier value = 2 }
		}
		set_temp_variable = { tensions_rising_stage = constant:tensions_rising_event_log.stage_2 }
	}
	else_if = {
		limit = {
			has_global_flag = { flag = chaos_tier value = 1 }
		}
		set_temp_variable = { tensions_rising_stage = constant:tensions_rising_event_log.stage_1 }
	}
}

apply_tensions_rising_event_effect = {
	get_tensions_rising_stage = yes
	if = {
		limit = { check_variable = { tensions_rising_stage < constant:tensions_rising_event_log.stage_1 } }
		set_temp_variable = { tensions_rising_world_tension_packet = constant:tensions_rising_world_tension_gain.baseline }
		apply_tensions_rising_distributed_world_tension = yes
		tensions_rising_sync_world_tension_tracker = yes
	}
	else = {
		record_tensions_rising_evolution_if_needed = yes

		apply_tensions_rising_ai_posture = yes
		schedule_tensions_rising_followup = yes
		update_tensions_rising_achievement_tracking = yes
	}
}

apply_tensions_rising_distributed_world_tension = {
	set_temp_variable = { tensions_rising_world_tension_country_count = 0 }
	every_country = {
		limit = { is_tensions_rising_world_tension_source_country = yes }
		add_to_temp_variable = { tensions_rising_world_tension_country_count = 1 }
	}

	if = {
		limit = { check_variable = { tensions_rising_world_tension_country_count > 0 } }
		if = {
			limit = {
				check_variable = { tensions_rising_world_tension_packet = constant:tensions_rising_world_tension_gain.stage_4 }
				check_variable = { tensions_rising_world_tension_country_count > @TENSIONS_RISING_WORLD_TENSION_100_MAX_RECIPIENTS }
			}
			every_country = {
				limit = { is_tensions_rising_world_tension_source_country = yes }
				random_select_amount = @TENSIONS_RISING_WORLD_TENSION_100_MAX_RECIPIENTS
				add_named_threat = {
					threat = constant:tensions_rising_world_tension_gain.minimum_country_slice
					name = tensions_rising_global_panic_threat
				}
			}
		}
		else_if = {
			limit = {
				check_variable = { tensions_rising_world_tension_packet = constant:tensions_rising_world_tension_gain.stage_3 }
				check_variable = { tensions_rising_world_tension_country_count > @TENSIONS_RISING_WORLD_TENSION_50_MAX_RECIPIENTS }
			}
			every_country = {
				limit = { is_tensions_rising_world_tension_source_country = yes }
				random_select_amount = @TENSIONS_RISING_WORLD_TENSION_50_MAX_RECIPIENTS
				add_named_threat = {
					threat = constant:tensions_rising_world_tension_gain.minimum_country_slice
					name = tensions_rising_global_panic_threat
				}
			}
		}
		else_if = {
			limit = {
				check_variable = { tensions_rising_world_tension_packet = constant:tensions_rising_world_tension_gain.stage_2 }
				check_variable = { tensions_rising_world_tension_country_count > @TENSIONS_RISING_WORLD_TENSION_20_MAX_RECIPIENTS }
			}
			every_country = {
				limit = { is_tensions_rising_world_tension_source_country = yes }
				random_select_amount = @TENSIONS_RISING_WORLD_TENSION_20_MAX_RECIPIENTS
				add_named_threat = {
					threat = constant:tensions_rising_world_tension_gain.minimum_country_slice
					name = tensions_rising_global_panic_threat
				}
			}
		}
		else_if = {
			limit = {
				OR = {
					check_variable = { tensions_rising_world_tension_packet = constant:tensions_rising_world_tension_gain.baseline }
					check_variable = { tensions_rising_world_tension_packet = constant:tensions_rising_world_tension_gain.stage_1 }
				}
				check_variable = { tensions_rising_world_tension_country_count > @TENSIONS_RISING_WORLD_TENSION_10_MAX_RECIPIENTS }
			}
			every_country = {
				limit = { is_tensions_rising_world_tension_source_country = yes }
				random_select_amount = @TENSIONS_RISING_WORLD_TENSION_10_MAX_RECIPIENTS
				add_named_threat = {
					threat = constant:tensions_rising_world_tension_gain.minimum_country_slice
					name = tensions_rising_global_panic_threat
				}
			}
		}
		else = {
			set_temp_variable = { tensions_rising_world_tension_slice = tensions_rising_world_tension_packet }
			divide_temp_variable = { tensions_rising_world_tension_slice = tensions_rising_world_tension_country_count }
			every_country = {
				limit = { is_tensions_rising_world_tension_source_country = yes }
				add_named_threat = {
					threat = tensions_rising_world_tension_slice
					name = tensions_rising_global_panic_threat
				}
			}
		}
	}
}

tensions_rising_sync_world_tension_tracker = {
	set_variable = { global.current_world_tension = global.threat }
	multiply_variable = { global.current_world_tension = 100 }
	if = {
		limit = { NOT = { has_variable = global.highest_world_tension_recorded } }
		set_variable = { global.highest_world_tension_recorded = global.current_world_tension }
	}
	else_if = {
		limit = {
			check_variable = {
				var = global.current_world_tension
				value = global.highest_world_tension_recorded
				compare = greater_than
			}
		}
		set_variable = { global.highest_world_tension_recorded = global.current_world_tension }
	}
}

record_tensions_rising_evolution_if_needed = {
	set_temp_variable = { events_log_evolution_event_id = constant:tensions_rising_event_log.event_id }
	set_temp_variable = { events_log_evolution_type = constant:tensions_rising_event_log.evolution_type }
	set_temp_variable = { events_log_evolution_stage = tensions_rising_stage }
	set_temp_variable = { events_log_evolution_tier = tensions_rising_stage }
	if = {
		limit = {
			is_current_evolution_enabled = yes
			OR = {
				AND = {
					check_variable = { tensions_rising_stage = constant:tensions_rising_event_log.stage_1 }
					NOT = { has_global_flag = tensions_rising_stage_1_recorded }
				}
				AND = {
					check_variable = { tensions_rising_stage = constant:tensions_rising_event_log.stage_2 }
					NOT = { has_global_flag = tensions_rising_stage_2_recorded }
				}
				AND = {
					check_variable = { tensions_rising_stage = constant:tensions_rising_event_log.stage_3 }
					NOT = { has_global_flag = tensions_rising_stage_3_recorded }
				}
				AND = {
					check_variable = { tensions_rising_stage = constant:tensions_rising_event_log.stage_4 }
					NOT = { has_global_flag = tensions_rising_stage_4_recorded }
				}
			}
		}
		record_events_log_evolution_entry = yes
		if = {
			limit = { check_variable = { tensions_rising_stage = constant:tensions_rising_event_log.stage_4 } }
			set_global_flag = tensions_rising_stage_4_recorded
		}
		else_if = {
			limit = { check_variable = { tensions_rising_stage = constant:tensions_rising_event_log.stage_3 } }
			set_global_flag = tensions_rising_stage_3_recorded
		}
		else_if = {
			limit = { check_variable = { tensions_rising_stage = constant:tensions_rising_event_log.stage_2 } }
			set_global_flag = tensions_rising_stage_2_recorded
		}
		else = {
			set_global_flag = tensions_rising_stage_1_recorded
		}
	}
}

apply_tensions_rising_timer_pulse = {
	set_temp_variable = { tensions_rising_refresh_pulse_flag = 0 }
	if = {
		limit = { check_variable = { tensions_rising_stage = constant:tensions_rising_event_log.stage_4 } }
		set_temp_variable = { tensions_rising_pulse_strength = constant:tensions_rising_timer_pulse.stage_4_strength }
		set_temp_variable_to_random = {
			var = tensions_rising_pulse_days
			min = constant:tensions_rising_timer_pulse.stage_4_min_days
			max = constant:tensions_rising_timer_pulse.stage_4_max_days
			integer = yes
		}
		set_temp_variable = { tensions_rising_pulse_extension_days = constant:tensions_rising_timer_pulse.stage_4_extension_days }
		set_temp_variable = { tensions_rising_pulse_cap_days = constant:tensions_rising_timer_pulse.stage_4_cap_days }
	}
	else_if = {
		limit = { check_variable = { tensions_rising_stage = constant:tensions_rising_event_log.stage_3 } }
		set_temp_variable = { tensions_rising_pulse_strength = constant:tensions_rising_timer_pulse.stage_3_strength }
		set_temp_variable_to_random = {
			var = tensions_rising_pulse_days
			min = constant:tensions_rising_timer_pulse.stage_3_min_days
			max = constant:tensions_rising_timer_pulse.stage_3_max_days
			integer = yes
		}
		set_temp_variable = { tensions_rising_pulse_extension_days = constant:tensions_rising_timer_pulse.stage_3_extension_days }
		set_temp_variable = { tensions_rising_pulse_cap_days = constant:tensions_rising_timer_pulse.stage_3_cap_days }
	}
	else_if = {
		limit = { check_variable = { tensions_rising_stage = constant:tensions_rising_event_log.stage_2 } }
		set_temp_variable = { tensions_rising_pulse_strength = constant:tensions_rising_timer_pulse.stage_2_strength }
		set_temp_variable_to_random = {
			var = tensions_rising_pulse_days
			min = constant:tensions_rising_timer_pulse.stage_2_min_days
			max = constant:tensions_rising_timer_pulse.stage_2_max_days
			integer = yes
		}
		set_temp_variable = { tensions_rising_pulse_extension_days = constant:tensions_rising_timer_pulse.stage_2_extension_days }
		set_temp_variable = { tensions_rising_pulse_cap_days = constant:tensions_rising_timer_pulse.stage_2_cap_days }
	}
	else = {
		set_temp_variable = { tensions_rising_pulse_strength = constant:tensions_rising_timer_pulse.stage_1_strength }
		set_temp_variable_to_random = {
			var = tensions_rising_pulse_days
			min = constant:tensions_rising_timer_pulse.stage_1_min_days
			max = constant:tensions_rising_timer_pulse.stage_1_max_days
			integer = yes
		}
		set_temp_variable = { tensions_rising_pulse_extension_days = constant:tensions_rising_timer_pulse.stage_1_extension_days }
		set_temp_variable = { tensions_rising_pulse_cap_days = constant:tensions_rising_timer_pulse.stage_1_cap_days }
	}

	if = {
		limit = { NOT = { has_global_flag = tensions_rising_timer_pulse_active } }
		set_variable = { global.tensions_rising_timer_pulse_strength = tensions_rising_pulse_strength }
		set_variable = { global.tensions_rising_timer_pulse_days = tensions_rising_pulse_days }
		set_temp_variable = { tensions_rising_refresh_pulse_flag = 1 }
	}
	else_if = {
		limit = {
			check_variable = {
				var = tensions_rising_pulse_strength
				value = global.tensions_rising_timer_pulse_strength
				compare = greater_than
			}
		}
		set_variable = { global.tensions_rising_timer_pulse_strength = tensions_rising_pulse_strength }
		set_variable = { global.tensions_rising_timer_pulse_days = tensions_rising_pulse_days }
		set_temp_variable = { tensions_rising_refresh_pulse_flag = 1 }
	}
	else_if = {
		limit = {
			check_variable = {
				var = tensions_rising_pulse_strength
				value = global.tensions_rising_timer_pulse_strength
				compare = equals
			}
		}
		if = {
			limit = { NOT = { has_variable = global.tensions_rising_timer_pulse_days } }
			set_variable = { global.tensions_rising_timer_pulse_days = tensions_rising_pulse_days }
		}
		add_to_variable = { global.tensions_rising_timer_pulse_days = tensions_rising_pulse_extension_days }
		clamp_variable = {
			var = global.tensions_rising_timer_pulse_days
			min = 0
			max = tensions_rising_pulse_cap_days
		}
		set_temp_variable = { tensions_rising_refresh_pulse_flag = 1 }
	}

	if = {
		limit = { check_variable = { tensions_rising_refresh_pulse_flag > 0 } }
		refresh_tensions_rising_timer_pulse_active_flag = yes
	}
}

apply_tensions_rising_relation_pair = {
	set_temp_variable = { tensions_rising_relation_actor_found = 0 }
	set_temp_variable = { tensions_rising_relation_target_found = 0 }

	random_country = {
		limit = {
			is_tensions_rising_priority_relation_actor = yes
			NOT = { has_country_flag = tensions_rising_recent_relation_actor }
		}
		save_event_target_as = tensions_rising_first_actor
		ROOT = { set_temp_variable = { tensions_rising_relation_actor_found = 1 } }
	}
	if = {
		limit = { check_variable = { tensions_rising_relation_actor_found < 1 } }
		random_country = {
			limit = {
				is_tensions_rising_valid_diplomatic_country = yes
				NOT = { has_country_flag = tensions_rising_recent_relation_actor }
			}
			save_event_target_as = tensions_rising_first_actor
			ROOT = { set_temp_variable = { tensions_rising_relation_actor_found = 1 } }
		}
	}
	if = {
		limit = { check_variable = { tensions_rising_relation_actor_found < 1 } }
		random_country = {
			limit = { is_tensions_rising_valid_diplomatic_country = yes }
			save_event_target_as = tensions_rising_first_actor
			ROOT = { set_temp_variable = { tensions_rising_relation_actor_found = 1 } }
		}
	}

	if = {
		limit = { check_variable = { tensions_rising_relation_actor_found > 0 } }
		event_target:tensions_rising_first_actor = {
			random_country = {
				limit = {
					is_tensions_rising_relation_target_for_prev = yes
					is_neighbor_of = PREV
					NOT = { has_country_flag = tensions_rising_recent_relation_actor }
				}
				save_event_target_as = tensions_rising_second_actor
				ROOT = { set_temp_variable = { tensions_rising_relation_target_found = 1 } }
			}
			if = {
				limit = { ROOT = { check_variable = { tensions_rising_relation_target_found < 1 } } }
				random_country = {
					limit = {
						is_tensions_rising_relation_target_for_prev = yes
						OR = {
							is_major = yes
							is_faction_leader = yes
						}
						NOT = { has_country_flag = tensions_rising_recent_relation_actor }
					}
					save_event_target_as = tensions_rising_second_actor
					ROOT = { set_temp_variable = { tensions_rising_relation_target_found = 1 } }
				}
			}
			if = {
				limit = { ROOT = { check_variable = { tensions_rising_relation_target_found < 1 } } }
				random_country = {
					limit = {
						is_tensions_rising_relation_target_for_prev = yes
						NOT = { has_country_flag = tensions_rising_recent_relation_actor }
					}
					save_event_target_as = tensions_rising_second_actor
					ROOT = { set_temp_variable = { tensions_rising_relation_target_found = 1 } }
				}
			}
			if = {
				limit = { ROOT = { check_variable = { tensions_rising_relation_target_found < 1 } } }
				random_country = {
					limit = { is_tensions_rising_relation_target_for_prev = yes }
					save_event_target_as = tensions_rising_second_actor
					ROOT = { set_temp_variable = { tensions_rising_relation_target_found = 1 } }
				}
			}
		}
	}

	if = {
		limit = { check_variable = { tensions_rising_relation_target_found > 0 } }
		apply_tensions_rising_relation_modifier_to_targets = yes
		add_to_variable = { global.tensions_rising_relation_pair_count = 1 }
		add_to_variable = { global.tensions_rising_active_opinion_modifiers = 2 }
		set_global_flag = { flag = tensions_rising_recent_relation_damage days = @TENSIONS_RISING_RELATION_FOLLOWUP_WINDOW_DAYS }
	}
