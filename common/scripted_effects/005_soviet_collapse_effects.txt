soviet_collapse_recalculate_total_threat = {
	soviet_collapse_clamp_crisis_components = yes
	set_variable = { soviet_collapse_total_collapse_threat = soviet_collapse_republic_confidence }
	add_to_variable = { soviet_collapse_total_collapse_threat = soviet_collapse_depot_vulnerability }
	add_to_variable = { soviet_collapse_total_collapse_threat = soviet_collapse_foreign_appetite }
	add_to_variable = { soviet_collapse_total_collapse_threat = soviet_collapse_league_cohesion }
	add_to_variable = { soviet_collapse_total_collapse_threat = soviet_collapse_evolution_weirdness }
	add_to_variable = { soviet_collapse_total_collapse_threat = 100 }
	subtract_from_variable = { soviet_collapse_total_collapse_threat = soviet_collapse_moscow_authority }
	subtract_from_variable = { soviet_collapse_total_collapse_threat = soviet_collapse_military_obedience }
	multiply_variable = { var = soviet_collapse_total_collapse_threat value = 0.33 }
	clamp_variable = { var = soviet_collapse_total_collapse_threat min = constant:soviet_collapse_baseline.total_threat_floor max = constant:soviet_collapse_baseline.total_threat_ceiling }
}

		add_to_variable = { soviet_collapse_depot_vulnerability = constant:soviet_collapse_opening_pressure.chaos_tier_1 }
	}
	if = {
		limit = { has_global_flag = { flag = chaos_tier value = 2 } }
		add_to_variable = { soviet_collapse_republic_confidence = constant:soviet_collapse_opening_pressure.chaos_tier_2 }
		add_to_variable = { soviet_collapse_depot_vulnerability = constant:soviet_collapse_opening_pressure.chaos_tier_2 }
		add_to_variable = { soviet_collapse_foreign_appetite = 3 }
	}
	if = {
		limit = { has_global_flag = { flag = chaos_tier value = 3 } }
		add_to_variable = { soviet_collapse_republic_confidence = constant:soviet_collapse_opening_pressure.chaos_tier_3 }
		add_to_variable = { soviet_collapse_depot_vulnerability = constant:soviet_collapse_opening_pressure.chaos_tier_3 }

		add_to_variable = { soviet_collapse_evolution_weirdness = 4 }

		limit = { has_global_flag = { flag = chaos_tier value = 4 } }

		add_to_variable = { soviet_collapse_depot_vulnerability = constant:soviet_collapse_opening_pressure.chaos_tier_final }
		add_to_variable = { soviet_collapse_foreign_appetite = 16 }
		add_to_variable = { soviet_collapse_evolution_weirdness = 14 }

		limit = { has_stability < constant:soviet_collapse_soviet_objective.good_stability }
		add_to_variable = { soviet_collapse_republic_confidence = constant:soviet_collapse_opening_pressure.low_stability }
		add_to_variable = { soviet_collapse_moscow_authority = -6 }

		limit = { has_war_support < constant:soviet_collapse_soviet_objective.good_war_support }

		add_to_variable = { soviet_collapse_republic_confidence = constant:soviet_collapse_opening_pressure.low_war_support }

		limit = { has_war = yes }
		add_to_variable = { soviet_collapse_depot_vulnerability = constant:soviet_collapse_opening_pressure.active_war }
		add_to_variable = { soviet_collapse_foreign_appetite = constant:soviet_collapse_opening_pressure.active_war }
	}
	if = {
		limit = { NOT = { capital_scope = { is_controlled_by = ROOT } } }
		add_to_variable = { soviet_collapse_moscow_authority = -18 }
		add_to_variable = { soviet_collapse_republic_confidence = constant:soviet_collapse_opening_pressure.capital_lost }
	}

	soviet_collapse_recalculate_total_threat = yes
}

soviet_collapse_setup_breakaway_country = {
	set_country_flag = soviet_collapse_breakaway
	add_to_array = {
		array = global.soviet_collapse_breakaway_countries
		value = THIS
	}
	add_to_variable = { global.soviet_collapse_breakaway_count = 1 }
	if = {
		limit = { SOV = { exists = yes } }
		SOV = { add_to_variable = { soviet_collapse_breakaway_count = 1 } }
	}

	add_manpower = constant:soviet_collapse_breakaway_support.base_manpower
	add_equipment_to_stockpile = { type = infantry_equipment amount = constant:soviet_collapse_breakaway_support.base_infantry_equipment }
	add_equipment_to_stockpile = { type = support_equipment_1 amount = constant:soviet_collapse_breakaway_support.base_support_equipment }
	add_ideas = soviet_collapse_captured_soviet_depots
	add_ideas = soviet_collapse_defensive_coordination

	division_template = {
		name = "Emergency Republican Guard"
		is_locked = yes
		force_allow_recruiting = yes
		regiments = {
			infantry = { x = 0 y = 0 }
			infantry = { x = 0 y = 1 }
			infantry = { x = 1 y = 0 }
			infantry = { x = 1 y = 1 }
			infantry = { x = 2 y = 0 }
			infantry = { x = 2 y = 1 }
		}
	}
	capital_scope = {
		create_unit = {
			division = "division_template = \"Emergency Republican Guard\" start_experience_factor = 0.15 start_equipment_factor = 0.9"
			owner = PREV
			count = 3
		}
	}
	if = {
		limit = {
			OR = {
				has_global_flag = { flag = chaos_tier value = 3 }
				has_global_flag = { flag = chaos_tier value = 4 }
				has_global_flag = { flag = chaos_tier value = 5 }
			}
		}
		add_manpower = constant:soviet_collapse_breakaway_support.extra_manpower_per_chaos_band
		add_equipment_to_stockpile = { type = infantry_equipment amount = constant:soviet_collapse_breakaway_support.extra_infantry_equipment_per_chaos_band }
		add_equipment_to_stockpile = { type = support_equipment_1 amount = constant:soviet_collapse_breakaway_support.extra_support_equipment_per_chaos_band }
		capital_scope = {
			create_unit = {
				division = "division_template = \"Emergency Republican Guard\" start_experience_factor = 0.2 start_equipment_factor = 0.9"
				owner = PREV
				count = 1
			}
		}
	}
}
