fury_sync_evolution_unlocks = {
	set_temp_variable = { events_log_evolution_event_id = constant:fury_event_log.event_id }
	set_temp_variable = { events_log_evolution_type = constant:fury_event_log.evolution_type }
	set_temp_variable = { events_log_evolution_stage = constant:fury_event_log.hardened_stage }
	if = {
		limit = {
			check_variable = {
				var = global.chaos_meter_value
				value = constant:chaos_meter_tier_range.tier_1.min
				compare = greater_than_or_equals
			}
			is_current_evolution_enabled = yes
		}
		set_global_flag = fury_evolution_i_unlocked
	}
	else = {
		clr_global_flag = fury_evolution_i_unlocked
	}
	set_temp_variable = { events_log_evolution_stage = constant:fury_event_log.two_fires_stage }
	if = {
		limit = {
			check_variable = {
				var = global.chaos_meter_value
				value = constant:chaos_meter_tier_range.tier_2.min
				compare = greater_than_or_equals
			}
			is_current_evolution_enabled = yes
		}
		set_global_flag = fury_evolution_ii_unlocked
	}
	else = {
		clr_global_flag = fury_evolution_ii_unlocked
	}
	set_temp_variable = { events_log_evolution_stage = constant:fury_event_log.all_borders_stage }
	if = {
		limit = {
			check_variable = {
				var = global.chaos_meter_value
				value = constant:chaos_meter_tier_range.tier_3.min
				compare = greater_than_or_equals
			}
			is_current_evolution_enabled = yes
		}
		set_global_flag = fury_evolution_iii_unlocked
	}
	else = {
		clr_global_flag = fury_evolution_iii_unlocked
	}
}

fury_prepare_random_event_fire = {
	fury_sync_evolution_unlocks = yes
	if = {
		limit = { has_global_flag = fury_evolution_iii_unlocked }
		fury_ensure_actor_count_three = yes
	}
	else_if = {
		limit = { has_global_flag = fury_evolution_ii_unlocked }
		fury_ensure_actor_count_two = yes
	}
	else_if = {
		limit = { NOT = { any_country = { has_country_flag = fury_actor } } }
		fury_select_ordinary_actor = yes
	}
}

fury_select_ordinary_actor = {
	if = {
		limit = { any_country = { fury_can_be_selected_one_state = yes } }
		random_country = {
			limit = { fury_can_be_selected_one_state = yes }
			fury_apply_package = yes
		}
	}
	else_if = {
		limit = { any_country = { fury_can_be_selected_two_state = yes } }
		random_country = {
			limit = { fury_can_be_selected_two_state = yes }
			fury_apply_package = yes
		}
	}
	else_if = {
		limit = { any_country = { fury_can_be_selected_three_state = yes } }
		random_country = {
			limit = { fury_can_be_selected_three_state = yes }
			fury_apply_package = yes
		}
	}
	else = {
		random_country = {
			limit = { fury_can_be_selected = yes }
			fury_apply_package = yes
		}
	}
}

fury_score_current_target_candidate = {
	set_variable = { fury_target_score = constant:fury_target_score.base_neighbor }
	if = {
		limit = { num_of_controlled_states < constant:fury_target_score.one_state_max }
		add_to_variable = { fury_target_score = constant:fury_target_score.one_state_bonus }
	}
	else_if = {
		limit = { num_of_controlled_states < constant:fury_target_score.two_state_max }
		add_to_variable = { fury_target_score = constant:fury_target_score.two_state_bonus }
	}
	if = {
		limit = { num_of_factories < constant:fury_target_score.weak_factory_max }
		add_to_variable = { fury_target_score = constant:fury_target_score.weak_industry_bonus }
	}
	if = {
		limit = { num_divisions < constant:fury_target_score.low_division_max }
		add_to_variable = { fury_target_score = constant:fury_target_score.low_division_bonus }
	}
	else_if = {
		limit = { num_divisions > constant:fury_target_score.high_division_min }
		subtract_from_variable = { fury_target_score = constant:fury_target_score.high_division_penalty }
	}
	ROOT = { set_variable = { fury_actor_divisions_for_target_score = num_divisions } }
	set_variable = { fury_target_relative_divisions = num_divisions }
	subtract_from_variable = { fury_target_relative_divisions = ROOT.fury_actor_divisions_for_target_score }
	if = {
		limit = {
			check_variable = {
				var = fury_target_relative_divisions
				value = 0
				compare = greater_than
			}
		}
		subtract_from_variable = { fury_target_score = constant:fury_target_score.stronger_than_fury_penalty }
	}
	else_if = {
		limit = {
			check_variable = {
				var = fury_target_relative_divisions
				value = 0
				compare = less_than
			}
		}
		add_to_variable = { fury_target_score = constant:fury_target_score.weaker_than_fury_bonus }
	}
	clear_variable = fury_target_relative_divisions
	ROOT = { clear_variable = fury_actor_divisions_for_target_score }
	if = {
		limit = { has_manpower < constant:fury_target_score.low_manpower_gate }
		add_to_variable = { fury_target_score = constant:fury_target_score.low_manpower_bonus }
	}
	else_if = {
		limit = { has_manpower > constant:fury_target_score.high_manpower_gate }
		subtract_from_variable = { fury_target_score = constant:fury_target_score.high_manpower_penalty }
	}
	if = {
		limit = { has_war = yes }
		add_to_variable = { fury_target_score = constant:fury_target_score.target_at_war_bonus }
	}
	if = {
		limit = { is_in_faction = yes }
		subtract_from_variable = { fury_target_score = constant:fury_target_score.faction_penalty }
	}
	if = {
		limit = { is_major = yes }
		subtract_from_variable = { fury_target_score = constant:fury_target_score.major_penalty }
	}
	if = {
		limit = {
			OR = {
				has_terrain = mountain
				has_terrain = jungle
				has_terrain = marsh
			}
		}
		subtract_from_variable = { fury_target_score = constant:fury_target_score.rough_terrain_penalty }
	}
	if = {
		limit = { num_of_supply_nodes < constant:fury_target_score.weak_supply_node_gate }
		subtract_from_variable = { fury_target_score = constant:fury_target_score.weak_supply_penalty }
	}
	if = {
		limit = {
			ROOT = {
				check_variable = {
					var = fury_last_target_id
					value = PREV.id
					compare = equals
				}
			}
		}
		subtract_from_variable = { fury_target_score = constant:fury_target_score.last_target_churn_penalty }
	}
	if = {
		limit = {
			has_equipment = { infantry_equipment > constant:fury_target_score.high_equipment_gate }
			ROOT = {
				NOT = { has_equipment = { infantry_equipment > constant:fury_target_score.high_equipment_gate } }
			}
		}
		subtract_from_variable = { fury_target_score = constant:fury_target_score.target_equipment_depth_penalty }
	}
	if = {
		limit = {
			ROOT = {
				check_variable = {
					var = fury_overextension
					value = constant:fury_target_score.overextension_high
					compare = greater_than
				}
			}
		}
		subtract_from_variable = { fury_target_score = constant:fury_target_score.overextension_penalty }
	}
	if = {
		limit = {
			ROOT = {
				check_variable = {
					var = fury_overextension
					value = constant:fury_target_score.overextension_extreme
					compare = greater_than
				}
			}
		}
		subtract_from_variable = { fury_target_score = constant:fury_target_score.overextension_penalty }
	}
	if = {
		limit = {
			ROOT = {
				check_variable = {
					var = fury_momentum
					value = 0
					compare = greater_than
				}
			}
		}
		add_to_variable = { fury_target_score = constant:fury_target_score.momentum_bonus }
	}
	if = {
		limit = {
			ROOT = {
				check_variable = {
					var = fury_target_preparation
					value = 0
					compare = greater_than
				}
			}
		}
		set_variable = { fury_target_preparation_score = ROOT.fury_target_preparation }
		multiply_variable = {
			var = fury_target_preparation_score
			value = constant:fury_target_score.preparation_bonus
		}
		add_to_variable = { fury_target_score = fury_target_preparation_score }
		clear_variable = fury_target_preparation_score
	}
	if = {
		limit = { ROOT = { has_country_flag = fury_evolution_i_applied } }
		add_to_variable = { fury_target_score = constant:fury_target_score.evolution_i_bonus }
	}
	if = {
		limit = { ROOT = { has_country_flag = fury_evolution_iii_applied } }
		add_to_variable = { fury_target_score = constant:fury_target_score.evolution_iii_bonus }
	}
}

fury_choose_next_target = {
	clr_country_flag = fury_has_current_target
	clear_variable = fury_current_target
	set_variable = { fury_best_target_score = 0 }
	every_neighbor_country = {
		limit = { fury_is_valid_target = yes }
		fury_score_current_target_candidate = yes
		if = {
			limit = {
				fury_candidate_passes_score_gate = yes
				check_variable = {
					var = fury_target_score
					value = ROOT.fury_best_target_score
					compare = greater_than
				}
			}
			ROOT = {
				set_variable = { fury_current_target = PREV.id }
				set_variable = { fury_best_target_score = PREV.fury_target_score }
				set_country_flag = fury_has_current_target
			}
		}
		clear_variable = fury_target_score
	}
	if = {
		limit = { has_country_flag = fury_has_current_target }
		set_variable = { fury_failed_target_scans = 0 }
		clr_country_flag = fury_no_valid_neighbors
	}
	else = {
		add_to_variable = { fury_failed_target_scans = 1 }
		set_country_flag = fury_no_valid_neighbors
	}
}

fury_try_terminal_player_target = {
	if = {
		limit = {
			has_global_flag = fury_world_end_active
			has_country_flag = fury_world_end_actor
			NOT = { has_war = yes }
			NOT = { any_neighbor_country = { fury_is_valid_target = yes } }
			fury_has_terminal_player_target = yes
		}
		fury_choose_terminal_player_target = yes
		if = {
			limit = { has_country_flag = fury_has_current_target }
			every_country = {
				limit = { is_ai = no }
				country_event = { id = chaosx.nr7.52 }
			}
			fury_declare_on_current_target = yes
			add_to_variable = { fury_momentum = constant:fury_terminal.terminal_player_threat_momentum }
		}
	}
}

fury_declare_on_current_target = {
	if = {
		limit = { has_country_flag = fury_has_current_target }
		set_variable = { fury_last_target_id = fury_current_target }
		declare_war_on = {
			type = annex_everything
			target = var:fury_current_target
		}
		clr_country_flag = fury_no_valid_neighbors
		add_to_variable = { fury_momentum = 1 }
	}
}

	set_variable = { fury_registered_state_count = 0 }
	set_variable = { fury_occupation_pressure = fury_overextension }
	every_controlled_state = {
		limit = {
			NOT = { is_core_of = ROOT }
		}
		ROOT = { add_to_variable = { fury_occupation_pressure = constant:fury_occupation.overextension_per_conquered_state } }
		if = {
			limit = { has_state_flag = fury_settlement_registered }
			ROOT = {
				add_to_variable = { fury_registered_state_count = 1 }
				subtract_from_variable = { fury_occupation_pressure = constant:fury_occupation.registered_state_relief }
			}
		}
		if = {
			limit = { has_state_flag = fury_rail_registry_complete }
			ROOT = { subtract_from_variable = { fury_occupation_pressure = constant:fury_occupation.rail_registry_bonus } }
		}
		if = {
			limit = { resistance > 30 }
			ROOT = { add_to_variable = { fury_occupation_pressure = constant:fury_occupation.resistance_failure_pressure } }
		}
	}
	if = {
		limit = { has_war = yes }
		add_to_variable = { fury_occupation_pressure = constant:fury_occupation.overextension_per_active_war }
	}
	set_variable = { fury_garrison_capacity = num_divisions }
	multiply_variable = {
		var = fury_garrison_capacity
		value = constant:fury_occupation.garrison_capacity_per_unit
	}
	subtract_from_variable = { fury_occupation_pressure = fury_garrison_capacity }
	clamp_variable = { var = fury_occupation_pressure min = 0 }
	if = {
		limit = { fury_occupation_pressure_high = yes }
		add_ideas = fury_overextension
	}
	else = {
		remove_ideas = fury_overextension
	}
	if = {
		limit = { fury_occupation_pressure_extreme = yes }
		set_country_flag = fury_occupation_crisis_active
	}
	else = {
		clr_country_flag = fury_occupation_crisis_active
	}
}

trigger_world_in_fury_scenario = {
	fury_prepare_triggerable_scenario_actor_count = yes
	set_global_flag = fury_triggerable_scenario_launched
	set_global_flag = fury_triggerable_scenario_launch_active
	set_variable = { global.fury_triggerable_scenario_intensity = triggerable_scenarios_intensity }
	set_variable = { global.fury_triggerable_scenario_type = triggerable_scenarios_fury_type }
	clr_global_flag = fury_triggerable_scenario_maximum
	clr_global_flag = fury_triggerable_scenario_pact
	clr_global_flag = fury_triggerable_scenario_hostile
	clr_global_flag = fury_triggerable_scenario_used_every_safe_candidate
	clr_global_flag = fury_triggerable_scenario_short_of_maximum
	if = {
		limit = { check_variable = { global.fury_triggerable_scenario_intensity = constant:triggerable_scenario_intensity.maximum } }
		set_global_flag = fury_triggerable_scenario_maximum
	}
	if = {
		limit = { check_variable = { global.fury_triggerable_scenario_type = constant:fury_scenario_type.hostile } }
		set_global_flag = fury_triggerable_scenario_hostile
	}
	else = {
		set_global_flag = fury_triggerable_scenario_pact
	}
	set_temp_variable = { fury_triggerable_actor_count = 0 }
	fury_seed_triggerable_scenario_continental_one_state = yes
	fury_seed_triggerable_scenario_continental_one_state = yes
	fury_seed_triggerable_scenario_continental_two_state = yes
	fury_seed_triggerable_scenario_continental_two_state = yes
	fury_seed_triggerable_scenario_continental_general = yes
	fury_seed_triggerable_scenario_continental_general = yes
	while_loop_effect = {
		limit = {
			check_variable = { fury_triggerable_actor_count < fury_triggerable_actor_goal }
			any_country = { fury_can_be_selected = yes }
		}
		random_country = {
			limit = { fury_can_be_selected = yes }
			fury_apply_triggerable_scenario_actor_package = yes
		}
	}
	set_variable = { global.fury_triggerable_scenario_actor_count = fury_triggerable_actor_count }
	if = {
		limit = {
			check_variable = { global.fury_triggerable_scenario_intensity = constant:triggerable_scenario_intensity.maximum }
			check_variable = {
				var = fury_triggerable_actor_count
				value = constant:fury_scenario_actor_count.maximum
				compare = less_than
			}
			NOT = { any_country = { fury_can_be_selected = yes } }
		}
		set_global_flag = fury_triggerable_scenario_used_every_safe_candidate
		set_global_flag = fury_triggerable_scenario_short_of_maximum
	}
	if = {
		limit = { check_variable = { global.fury_triggerable_scenario_type = constant:fury_scenario_type.pact } }
		set_global_flag = fury_pact_formed
		random_country = {
			limit = { has_country_flag = fury_triggerable_scenario_actor }
			create_faction = "The Fury Pact"
			save_global_event_target_as = fury_pact_leader
		}
		every_country = {
			limit = {
				has_country_flag = fury_triggerable_scenario_actor
				NOT = { tag = event_target:fury_pact_leader }
			}
			event_target:fury_pact_leader = { add_to_faction = PREV }
			set_country_flag = fury_pact_member
		}
	}
	else = {
		every_country = {
			limit = { has_country_flag = fury_triggerable_scenario_actor }
			if = {
				limit = { check_variable = { global.fury_triggerable_scenario_intensity = constant:triggerable_scenario_intensity.maximum } }
				fury_declare_all_valid_neighbors = yes
				fury_declare_on_rival_actor = yes
			}
			else = {
				country_event = { id = chaosx.nr7.10 days = 1 }
			}
		}
	}
	clr_global_flag = fury_triggerable_scenario_launch_active
}
