sanitize_chaos_meter_history_actor_entries = {
	if = {
		limit = {
			check_variable = {
				var = global.chaos_meter_history_sequence_entries^num
				value = global.chaos_meter_history_actor_entries^num
				compare = greater_than
			}
		}
		set_temp_variable = { missing_actor_entries = global.chaos_meter_history_sequence_entries^num }
		subtract_from_temp_variable = { missing_actor_entries = global.chaos_meter_history_actor_entries^num }

		while_loop_effect = {
			limit = { check_variable = { missing_actor_entries > 0 } }
			add_to_array = {
				array = global.chaos_meter_history_actor_entries
				value = 0
			}
			add_to_array = {
				array = global.chaos_meter_history_has_actor_entries
				value = 0
			}
			subtract_from_temp_variable = { missing_actor_entries = 1 }
		}
	}

	if = {
		limit = {
			check_variable = {
				var = global.chaos_meter_history_sequence_entries^num
				value = global.chaos_meter_history_has_actor_entries^num
				compare = greater_than
			}
		}
		set_temp_variable = { missing_has_actor_entries = global.chaos_meter_history_sequence_entries^num }
		subtract_from_temp_variable = { missing_has_actor_entries = global.chaos_meter_history_has_actor_entries^num }

		while_loop_effect = {
			limit = { check_variable = { missing_has_actor_entries > 0 } }
			add_to_array = {
				array = global.chaos_meter_history_has_actor_entries
				value = 0
			}
			subtract_from_temp_variable = { missing_has_actor_entries = 1 }
		}
	}

	set_temp_variable = { chaos_history_sanitize_index = 0 }
	while_loop_effect = {
		limit = {
			check_variable = {
				var = chaos_history_sanitize_index
				value = global.chaos_meter_history_sequence_entries^num
				compare = less_than
			}
		}

		if = {
			limit = {
				OR = {
					check_variable = {
						var = global.chaos_meter_history_reason_entries^chaos_history_sanitize_index
						value = constant:chaos_meter_history_reason.generic.increase
						compare = equals
					}
					check_variable = {
						var = global.chaos_meter_history_reason_entries^chaos_history_sanitize_index
						value = constant:chaos_meter_history_reason.generic.decrease
						compare = equals
					}
					check_variable = {
						var = global.chaos_meter_history_reason_entries^chaos_history_sanitize_index
						value = constant:chaos_meter_history_reason.settings.manual_update
						compare = equals
					}
					check_variable = {
						var = global.chaos_meter_history_reason_entries^chaos_history_sanitize_index
						value = constant:chaos_meter_history_reason.settings.disabled
						compare = equals
					}
					check_variable = {
						var = global.chaos_meter_history_reason_entries^chaos_history_sanitize_index
						value = constant:chaos_meter_history_reason.settings.reenabled
						compare = equals
					}
					AND = {
						check_variable = {
							var = global.chaos_meter_history_reason_entries^chaos_history_sanitize_index
							value = constant:chaos_meter_history_reason.world_state.world_tension_rise
							compare = greater_than_or_equals
						}
						check_variable = {
							var = global.chaos_meter_history_reason_entries^chaos_history_sanitize_index
							value = constant:chaos_meter_history_reason.world_state.deaths_rise
							compare = less_than_or_equals
						}
					}
				}
			}
			set_variable = { global.chaos_meter_history_actor_entries^chaos_history_sanitize_index = 0 }
			set_variable = { global.chaos_meter_history_has_actor_entries^chaos_history_sanitize_index = 0 }
		}

		if = {
			limit = {
				check_variable = {
					var = global.chaos_meter_history_has_actor_entries^chaos_history_sanitize_index
					value = 1
					compare = less_than
				}
			}
			set_variable = { global.chaos_meter_history_actor_entries^chaos_history_sanitize_index = 0 }
			set_variable = { global.chaos_meter_history_has_actor_entries^chaos_history_sanitize_index = 0 }
		}

		if = {
			limit = {
				check_variable = {
					var = global.chaos_meter_history_has_actor_entries^chaos_history_sanitize_index
					value = 0
					compare = greater_than
				}
				check_variable = {
					var = global.chaos_meter_history_actor_entries^chaos_history_sanitize_index
					value = 1
					compare = less_than
				}
			}
			set_variable = { global.chaos_meter_history_actor_entries^chaos_history_sanitize_index = 0 }
			set_variable = { global.chaos_meter_history_has_actor_entries^chaos_history_sanitize_index = 0 }
		}

		add_to_temp_variable = { chaos_history_sanitize_index = 1 }
	}
}

chaos_meter_resolve_country_sort_index = {
	set_temp_variable = { chaos_deaths_country_sort_index = global.countries^num }
	set_temp_variable = { chaos_deaths_country_sort_scan_index = 0 }

	while_loop_effect = {
		limit = {
			check_variable = {
				var = chaos_deaths_country_sort_scan_index
				value = global.countries^num
				compare = less_than
			}
			check_variable = {
				var = chaos_deaths_country_sort_index
				value = global.countries^num
				compare = equals
			}
		}
		if = {
			limit = {
				check_variable = {
					var = global.countries^chaos_deaths_country_sort_scan_index
					value = chaos_deaths_aggregate_country
					compare = equals
				}
			}
			set_temp_variable = { chaos_deaths_country_sort_index = chaos_deaths_country_sort_scan_index }
		}
		add_to_temp_variable = { chaos_deaths_country_sort_scan_index = 1 }
	}
}

	if = {
		limit = {
			check_variable = { chaos_deaths_change > 0 }
		}
		add_to_variable = { global.chaos_meter_deaths_chaos_buffer = chaos_deaths_change }

		set_temp_variable = { chaos_change = 0 }
		while_loop_effect = {
			limit = {
				check_variable = {
					var = global.chaos_meter_deaths_chaos_buffer
					value = constant:chaos_meter_deaths.million_for_chaos
					compare = greater_than_or_equals

			}

			limit = { check_variable = { chaos_change > 0 } }
			set_temp_variable = { chaos_history_reason = constant:chaos_meter_history_reason.world_state.deaths_rise }
			set_temp_variable = { chaos_history_reason_custom = 1 }
			add_chaos_meter_value = yes

		}
	}

	if = {
		limit = {
			NOT = { has_variable = chaos_deaths_change }
		}
		set_temp_variable = { chaos_deaths_change = 0 }
	}
	if = {
		limit = { NOT = { has_variable = chaos_deaths_target_country } }
		set_temp_variable = { chaos_deaths_target_country = 0 }
	}
	if = {
		limit = { NOT = { has_variable = chaos_deaths_has_target_country } }
		set_temp_variable = { chaos_deaths_has_target_country = 0 }
	}
	round_temp_variable = chaos_deaths_change

	if = {
		limit = {
			NOT = { has_global_flag = settings_chaos_deaths_disabled }
			check_variable = { chaos_deaths_change > 0 }
		}
		add_to_variable = { global.chaos_meter_deaths_total = chaos_deaths_change }
		set_variable = { global.chaos_meter_deaths_last_change = chaos_deaths_change }

		if = {
			limit = {
				has_variable = chaos_deaths_is_civilian
				check_variable = { chaos_deaths_is_civilian > 0 }
			}
			add_to_variable = { global.chaos_meter_deaths_civilian_total = chaos_deaths_change }

			if = {
				limit = {
					has_variable = chaos_deaths_apply_state_pop
					check_variable = { chaos_deaths_apply_state_pop > 0 }
				}
				set_temp_variable = { chaos_deaths_state_pop_delta = chaos_deaths_change }
				multiply_temp_variable = { chaos_deaths_state_pop_delta = -1 }
				add_manpower = var:chaos_deaths_state_pop_delta
			}
		}
		else = {
			add_to_variable = { global.chaos_meter_deaths_military_total = chaos_deaths_change }
		}

		chaos_meter_sync_chaos_from_deaths_delta = yes

	}
	else_if = {
		limit = { has_global_flag = settings_chaos_deaths_disabled }
		set_variable = { global.chaos_meter_deaths_last_change = 0 }
	}

	set_temp_variable = { chaos_deaths_target_country = 0 }
	set_temp_variable = { chaos_deaths_has_target_country = 0 }
}

chaos_meter_apply_nuclear_fallout_to_state = {
	set_temp_variable = { chaos_fallout_current_day = global.num_days }
	set_temp_variable = { chaos_fallout_existing_intensity = 0 }
	set_temp_variable = { chaos_fallout_existing_expiry = 0 }

	if = {
		limit = { has_variable = nuclear_fallout_intensity }
		set_temp_variable = { chaos_fallout_existing_intensity = nuclear_fallout_intensity }
	}
	if = {
		limit = { has_variable = nuclear_fallout_expiry_day }
		set_temp_variable = { chaos_fallout_existing_expiry = nuclear_fallout_expiry_day }
	}
	if = {
		limit = {
			check_variable = {
				var = chaos_fallout_existing_expiry
				value = chaos_fallout_current_day
				compare = less_than_or_equals
			}
		}
		set_temp_variable = { chaos_fallout_existing_intensity = 0 }
		set_temp_variable = { chaos_fallout_existing_expiry = chaos_fallout_current_day }
	}

	set_temp_variable = { chaos_fallout_new_intensity = chaos_fallout_existing_intensity }
	add_to_temp_variable = { chaos_fallout_new_intensity = ROOT.chaos_fallout_intensity_add }
	clamp_temp_variable = {
		var = chaos_fallout_new_intensity
		min = 0
		max = constant:chaos_meter_nuclear_fallout.intensity_max
	}
	set_variable = { nuclear_fallout_intensity = chaos_fallout_new_intensity }

	set_temp_variable = { chaos_fallout_new_expiry = chaos_fallout_current_day }
	add_to_temp_variable = { chaos_fallout_new_expiry = ROOT.chaos_fallout_duration_days }
	if = {
		limit = {
			check_variable = {
				var = chaos_fallout_existing_expiry
				value = chaos_fallout_new_expiry
				compare = greater_than
			}
		}
		set_temp_variable = { chaos_fallout_new_expiry = chaos_fallout_existing_expiry }
	}
	set_variable = { nuclear_fallout_expiry_day = chaos_fallout_new_expiry }

	set_temp_variable = { chaos_fallout_days_left = chaos_fallout_new_expiry }
	subtract_from_temp_variable = { chaos_fallout_days_left = chaos_fallout_current_day }
	round_temp_variable = chaos_fallout_days_left
	clamp_temp_variable = { var = chaos_fallout_days_left min = 1 max = 5000 }
	add_dynamic_modifier = { modifier = nuclear_fallout_state days = chaos_fallout_days_left }
}

chaos_meter_register_nuclear_condemnation = {
	if = {
		limit = {
			tag = USA
			FROM = {
				OR = {
					OWNER = { tag = JAP }
					CONTROLLER = { tag = JAP }
					OWNER = { original_tag = JAP }

			}

				limit = { check_variable = { state_population_k > 10000 } }
				multiply_temp_variable = { chaos_nuke_condemnation_density_mult = constant:chaos_meter_nuclear_fallout.condemnation_populous_10m }
			}
			else_if = {
				limit = { check_variable = { state_population_k > 5000 } }
				multiply_temp_variable = { chaos_nuke_condemnation_density_mult = constant:chaos_meter_nuclear_fallout.condemnation_populous_5m }
			}
		}

		multiply_temp_variable = { chaos_nuke_condemnation_pop_millions = constant:chaos_meter_nuclear_fallout.condemnation_per_million_population }
		add_to_temp_variable = { chaos_nuke_condemnation_add = chaos_nuke_condemnation_pop_millions }
		multiply_temp_variable = { chaos_nuke_condemnation_add = chaos_nuke_condemnation_density_mult }

		if = {
			limit = { has_tech = thermonuclear_bombs }
			multiply_temp_variable = { chaos_nuke_condemnation_add = constant:chaos_meter_nuclear_fallout.condemnation_thermonuclear_mult }
		}

		if = {
			limit = { check_variable = { chaos_nuke_condemnation_add > 0 } }
			if = {
				limit = { NOT = { has_variable = chem_warfare_condemnation } }
				set_variable = { chem_warfare_condemnation = 0 }
			}
			add_to_variable = { chem_warfare_condemnation = chaos_nuke_condemnation_add }
			chem_apply_condemnation_diplomatic_consequences = yes
			refresh_chaos_meter_condemnation_view_if_open = yes
		}
	}

	else_if = {

chaos_meter_history_append_rows_by_actor_sort = {
	clear_array = temp_chaos_history_unique_actors

	# Build a unique actor list from rows that actually have actors.
	set_temp_variable = { chaos_history_actor_collect_index = 0 }
	while_loop_effect = {
		limit = { check_variable = { chaos_history_actor_collect_index < chaos_history_source_count } }
		set_temp_variable = { chaos_history_source_index = chaos_history_actor_collect_index }
		if = {
			limit = {
				check_variable = {
					var = global.chaos_meter_history_has_actor_entries^chaos_history_source_index
					value = 0
					compare = greater_than
				}
				NOT = {
					is_in_array = {
						temp_chaos_history_unique_actors = global.chaos_meter_history_actor_entries^chaos_history_source_index
					}
				}
			}
			add_to_array = {
				array = temp_chaos_history_unique_actors
				value = global.chaos_meter_history_actor_entries^chaos_history_source_index
			}
		}
		add_to_temp_variable = { chaos_history_actor_collect_index = 1 }
	}

	# Actor sort now uses deterministic actor-id passes.
	while_loop_effect = {
		limit = { check_variable = { temp_chaos_history_unique_actors^num > 0 } }

		if = {
			limit = { has_country_flag = chaos_meter_history_sort_descending }
			find_highest_in_array = {
				array = temp_chaos_history_unique_actors
				value = chaos_history_actor_sort_target
			}
		}
		else = {
			find_lowest_in_array = {
				array = temp_chaos_history_unique_actors
				value = chaos_history_actor_sort_target
			}
		}

		if = {
			limit = { has_country_flag = chaos_meter_history_sort_descending }
			set_temp_variable = { chaos_history_actor_member_index = 0 }
			while_loop_effect = {
				limit = { check_variable = { chaos_history_actor_member_index < chaos_history_source_count } }
				set_temp_variable = { chaos_history_source_index = chaos_history_actor_member_index }
				if = {
					limit = {
						check_variable = {
							var = global.chaos_meter_history_has_actor_entries^chaos_history_source_index
							value = 0
							compare = greater_than
						}
						check_variable = {
							var = global.chaos_meter_history_actor_entries^chaos_history_source_index
							value = chaos_history_actor_sort_target
							compare = equals
						}
					}
					chaos_meter_history_row_matches_active_filter = yes
					if = {
						limit = { check_variable = { chaos_history_row_matches_filter > 0 } }
						chaos_meter_history_append_view_row_from_source = yes
					}
				}
				add_to_temp_variable = { chaos_history_actor_member_index = 1 }
			}
		}
		else = {
			set_temp_variable = { chaos_history_actor_member_index = chaos_history_source_count }
			while_loop_effect = {
				limit = { check_variable = { chaos_history_actor_member_index > 0 } }
				subtract_from_temp_variable = { chaos_history_actor_member_index = 1 }
				set_temp_variable = { chaos_history_source_index = chaos_history_actor_member_index }
				if = {
					limit = {
						check_variable = {
							var = global.chaos_meter_history_has_actor_entries^chaos_history_source_index
							value = 0
							compare = greater_than
						}
						check_variable = {
							var = global.chaos_meter_history_actor_entries^chaos_history_source_index
							value = chaos_history_actor_sort_target
							compare = equals
						}
					}
					chaos_meter_history_row_matches_active_filter = yes
					if = {
						limit = { check_variable = { chaos_history_row_matches_filter > 0 } }
						chaos_meter_history_append_view_row_from_source = yes
					}
				}
			}
		}

		remove_from_array = {
			array = temp_chaos_history_unique_actors
			value = chaos_history_actor_sort_target
		}
	}

	# Non-actor rows are always appended after actor rows.
	if = {
		limit = { has_country_flag = chaos_meter_history_sort_descending }
		set_temp_variable = { chaos_history_source_index = 0 }
		while_loop_effect = {
			limit = { check_variable = { chaos_history_source_index < chaos_history_source_count } }
			if = {
				limit = {
					check_variable = {
						var = global.chaos_meter_history_has_actor_entries^chaos_history_source_index
						value = 1
						compare = less_than
					}
				}
				chaos_meter_history_row_matches_active_filter = yes
				if = {
					limit = { check_variable = { chaos_history_row_matches_filter > 0 } }
					chaos_meter_history_append_view_row_from_source = yes
				}
			}
			add_to_temp_variable = { chaos_history_source_index = 1 }
		}
	}
	else = {
		set_temp_variable = { chaos_history_source_index = chaos_history_source_count }
		while_loop_effect = {
			limit = { check_variable = { chaos_history_source_index > 0 } }
			subtract_from_temp_variable = { chaos_history_source_index = 1 }
			if = {
				limit = {
					check_variable = {
						var = global.chaos_meter_history_has_actor_entries^chaos_history_source_index
						value = 1
						compare = less_than
					}
				}
				chaos_meter_history_row_matches_active_filter = yes
				if = {
					limit = { check_variable = { chaos_history_row_matches_filter > 0 } }
					chaos_meter_history_append_view_row_from_source = yes
				}
			}
		}
	}
}

rebuild_chaos_meter_history_view = {
	sanitize_chaos_meter_history_actor_entries = yes

	if = {
		limit = { NOT = { has_variable = settings_chaos_history_sort_mode } }
		set_variable = { settings_chaos_history_sort_mode = 0 }
	}
	if = {
		limit = { check_variable = { settings_chaos_history_sort_mode > 2 } }
		set_variable = { settings_chaos_history_sort_mode = 0 }
	}
	if = {
		limit = { check_variable = { settings_chaos_history_sort_mode < 0 } }
		set_variable = { settings_chaos_history_sort_mode = 0 }
	}
	if = {
		limit = { NOT = { has_variable = settings_chaos_history_filter_type } }
		set_variable = { settings_chaos_history_filter_type = 0 }
	}
	if = {
		limit = { check_variable = { settings_chaos_history_filter_type > 3 } }
		set_variable = { settings_chaos_history_filter_type = 0 }
	}
	if = {
		limit = { check_variable = { settings_chaos_history_filter_type < 0 } }
		set_variable = { settings_chaos_history_filter_type = 0 }
	}

	clear_array = global.chaos_meter_history_view_sequence_entries
	clear_array = global.chaos_meter_history_view_delta_entries
	clear_array = global.chaos_meter_history_view_total_entries
	clear_array = global.chaos_meter_history_view_reason_entries
	clear_array = global.chaos_meter_history_view_date_entries
	clear_array = global.chaos_meter_history_view_target_count_entries
	clear_array = global.chaos_meter_history_view_actor_entries
	clear_array = global.chaos_meter_history_view_has_actor_entries

	set_temp_variable = { chaos_history_source_count = global.chaos_meter_history_sequence_entries^num }

	if = {
		limit = { check_variable = { chaos_history_source_count > 0 } }
		if = {
			limit = { check_variable = { settings_chaos_history_sort_mode = 0 } }
			if = {
				limit = { has_country_flag = chaos_meter_history_sort_descending }
				for_loop_effect = {
					start = 0
					end = chaos_history_source_count
					value = i
					set_temp_variable = { chaos_history_source_index = i }
					chaos_meter_history_row_matches_active_filter = yes
					if = {
						limit = { check_variable = { chaos_history_row_matches_filter > 0 } }
						chaos_meter_history_append_view_row_from_source = yes
					}
				}
			}
			else = {
				set_temp_variable = { chaos_history_asc_index = chaos_history_source_count }
				while_loop_effect = {
					limit = { check_variable = { chaos_history_asc_index > 0 } }
					subtract_from_temp_variable = { chaos_history_asc_index = 1 }
					set_temp_variable = { chaos_history_source_index = chaos_history_asc_index }
					chaos_meter_history_row_matches_active_filter = yes
					if = {
						limit = { check_variable = { chaos_history_row_matches_filter > 0 } }
						chaos_meter_history_append_view_row_from_source = yes
					}
				}
			}
		}
		else_if = {
			limit = { check_variable = { settings_chaos_history_sort_mode = 1 } }
			chaos_meter_history_append_rows_by_actor_sort = yes
		}
		else = {
			if = {
				limit = { has_country_flag = chaos_meter_history_sort_descending }
				set_temp_variable = { chaos_history_sort_delta = 200 }
				while_loop_effect = {
					limit = { check_variable = { chaos_history_sort_delta > -201 } }
					for_loop_effect = {
						start = 0
						end = chaos_history_source_count
						value = i
						set_temp_variable = { chaos_history_source_index = i }
						if = {
							limit = {
								check_variable = {
									var = global.chaos_meter_history_delta_entries^chaos_history_source_index
									value = chaos_history_sort_delta
									compare = equals
								}
							}
							chaos_meter_history_row_matches_active_filter = yes
							if = {
								limit = { check_variable = { chaos_history_row_matches_filter > 0 } }
								chaos_meter_history_append_view_row_from_source = yes
							}
						}
					}
					subtract_from_temp_variable = { chaos_history_sort_delta = 1 }
				}
			}
			else = {
				for_loop_effect = {
					start = -200
					end = 201
					value = chaos_history_sort_delta
					for_loop_effect = {
						start = 0
						end = chaos_history_source_count
						value = i
						set_temp_variable = { chaos_history_source_index = i }
						if = {
							limit = {
								check_variable = {
									var = global.chaos_meter_history_delta_entries^chaos_history_source_index
									value = chaos_history_sort_delta
									compare = equals
								}
							}
							chaos_meter_history_row_matches_active_filter = yes
							if = {
								limit = { check_variable = { chaos_history_row_matches_filter > 0 } }
								chaos_meter_history_append_view_row_from_source = yes
							}
						}
					}
				}
			}
		}
	}
}

record_chaos_meter_history_entry = {
	if = {
		limit = { NOT = { has_variable = global.chaos_meter_history_sequence } }
		set_variable = { global.chaos_meter_history_sequence = 0 }
	}

	if = {
		limit = { NOT = { has_variable = chaos_history_reason } }
		set_temp_variable = { chaos_history_reason = constant:chaos_meter_history_reason.generic.increase }
	}

	set_temp_variable = { chaos_history_reason_uses_actor = 1 }
	if = {
		limit = {
			OR = {
				check_variable = {
					var = chaos_history_reason
					value = constant:chaos_meter_history_reason.generic.increase
					compare = equals
				}
				check_variable = {
					var = chaos_history_reason
					value = constant:chaos_meter_history_reason.generic.decrease
					compare = equals
				}
				check_variable = {
					var = chaos_history_reason
					value = constant:chaos_meter_history_reason.settings.manual_update
					compare = equals
				}
				check_variable = {
					var = chaos_history_reason
					value = constant:chaos_meter_history_reason.settings.disabled
					compare = equals
				}
				check_variable = {
					var = chaos_history_reason
					value = constant:chaos_meter_history_reason.settings.reenabled
					compare = equals
				}
					AND = {
						check_variable = {
							var = chaos_history_reason
							value = constant:chaos_meter_history_reason.world_state.world_tension_rise
							compare = greater_than_or_equals
						}
						check_variable = {
							var = chaos_history_reason
							value = constant:chaos_meter_history_reason.world_state.deaths_rise
							compare = less_than_or_equals
						}
					}
				}
			}
		set_temp_variable = { chaos_history_reason_uses_actor = 0 }
	}

	if = {
		limit = { NOT = { has_variable = chaos_history_target_count } }
		set_temp_variable = { chaos_history_target_count = 1 }
	}

	sanitize_chaos_meter_history_actor_entries = yes

	set_temp_variable = { chaos_history_actor_from_event_target = 0 }
	if = {
		limit = {
			check_variable = { chaos_history_reason_uses_actor > 0 }
			has_event_target = chaos_history_actor
		}
		set_temp_variable = { chaos_history_actor_from_event_target = 1 }
	}

	set_temp_variable = { merge_with_latest_war_entry = 0 }
	if = {
		limit = {
			check_variable = { global.chaos_meter_history_reason_entries^num > 0 }
			check_variable = { global.chaos_meter_history_has_actor_entries^num > 0 }
			OR = {
				check_variable = {
					var = chaos_history_reason
					value = constant:chaos_meter_history_reason.war.major
					compare = equals
				}
				check_variable = {
					var = chaos_history_reason
					value = constant:chaos_meter_history_reason.war.minor
					compare = equals
				}
			}
			check_variable = {
				var = global.chaos_meter_history_reason_entries^0
				value = chaos_history_reason
				compare = equals
			}
			check_variable = {
				var = global.chaos_meter_history_date_entries^0
				value = global.date
				compare = equals
			}
				OR = {
					AND = {
						check_variable = { chaos_history_actor_from_event_target > 0 }
						check_variable = {
							var = global.chaos_meter_history_actor_entries^0
							value = event_target:chaos_history_actor
							compare = equals
						}
					}
					AND = {
						check_variable = { chaos_history_actor_from_event_target < 1 }
						check_variable = {
							var = global.chaos_meter_history_has_actor_entries^0
							value = 1
							compare = less_than
						}
					}
			}
		}
		set_temp_variable = { merge_with_latest_war_entry = 1 }
	}

	if = {
		limit = { check_variable = { merge_with_latest_war_entry > 0 } }
		add_to_variable = { global.chaos_meter_history_delta_entries^0 = chaos_change }
		set_variable = { global.chaos_meter_history_total_entries^0 = global.chaos_meter_value }
		add_to_variable = { global.chaos_meter_history_target_count_entries^0 = 1 }
	}
	else = {
		add_to_variable = { global.chaos_meter_history_sequence = 1 }

		add_to_array = {
			array = global.chaos_meter_history_sequence_entries
			value = global.chaos_meter_history_sequence
			index = 0
		}

		add_to_array = {
			array = global.chaos_meter_history_delta_entries
			value = chaos_change
			index = 0
		}

		add_to_array = {
			array = global.chaos_meter_history_total_entries
			value = global.chaos_meter_value
			index = 0
		}

		add_to_array = {
			array = global.chaos_meter_history_reason_entries
			value = chaos_history_reason
			index = 0
		}

		add_to_array = {
			array = global.chaos_meter_history_date_entries
			value = global.date
			index = 0
		}

		add_to_array = {
			array = global.chaos_meter_history_target_count_entries
			value = chaos_history_target_count
			index = 0
		}

		if = {
			limit = {
				check_variable = { chaos_history_reason_uses_actor > 0 }
				has_event_target = chaos_history_actor
			}
			event_target:chaos_history_actor = {
				add_to_array = {
					array = global.chaos_meter_history_actor_entries
					value = THIS
					index = 0
				}
			}
			add_to_array = {
				array = global.chaos_meter_history_has_actor_entries
				value = 1
				index = 0
			}
		}
		else = {
			add_to_array = {
				array = global.chaos_meter_history_actor_entries
				value = 0
				index = 0
			}
			add_to_array = {
				array = global.chaos_meter_history_has_actor_entries
				value = 0
				index = 0
			}
		}
	}

	every_country = {
		limit = {
			is_ai = no
			has_country_flag = chaos_meter_window_open
			has_country_flag = chaos_meter_history_tab
		}
		rebuild_chaos_meter_history_view = yes
	}
}

add_chaos_meter_value = {
    # Input: temp variable chaos_change

    # Check if chaos meter is disabled
    if = {
        limit = { has_global_flag = settings_chaos_meter_disabled }
        log = "CHAOS METER: Update blocked - chaos meter is disabled"
    }
    else_if = {
        limit = { check_variable = { global.chaos_meter_value < constant:chaos_meter_tier_range.tier_final.min } }

        round_temp_variable = chaos_change

        set_temp_variable = { generic_history_reason = constant:chaos_meter_history_reason.generic.increase }
        if = {
            limit = { check_variable = { chaos_change < 0 } }
            set_temp_variable = { generic_history_reason = constant:chaos_meter_history_reason.generic.decrease }
        }

        if = {
            limit = {
                has_variable = chaos_history_reason_custom
                check_variable = { chaos_history_reason_custom > 0 }
                has_variable = chaos_history_reason
            }
            set_temp_variable = { active_history_reason = chaos_history_reason }
        }
        else = {
            set_temp_variable = { active_history_reason = generic_history_reason }
        }
        set_temp_variable = { chaos_history_reason = active_history_reason }
        set_temp_variable = { chaos_history_reason_custom = 0 }

        if = {
            limit = { NOT = { has_variable = chaos_history_target_count } }
            set_temp_variable = { chaos_history_target_count = 1 }
        }

        add_to_variable = { global.chaos_meter_value = chaos_change }

        clamp_variable = {
            var = global.chaos_meter_value
            min = 0
        }

        round_variable = global.chaos_meter_value
        update_chaos_meter_tier = yes

        if = {
            limit = {
                OR = {
                    check_variable = { chaos_change > 0 }
                    check_variable = { chaos_change < 0 }
                }
            }
            record_chaos_meter_history_entry = yes
        }

        if = {
            limit = { check_variable = { chaos_change > 0 } }
            log = "CHAOS METER: Increased by [?chaos_change] to [?global.chaos_meter_value]"
        }
        else = {
            set_temp_variable = { abs_change = chaos_change }
            multiply_temp_variable = { abs_change = -1 }
            log = "CHAOS METER: Decreased by [?abs_change] to [?global.chaos_meter_value]"
        }
    }
    else = {
        log = "CHAOS METER: World Collapse active - chaos meter locked at [?global.chaos_meter_value]"
    }
}

# Clear all tier flags (no longer needed with integer flag system)

    clr_global_flag = chaos_tier

update_chaos_meter_tier = {

    # Store current tier state for change detection
    set_temp_variable = { old_tier_value = 0 }
    if = { limit = { has_global_flag = { flag = chaos_tier value = 5 } } set_temp_variable = { old_tier_value = 5 } }
    else_if = { limit = { has_global_flag = { flag = chaos_tier value = 4 } } set_temp_variable = { old_tier_value = 4 } }
    else_if = { limit = { has_global_flag = { flag = chaos_tier value = 3 } } set_temp_variable = { old_tier_value = 3 } }
    else_if = { limit = { has_global_flag = { flag = chaos_tier value = 2 } } set_temp_variable = { old_tier_value = 2 } }
    else_if = { limit = { has_global_flag = { flag = chaos_tier value = 1 } } set_temp_variable = { old_tier_value = 1 } }

    # Clear the flag and set new value
    clr_global_flag = chaos_tier

    if = {
        limit = { check_variable = { var = global.chaos_meter_value value = constant:chaos_meter_tier_range.tier_final.min compare = greater_than_or_equals } }
        set_global_flag = { flag = chaos_tier value = 5 }
        if = {
            limit = { NOT = { check_variable = { old_tier_value = 5 } } }
            log = "CHAOS METER: Tier changed to WORLD COLLAPSE"
        }
    }

air_contamination_sync_chaos_from_delta = {
	if = {
		limit = {
			OR = {
				check_variable = { air_contamination_delta_bp > 0 }
				check_variable = { air_contamination_delta_bp < 0 }
			}
		}
		add_to_variable = { global.air_contamination_chaos_buffer_bp = air_contamination_delta_bp }

		set_temp_variable = { chaos_change = 0 }

		while_loop_effect = {
			limit = {
				check_variable = {
					var = global.air_contamination_chaos_buffer_bp
					value = 100
					compare = greater_than_or_equals
				}
			}
			add_to_temp_variable = { chaos_change = 1 }
			subtract_from_variable = { global.air_contamination_chaos_buffer_bp = 100 }
		}

		while_loop_effect = {
			limit = {
				check_variable = {
					var = global.air_contamination_chaos_buffer_bp
					value = -100
					compare = less_than_or_equals
				}
			}
			subtract_from_temp_variable = { chaos_change = 1 }
			add_to_variable = { global.air_contamination_chaos_buffer_bp = 100 }
		}

		if = {
			limit = {
				OR = {
					check_variable = { chaos_change > 0 }
					check_variable = { chaos_change < 0 }
				}
			}
			set_temp_variable = { chaos_history_reason = constant:chaos_meter_history_reason.world_state.air_contamination_rise }
			if = {
				limit = { check_variable = { chaos_change < 0 } }
				set_temp_variable = { chaos_history_reason = constant:chaos_meter_history_reason.world_state.air_contamination_fall }
			}

			if = {
				limit = {
					check_variable = { chaos_change > 0 }
					has_variable = air_chaos_reason_increase
					check_variable = { air_chaos_reason_increase > 0 }
				}
				set_temp_variable = { chaos_history_reason = air_chaos_reason_increase }
			}
			if = {
				limit = {
					check_variable = { chaos_change < 0 }
					has_variable = air_chaos_reason_decrease
					check_variable = { air_chaos_reason_decrease > 0 }
				}
				set_temp_variable = { chaos_history_reason = air_chaos_reason_decrease }
			}

			set_temp_variable = { chaos_history_reason_custom = 1 }
			add_chaos_meter_value = yes
		}
	}
}

		set_temp_variable = { chaos_state_deaths_percent = ROOT.chaos_nuke_state_deaths_percent }
		set_temp_variable = { chaos_state_deaths_mult = 1 }
		chaos_meter_apply_nuclear_state_death_environment_multiplier = yes

		chaos_meter_register_state_civilian_deaths_percent = yes
		chaos_meter_apply_nuclear_fallout_to_state = yes
	}

	if = {
		limit = {
			check_variable = {
				var = air_cleanliness_system_enabled
				value = 0
				compare = greater_than
			}
		}
		set_temp_variable = { air_contamination_delta_bp = constant:air_contamination_delta_bp.nuke_normal }
		set_temp_variable = { air_chaos_reason_increase = constant:chaos_meter_history_reason.nuke.base }
		if = {
			limit = { has_tech = thermonuclear_bombs }
			set_temp_variable = { air_contamination_delta_bp = constant:air_contamination_delta_bp.nuke_thermonuclear }
			set_temp_variable = { air_chaos_reason_increase = constant:chaos_meter_history_reason.nuke.thermonuclear }
			log = "AIR CONTAMINATION: Thermonuclear strike by [THIS.GetName] on [FROM.GetName] (+[?air_contamination_delta_bp] bp)"
		}
		else = {
			log = "AIR CONTAMINATION: Nuclear strike by [THIS.GetName] on [FROM.GetName] (+[?air_contamination_delta_bp] bp)"
		}

		set_variable = { global.air_contamination_last_nuke_bp = air_contamination_delta_bp }
		air_contamination_apply_delta_bp = yes
		air_contamination_apply_country_modifier = yes
	}
	else = {

chaos_meter_update_world_tension = {
    set_temp_variable = { current_world_tension = global.threat }
    multiply_temp_variable = { current_world_tension = 100 }
    round_temp_variable = current_world_tension
    set_variable = { global.current_world_tension = current_world_tension }

    if = {
        limit = { check_variable = { current_world_tension > global.highest_world_tension_recorded } }
        set_temp_variable = { tension_increase = current_world_tension }
	        subtract_from_temp_variable = { tension_increase = global.highest_world_tension_recorded }

	        if = {
	            limit = { check_variable = { var = tension_increase value = constant:chaos_meter_update_threshold.min_tension_change compare = greater_than_or_equals } }
	            set_temp_variable = { chaos_change = tension_increase }
	            multiply_temp_variable = { chaos_change = constant:chaos_meter_delta.world_tension.base }
	            set_temp_variable = { chaos_history_reason = constant:chaos_meter_history_reason.world_state.world_tension_rise }
	            set_temp_variable = { chaos_history_reason_custom = 1 }
	            add_chaos_meter_value = yes
	            log = "CHAOS METER: World tension increased from [?global.highest_world_tension_recorded]% to [?current_world_tension]% ([?chaos_change|=] chaos)"
	            set_variable = { global.highest_world_tension_recorded = current_world_tension }
	        }
	    }
    else_if = {
        limit = { check_variable = { current_world_tension < global.highest_world_tension_recorded } }
        set_temp_variable = { tension_decrease = global.highest_world_tension_recorded }
	        subtract_from_temp_variable = { tension_decrease = current_world_tension }

	        if = {
	            limit = { check_variable = { var = tension_decrease value = constant:chaos_meter_update_threshold.min_tension_change compare = greater_than_or_equals } }
	            round_temp_variable = tension_decrease
	            set_temp_variable = { chaos_change = tension_decrease }
	            multiply_temp_variable = { chaos_change = constant:chaos_meter_delta.world_tension.base }
	            multiply_temp_variable = { chaos_change = -1 }
	            set_temp_variable = { chaos_history_reason = constant:chaos_meter_history_reason.world_state.world_tension_fall }
	            set_temp_variable = { chaos_history_reason_custom = 1 }
	            add_chaos_meter_value = yes

            set_temp_variable = { wt_decreased_from = current_world_tension }
            add_to_temp_variable = { wt_decreased_from = tension_decrease }
            log = "CHAOS METER: World tension decreased from [?wt_decreased_from]% to [?current_world_tension]% ([?chaos_change|=] chaos)"
            set_variable = { global.highest_world_tension_recorded = current_world_tension }
        }
    }
}

chaos_meter_update_casualties = {
	set_temp_variable = { current_casualties = 0 }
	every_country = {
		add_to_temp_variable = { current_casualties = casualties }
		if = {
			limit = { NOT = { has_variable = chaos_meter_last_casualty_count } }
			set_variable = { chaos_meter_last_casualty_count = casualties }
		}

		set_temp_variable = { casualty_increase = casualties }
		subtract_from_temp_variable = { casualty_increase = chaos_meter_last_casualty_count }

		if = {
			limit = { check_variable = { casualty_increase > 0 } }
			set_temp_variable = { chaos_deaths_change = casualty_increase }
			set_temp_variable = { chaos_deaths_reason = constant:chaos_meter_deaths_reason.military_casualties }
			set_temp_variable = { chaos_deaths_is_civilian = 0 }
			set_temp_variable = { chaos_deaths_target_country = THIS }
			set_temp_variable = { chaos_deaths_has_target_country = 1 }
			chaos_meter_register_deaths = yes
			log = "CHAOS METER: Military casualties increased for [This.GetNameDef] by [?casualty_increase|.0], total deaths now [?global.chaos_meter_deaths_total|.0]"
		}

		set_variable = { chaos_meter_last_casualty_count = casualties }
	}

	set_variable = { global.last_casualty_count = current_casualties }
}

chaos_meter_update_stability = {
    set_temp_variable = { current_total_stability = 0 }
    set_temp_variable = { current_country_count = 0 }

    every_country = {
        add_to_temp_variable = { current_total_stability = stability }
        add_to_temp_variable = { current_country_count = 1 }
    }

    set_temp_variable = { current_average_stability = 0 }
    if = {
        limit = { check_variable = { current_country_count > 0 } }
        set_temp_variable = { current_average_stability = current_total_stability }
        divide_temp_variable = { current_average_stability = current_country_count }
    }

    set_temp_variable = { stability_change = current_average_stability }
    subtract_from_temp_variable = { stability_change = global.last_average_stability }
    set_temp_variable = { abs_stability_change = stability_change }

    if = {
        limit = { check_variable = { stability_change < 0 } }
        multiply_temp_variable = { abs_stability_change = -1 }
	    }

	    if = {
	        limit = { check_variable = { abs_stability_change > constant:chaos_meter_scale_fp.stability_change } }
	        set_temp_variable = { chaos_from_stability = stability_change }
	        divide_temp_variable = { chaos_from_stability = constant:chaos_meter_scale_fp.stability_change }
	        round_temp_variable = chaos_from_stability
	        set_temp_variable = { chaos_change = chaos_from_stability }
	        multiply_temp_variable = { chaos_change = -1 }

	        if = {
	            limit = {
	                OR = {
	                    check_variable = { var = chaos_change value = constant:chaos_meter_update_threshold.min_casualty_change_k compare = greater_than_or_equals }
	                    AND = {
	                        check_variable = { chaos_change < 0 }
	                        set_temp_variable = { neg_threshold = constant:chaos_meter_update_threshold.min_casualty_change_k }
	                        multiply_temp_variable = { neg_threshold = -1 }
	                        check_variable = { var = chaos_change value = neg_threshold compare = less_than_or_equals }
	                    }
	                }
	            }
	            if = {
	                limit = { check_variable = { stability_change > 0 } }
	                set_temp_variable = { chaos_history_reason = constant:chaos_meter_history_reason.world_state.stability_rise }
	            }
	            else = {
	                set_temp_variable = { chaos_history_reason = constant:chaos_meter_history_reason.world_state.stability_fall }
	            }
	            set_temp_variable = { chaos_history_reason_custom = 1 }
            add_chaos_meter_value = yes

            if = {
                limit = { check_variable = { stability_change > 0 } }
                log = "CHAOS METER: World stability improved by [?abs_stability_change] ([?chaos_change|=] chaos)"
            }
            else = {
                log = "CHAOS METER: World stability decreased by [?abs_stability_change] ([?chaos_change|=] chaos)"
            }
        }

        set_variable = { global.last_average_stability = current_average_stability }
    }
}
