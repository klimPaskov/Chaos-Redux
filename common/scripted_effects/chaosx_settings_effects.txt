append_event_id_manual_digit = {
    if = {
        limit = { NOT = { has_country_flag = event_id_manual_entry_active } }
        set_country_flag = event_id_manual_entry_active
    }

    if = {
        limit = { has_variable = settings_event_id_manual_buffer }
        set_temp_variable = { new_event_id_manual_buffer = settings_event_id_manual_buffer }
        multiply_temp_variable = { new_event_id_manual_buffer = 10 }
        add_to_temp_variable = { new_event_id_manual_buffer = event_id_manual_digit }
        set_variable = { settings_event_id_manual_buffer = new_event_id_manual_buffer }
    }
    else = {
        set_variable = { settings_event_id_manual_buffer = event_id_manual_digit }
    }

    update_event_id_display = yes
}

apply_event_id_manual_entry = {
    set_temp_variable = { event_id_manual_entry_applied = 0 }

    if = {
        limit = { has_variable = settings_event_id_manual_buffer }
        if = {
            limit = {
                check_variable = { settings_event_id_manual_buffer > 0 }
                OR = {
                    check_variable = { settings_event_id_manual_buffer < global.all_events^num }
                    check_variable = { settings_event_id_manual_buffer = global.all_events^num }
                }
            }
            set_variable = { settings_event_id_filter = constant:settings_event_filter.all }
            set_variable = { settings_selected_event_id = settings_event_id_manual_buffer }
            set_temp_variable = { event_id_manual_entry_applied = 1 }
        }
    }

    clr_country_flag = event_id_manual_entry_active
    clear_variable = settings_event_id_manual_buffer
    update_event_id_display = yes
}

            has_variable = settings_numeric_manual_target
        }

        if = {
            limit = { check_variable = { settings_numeric_manual_target = constant:settings_numeric_input_target.timer_min } }
            set_temp_variable = { new_timer_min = settings_numeric_manual_buffer }
            clamp_temp_variable = {
                var = new_timer_min
                min = constant:settings_timer_limit.min_min_value
                max = settings_timer_max_days
            }

            calculate_dynamic_caps = yes

            limit = { check_variable = { settings_numeric_manual_target = constant:settings_numeric_input_target.timer_max } }
            set_temp_variable = { new_timer_max = settings_numeric_manual_buffer }

                max = constant:settings_timer_limit.max_max_value

# Cycle to previous event filter type
cycle_event_filter_prev = {
    reset_event_id_manual_entry = yes

    if = {
        limit = { check_variable = { settings_event_id_filter = constant:settings_event_filter.all } }
        set_variable = { settings_event_id_filter = constant:settings_event_filter.fire_once }
    }
    else = {
        subtract_from_variable = { settings_event_id_filter = 1 }
    }

    adjust_event_id_for_filter = yes
    update_event_id_display = yes
}

# Cycle to next event filter type
cycle_event_filter_next = {
    reset_event_id_manual_entry = yes

    if = {
        limit = { check_variable = { settings_event_id_filter = constant:settings_event_filter.fire_once } }
        set_variable = { settings_event_id_filter = constant:settings_event_filter.all }
    }
    else = {
        add_to_variable = { settings_event_id_filter = 1 }
    }

    adjust_event_id_for_filter = yes
    update_event_id_display = yes
}

# Find index of event in specified array
find_event_in_array = {
    # Input: event_id, array_filter_type (1=major, 2=repeatable, 3=fire_once)
    # Output: temp_event_index (-1 if not found)

    set_temp_variable = { temp_event_index = -1 }

    if = {
        limit = { check_variable = { array_filter_type = constant:settings_event_filter.major } }
        # Major events
        for_loop_effect = {
            start = 0
            end = global.major_events^num
            value = i
            if = {
                limit = { check_variable = { event_id = global.major_events^i } }
                set_temp_variable = { temp_event_index = i }
            }
        }
    }
    else_if = {
        limit = { check_variable = { array_filter_type = constant:settings_event_filter.repeatable } }
        # Repeatable events
        for_loop_effect = {
            start = 0
            end = global.repeatable_events^num
            value = i
            if = {
                limit = { check_variable = { event_id = global.repeatable_events^i } }
                set_temp_variable = { temp_event_index = i }
            }
        }
    }
    else_if = {
        limit = { check_variable = { array_filter_type = constant:settings_event_filter.fire_once } }
        # Fire-once events
        for_loop_effect = {
            start = 0
            end = global.fire_once_events^num
            value = i
            if = {
                limit = { check_variable = { event_id = global.fire_once_events^i } }
                set_temp_variable = { temp_event_index = i }
            }
        }
    }
}

# Adjust the current event ID to be valid for the selected filter
adjust_event_id_for_filter = {
    if = {
        limit = { check_variable = { settings_event_id_filter = constant:settings_event_filter.all } }
        # All events - ensure ID is within valid range
        clamp_variable = {
            var = settings_selected_event_id
            min = 1
            max = global.all_events^num
        }
    }
    else_if = {
        limit = { check_variable = { settings_event_id_filter = constant:settings_event_filter.major } }
        # Major events - check if current ID is major, if not set to first major
        set_temp_variable = { event_id = settings_selected_event_id }
        set_temp_variable = { array_filter_type = constant:settings_event_filter.major }
        find_event_in_array = yes

        if = {
            limit = { check_variable = { temp_event_index = -1 } }
            if = {
                limit = { check_variable = { global.major_events^num > 0 } }
                set_variable = { settings_selected_event_id = global.major_events^0 }
            }
            else = {
                set_variable = { settings_selected_event_id = global.major_events^0 }
            }
        }
    }
    else_if = {
        limit = { check_variable = { settings_event_id_filter = constant:settings_event_filter.repeatable } }
        # Repeatable events
        set_temp_variable = { event_id = settings_selected_event_id }
        set_temp_variable = { array_filter_type = constant:settings_event_filter.repeatable }
        find_event_in_array = yes

        if = {
            limit = { check_variable = { temp_event_index = -1 } }
            if = {
                limit = { check_variable = { global.repeatable_events^num > 0 } }
                set_variable = { settings_selected_event_id = global.repeatable_events^0 }
            }
            else = {
                set_variable = { settings_selected_event_id = global.repeatable_events^0 }
            }
        }
    }
    else_if = {
        limit = { check_variable = { settings_event_id_filter = constant:settings_event_filter.fire_once } }
        # Fire-once events
        set_temp_variable = { event_id = settings_selected_event_id }
        set_temp_variable = { array_filter_type = constant:settings_event_filter.fire_once }
        find_event_in_array = yes

        if = {
            limit = { check_variable = { temp_event_index = -1 } }
            if = {
                limit = { check_variable = { global.fire_once_events^num > 0 } }
                set_variable = { settings_selected_event_id = global.fire_once_events^0 }
            }
            else = {
                set_variable = { settings_selected_event_id = global.fire_once_events^0 }
            }
        }
    }
}

# Get the next valid event ID for the current filter
get_next_valid_event_id = {
    # Simple implementation: set to next array element or wrap to first
    if = {
        limit = { check_variable = { settings_event_id_filter = constant:settings_event_filter.major } }
        # Major events - cycle through major events array
        if = {
            limit = { check_variable = { global.major_events^num > 0 } }

            set_temp_variable = { current_index = -1 }
            # Find current index in major events array
            for_loop_effect = {
                start = 0
                end = global.major_events^num
                value = i
                if = {
                    limit = { check_variable = { settings_selected_event_id = global.major_events^i } }
                    set_temp_variable = { current_index = i }
                }
            }

            # Move to next or wrap to first
            if = {
                limit = { check_variable = { current_index = -1 } }
                # Not found, go to first
                set_variable = { settings_selected_event_id = global.major_events^0 }
            }
            else = {
                add_to_temp_variable = { current_index = 1 }
                if = {
                    limit = { check_variable = { current_index < global.major_events^num } }
                    set_variable = { settings_selected_event_id = global.major_events^current_index }
                }
                else = {
                    # Wrap to first
                    set_variable = { settings_selected_event_id = global.major_events^0 }
                }

        }
    }
    else_if = {
        limit = { check_variable = { settings_event_id_filter = constant:settings_event_filter.repeatable } }

                if = {
                    limit = { check_variable = { current_index < global.repeatable_events^num } }
                    set_variable = { settings_selected_event_id = global.repeatable_events^current_index }
                }
                else = {
                    # Wrap to first
                    set_variable = { settings_selected_event_id = global.repeatable_events^0 }
                }
            }
        }
    }

            }
            else_if = {
                limit = { check_variable = { current_index = 0 } }
                # At first, wrap to last
                set_temp_variable = { last_index = global.repeatable_events^num }
                subtract_from_temp_variable = { last_index = 1 }
                set_variable = { settings_selected_event_id = global.repeatable_events^last_index }
            }
            else = {
                # Move to previous
                subtract_from_temp_variable = { current_index = 1 }
                set_variable = { settings_selected_event_id = global.repeatable_events^current_index }
            }
        }
    }
    else_if = {
        limit = { check_variable = { settings_event_id_filter = constant:settings_event_filter.fire_once } }
        # Fire-once events - cycle through fire-once events array backwards
        if = {
            limit = { check_variable = { global.fire_once_events^num > 0 } }

            set_temp_variable = { current_index = -1 }
            # Find current index in fire-once events array
            for_loop_effect = {
                start = 0
                end = global.fire_once_events^num
                value = i
                if = {
                    limit = { check_variable = { settings_selected_event_id = global.fire_once_events^i } }
                    set_temp_variable = { current_index = i }
                }
            }

            # Move to previous or wrap to last
            if = {
                limit = { check_variable = { current_index = -1 } }
                # Not found, go to last
                set_temp_variable = { last_index = global.fire_once_events^num }
                subtract_from_temp_variable = { last_index = 1 }
                set_variable = { settings_selected_event_id = global.fire_once_events^last_index }
            }
            else_if = {
                limit = { check_variable = { current_index = 0 } }
                # At first, wrap to last
                set_temp_variable = { last_index = global.fire_once_events^num }
                subtract_from_temp_variable = { last_index = 1 }
                set_variable = { settings_selected_event_id = global.fire_once_events^last_index }
            }
            else = {
                # Move to previous
                subtract_from_temp_variable = { current_index = 1 }
                set_variable = { settings_selected_event_id = global.fire_once_events^current_index }
            }
        }
    }
}

# Trigger selected event (checks if event can be fired unless force mode is enabled)
trigger_selected_event = {
    set_temp_variable = { event_id_manual_entry_can_trigger = 1 }

    if = {
        limit = { has_country_flag = event_id_manual_entry_active }
        set_temp_variable = { event_id_manual_entry_can_trigger = 0 }
        apply_event_id_manual_entry = yes
        if = {
            limit = { check_variable = { event_id_manual_entry_applied = 1 } }
            set_temp_variable = { event_id_manual_entry_can_trigger = 1 }
        }
    }

    if = {
        limit = {
            check_variable = { event_id_manual_entry_can_trigger = 1 }
            NOT = { check_variable = { settings_selected_event_id = constant:event_system_event_type.unknown } }
        }

        set_temp_variable = { event_id = settings_selected_event_id }

        if = {
            limit = {
                OR = {
                    has_country_flag = force_trigger_mode_enabled
                    has_variable = temp_bypass_checks
                }
            }
            fire_event_by_id = yes
            log = "CHAOS REDUX SETTINGS: FORCE triggered event ID [?settings_selected_event_id]"
        }
        else = {
            get_event_type = yes
            set_variable = { settings_event_type = event_type }

            set_temp_variable = { can_fire = 1 }
            if = {
                limit = {
                    OR = {
                        check_variable = { settings_event_type = constant:event_system_event_type.major }
                        check_variable = { settings_event_type = constant:event_system_event_type.fire_once }
                    }
                }
                if = {
                    limit = { is_in_array = { global.fired_events = event_id } }
                    set_temp_variable = { can_fire = 0 }
                }
            }

            if = {
                limit = { check_variable = { can_fire = 1 } }
                fire_event_by_id = yes
            }
            else = {
                log = "CHAOS REDUX SETTINGS: Cannot fire event [?event_id] - already fired. Enable force trigger mode to bypass this restriction."
            }
        }
    }
}

rebuild_events_log_events_view = {
    if = {
        limit = { NOT = { has_variable = settings_events_log_events_sort_mode } }
        set_variable = { settings_events_log_events_sort_mode = 0 }
    }
    if = {
        limit = { check_variable = { settings_events_log_events_sort_mode > 1 } }
        set_variable = { settings_events_log_events_sort_mode = 0 }
    }
    if = {
        limit = { check_variable = { settings_events_log_events_sort_mode < 0 } }
        set_variable = { settings_events_log_events_sort_mode = 0 }
    }
    if = {
        limit = { NOT = { has_variable = settings_events_log_events_filter_type } }
        set_variable = { settings_events_log_events_filter_type = 0 }
    }
    if = {
        limit = { check_variable = { settings_events_log_events_filter_type > 2 } }
        set_variable = { settings_events_log_events_filter_type = 0 }
    }
    if = {
        limit = { check_variable = { settings_events_log_events_filter_type < 0 } }
        set_variable = { settings_events_log_events_filter_type = 0 }
    }

    clear_array = global.events_log_view_sequence_entries
    clear_array = global.events_log_view_date_entries
    clear_array = global.events_log_view_event_id_entries
    clear_array = global.events_log_view_event_type_entries
    clear_array = global.events_log_view_actor_entries
    clear_array = global.events_log_view_has_actor_entries
    clear_array = global.events_log_events_view_fired_entries
    clear_array = global.events_log_events_view_enabled_entries
    clear_array = global.events_log_events_view_unique_entries

    clear_array = temp_events_log_events_source_event_id_entries
    clear_array = temp_events_log_events_source_event_type_entries
    clear_array = temp_events_log_events_source_fired_entries
    clear_array = temp_events_log_events_source_enabled_entries
    clear_array = temp_events_log_events_source_unique_entries

    set_temp_variable = { events_log_events_all_count = global.all_events^num }
    for_loop_effect = {
        start = 0
        end = events_log_events_all_count
        value = i

        set_temp_variable = { event_id = global.all_events^i }
        set_temp_variable = { events_log_events_is_enabled = 1 }
        if = {
            limit = { is_in_array = { global.disabled_events = event_id } }
            set_temp_variable = { events_log_events_is_enabled = 0 }
        }

        set_temp_variable = { events_log_events_include = 1 }
        if = {
            limit = {
                check_variable = { settings_events_log_events_filter_type = 1 }
                check_variable = { events_log_events_is_enabled = 0 }
            }
            set_temp_variable = { events_log_events_include = 0 }
        }
        else_if = {
            limit = {
                check_variable = { settings_events_log_events_filter_type = 2 }
                check_variable = { events_log_events_is_enabled > 0 }
            }
            set_temp_variable = { events_log_events_include = 0 }
        }

        if = {
            limit = { check_variable = { events_log_events_include > 0 } }
            get_event_type = yes

            set_temp_variable = { events_log_events_unique_flag = 0 }
            if = {
                limit = {
                    OR = {
                        check_variable = { event_type = constant:event_system_event_type.major }
                        check_variable = { event_type = constant:event_system_event_type.fire_once }
                    }
                }
                set_temp_variable = { events_log_events_unique_flag = 1 }
            }

            set_temp_variable = { events_log_events_fired_count = 0 }
            set_temp_variable = { events_log_events_history_scan_index = 0 }
            while_loop_effect = {
                limit = {
                    check_variable = {
                        events_log_events_history_scan_index < global.events_log_history_event_id_entries^num
                    }
                }
                if = {
                    limit = {
                        check_variable = {
                            var = global.events_log_history_event_id_entries^events_log_events_history_scan_index
                            value = event_id
                            compare = equals
                        }
                    }
                    add_to_temp_variable = { events_log_events_fired_count = 1 }
                }
                add_to_temp_variable = { events_log_events_history_scan_index = 1 }
            }

            add_to_array = {
                array = temp_events_log_events_source_event_id_entries
                value = event_id
            }
            add_to_array = {
                array = temp_events_log_events_source_event_type_entries
                value = event_type
            }
            add_to_array = {
                array = temp_events_log_events_source_fired_entries
                value = events_log_events_fired_count
            }
            add_to_array = {
                array = temp_events_log_events_source_enabled_entries
                value = events_log_events_is_enabled
            }
            add_to_array = {
                array = temp_events_log_events_source_unique_entries
                value = events_log_events_unique_flag
            }
        }
    }

    set_temp_variable = { events_log_events_source_count = temp_events_log_events_source_event_id_entries^num }
    set_temp_variable = { events_log_events_sorted_count = 0 }
    while_loop_effect = {
        limit = { check_variable = { events_log_events_sorted_count < events_log_events_source_count } }

        set_temp_variable = { events_log_events_best_index = -1 }
        set_temp_variable = { events_log_events_scan_index = 0 }
        while_loop_effect = {
            limit = { check_variable = { events_log_events_scan_index < events_log_events_source_count } }
            if = {
                limit = {
                    check_variable = {
                        var = temp_events_log_events_source_event_id_entries^events_log_events_scan_index
                        value = 0
                        compare = greater_than
                    }
                }

                set_temp_variable = { events_log_events_candidate_metric = temp_events_log_events_source_event_id_entries^events_log_events_scan_index }
                if = {
                    limit = { check_variable = { settings_events_log_events_sort_mode = 1 } }
                    set_temp_variable = { events_log_events_candidate_metric = temp_events_log_events_source_fired_entries^events_log_events_scan_index }
                }

                if = {
                    limit = { check_variable = { events_log_events_best_index < 0 } }
                    set_temp_variable = { events_log_events_best_index = events_log_events_scan_index }
                    set_temp_variable = { events_log_events_best_metric = events_log_events_candidate_metric }
                    set_temp_variable = { events_log_events_best_event_id = temp_events_log_events_source_event_id_entries^events_log_events_scan_index }
                }
                else_if = {
                    limit = {
                        has_country_flag = events_log_events_sort_descending
                        check_variable = { events_log_events_candidate_metric > events_log_events_best_metric }
                    }
                    set_temp_variable = { events_log_events_best_index = events_log_events_scan_index }
                    set_temp_variable = { events_log_events_best_metric = events_log_events_candidate_metric }
                    set_temp_variable = { events_log_events_best_event_id = temp_events_log_events_source_event_id_entries^events_log_events_scan_index }
                }
                else_if = {
                    limit = {
                        NOT = { has_country_flag = events_log_events_sort_descending }
                        check_variable = { events_log_events_candidate_metric < events_log_events_best_metric }
                    }
                    set_temp_variable = { events_log_events_best_index = events_log_events_scan_index }
                    set_temp_variable = { events_log_events_best_metric = events_log_events_candidate_metric }
                    set_temp_variable = { events_log_events_best_event_id = temp_events_log_events_source_event_id_entries^events_log_events_scan_index }
                }
                else_if = {
                    limit = { check_variable = { events_log_events_candidate_metric = events_log_events_best_metric } }
                    if = {
                        limit = {
                            has_country_flag = events_log_events_sort_descending
                            check_variable = {
                                var = temp_events_log_events_source_event_id_entries^events_log_events_scan_index
                                value = events_log_events_best_event_id
                                compare = greater_than
                            }
                        }
                        set_temp_variable = { events_log_events_best_index = events_log_events_scan_index }
                        set_temp_variable = { events_log_events_best_metric = events_log_events_candidate_metric }
                        set_temp_variable = { events_log_events_best_event_id = temp_events_log_events_source_event_id_entries^events_log_events_scan_index }
                    }
                    else_if = {
                        limit = {
                            NOT = { has_country_flag = events_log_events_sort_descending }
                            check_variable = {
                                var = temp_events_log_events_source_event_id_entries^events_log_events_scan_index
                                value = events_log_events_best_event_id
                                compare = less_than
                            }
                        }
                        set_temp_variable = { events_log_events_best_index = events_log_events_scan_index }
                        set_temp_variable = { events_log_events_best_metric = events_log_events_candidate_metric }
                        set_temp_variable = { events_log_events_best_event_id = temp_events_log_events_source_event_id_entries^events_log_events_scan_index }
                    }
                }
            }
            add_to_temp_variable = { events_log_events_scan_index = 1 }
        }

        if = {
            limit = { check_variable = { events_log_events_best_index > -1 } }
            add_to_array = {
                array = global.events_log_view_event_id_entries
                value = temp_events_log_events_source_event_id_entries^events_log_events_best_index
            }
            add_to_array = {
                array = global.events_log_view_event_type_entries
                value = temp_events_log_events_source_event_type_entries^events_log_events_best_index
            }
            add_to_array = {
                array = global.events_log_events_view_fired_entries
                value = temp_events_log_events_source_fired_entries^events_log_events_best_index
            }
            add_to_array = {
                array = global.events_log_events_view_enabled_entries
                value = temp_events_log_events_source_enabled_entries^events_log_events_best_index
            }
            add_to_array = {
                array = global.events_log_events_view_unique_entries
                value = temp_events_log_events_source_unique_entries^events_log_events_best_index
            }

            set_variable = { temp_events_log_events_source_event_id_entries^events_log_events_best_index = -1 }
            add_to_temp_variable = { events_log_events_sorted_count = 1 }
        }
        else = {
            set_temp_variable = { events_log_events_sorted_count = events_log_events_source_count }
        }
    }

# Get the next valid event ID by stepping through N positions in the filtered array
get_next_valid_event_id_by_steps = {
    set_temp_variable = { steps_to_move = settings_events_increment_mode }

    if = {
        limit = { check_variable = { settings_event_id_filter = constant:settings_event_filter.major } }
        # Major events - move by N steps in major events array
        if = {
            limit = { check_variable = { global.major_events^num > 0 } }

            set_temp_variable = { current_index = -1 }
            # Find current index in major events array
            for_loop_effect = {
                start = 0
                end = global.major_events^num
                value = i
                if = {
                    limit = { check_variable = { settings_selected_event_id = global.major_events^i } }
                    set_temp_variable = { current_index = i }
                }
            }

            # Move forward by steps_to_move or wrap around

            if = {
                limit = { check_variable = { current_index = -1 } }
                # Not found, go to first
                set_variable = { settings_selected_event_id = global.major_events^0 }
            }
            else = {
                add_to_temp_variable = { current_index = steps_to_move }
                # Handle wrapping
                if = {
                    limit = { OR = { check_variable = { current_index = global.major_events^num } check_variable = { current_index > global.major_events^num } } }
                    set_temp_variable = { wrap_amount = current_index }
                    subtract_from_temp_variable = { wrap_amount = global.major_events^num }
                    set_temp_variable = { current_index = wrap_amount }
                }
                set_variable = { settings_selected_event_id = global.major_events^current_index }
            }
        }
    }
    else_if = {
        limit = { check_variable = { settings_event_id_filter = constant:settings_event_filter.repeatable } }
        # Repeatable events - move by N steps in repeatable events array

                # Handle wrapping
                if = {
                    limit = { OR = { check_variable = { current_index > global.repeatable_events^num } check_variable = { current_index = global.repeatable_events^num } } }
                    set_temp_variable = { wrap_amount = current_index }
                    subtract_from_temp_variable = { wrap_amount = global.repeatable_events^num }
                    set_temp_variable = { current_index = wrap_amount }
                }
                set_variable = { settings_selected_event_id = global.repeatable_events^current_index }
            }
        }
    }
    else_if = {
        limit = { check_variable = { settings_event_id_filter = constant:settings_event_filter.fire_once } }
        # Fire-once events - move by N steps in fire-once events array

    set_variable = { global.unfired_major_events_count = 0 }
    for_loop_effect = {
        start = 0
        end = global.major_events^num
        value = i
        set_temp_variable = { check_event_id = global.major_events^i }

            limit = { NOT = { is_in_array = { global.fired_events = check_event_id } } }
            add_to_variable = { global.unfired_major_events_count = 1 }
        }
    }

    # Count unfired fire-once events
    set_variable = { global.unfired_fire_once_events_count = 0 }
    for_loop_effect = {
        start = 0
        end = global.fire_once_events^num
        value = i
        set_temp_variable = { check_event_id = global.fire_once_events^i }
        if = {
            limit = { NOT = { is_in_array = { global.fired_events = check_event_id } } }
            add_to_variable = { global.unfired_fire_once_events_count = 1 }
        }
    }

    # Count repeatable events that have never fired
    set_variable = { global.unfired_repeatable_events_count = 0 }
    for_loop_effect = {
        start = 0
        end = global.repeatable_events^num
        value = i
        set_temp_variable = { check_event_id = global.repeatable_events^i }
        set_temp_variable = { event_ever_fired = 0 }

        for_loop_effect = {
            start = 0
            end = global.fired_events^num
            value = j
            if = {
                limit = { check_variable = { global.fired_events^j = check_event_id } }
                set_temp_variable = { event_ever_fired = 1 }
            }
        }

        if = {
            limit = { check_variable = { event_ever_fired = 0 } }
            add_to_variable = { global.unfired_repeatable_events_count = 1 }
        }
    }

    # Total unique events left across all categories
    set_variable = { global.unfired_total_events_count = global.unfired_major_events_count }
    add_to_variable = { global.unfired_total_events_count = global.unfired_fire_once_events_count }
    add_to_variable = { global.unfired_total_events_count = global.unfired_repeatable_events_count }
}

# Calculate dynamic caps based on minValue and maxValue inputs
calculate_dynamic_caps = {
    # Input variables: settings_timer_min_days, settings_timer_max_days
    # Output variables: settings_timer_min_decrement_cap, settings_timer_max_decrement_cap

    set_temp_variable = { min_value = settings_timer_min_days }
    set_temp_variable = { max_value = settings_timer_max_days }

    # Calculate minDecrementCap = minValue - 10 (floor at 0)
    set_temp_variable = { calculated_min_cap = min_value }
    subtract_from_temp_variable = { calculated_min_cap = 10 }

    clamp_temp_variable = {
        var = calculated_min_cap
        min = 0
    }

    # Calculate maxDecrementCap based on maxValue
    if = {
        limit = { check_variable = { max_value < constant:settings_dynamic_cap.floor_threshold } }
        set_temp_variable = { calculated_max_cap = 0 }
    }
    else = {
        # Calculate deviation from base value
        set_temp_variable = { deviation = max_value }
        subtract_from_temp_variable = { deviation = constant:settings_defaults.events_max_days }

        # Calculate steps: deviation // 3
        set_temp_variable = { steps = 0 }
        set_temp_variable = { abs_deviation = deviation }

        if = {
            limit = { check_variable = { deviation < 0 } }
            multiply_temp_variable = { abs_deviation = -1 }
        }

        # Integer division by 3 using repeated subtraction
        while_loop_effect = {
            limit = { check_variable = { abs_deviation > 2 } }
            add_to_temp_variable = { steps = 1 }
            subtract_from_temp_variable = { abs_deviation = constant:settings_dynamic_cap.step_size }
        }

        # Apply the formula: base 5 + (or -) steps
        set_temp_variable = { calculated_max_cap = constant:settings_dynamic_cap.base_factor }

        if = {
            limit = { check_variable = { deviation < 0 } }
            subtract_from_temp_variable = { calculated_max_cap = steps }
        }
        else = {
            add_to_temp_variable = { calculated_max_cap = steps }
        }

        clamp_temp_variable = {
            var = calculated_max_cap
            min = 0
        }
    }

    # Apply gap-based clamping
    set_temp_variable = { gap = max_value }
    subtract_from_temp_variable = { gap = min_value }

    clamp_temp_variable = {
        var = calculated_max_cap
        max = gap
    }

    # Apply final constraint (minDecrementCap + maxDecrementCap ≤ maxValue)
    set_temp_variable = { total_caps = calculated_min_cap }
    add_to_temp_variable = { total_caps = calculated_max_cap }

    if = {
        limit = { check_variable = { total_caps > max_value } }
        set_temp_variable = { allowed_max_cap = max_value }
        subtract_from_temp_variable = { allowed_max_cap = calculated_min_cap }

        clamp_temp_variable = {
            var = allowed_max_cap
            min = 0
        }

        if = {
            limit = { check_variable = { allowed_max_cap = 0 } }
            set_temp_variable = { calculated_min_cap = max_value }
        }

        set_temp_variable = { calculated_max_cap = allowed_max_cap }
    }

    # Set final calculated values
    set_variable = { settings_timer_min_decrement_cap = calculated_min_cap }
    set_variable = { settings_timer_max_decrement_cap = calculated_max_cap }
}

# Get the current filtered country list based on filter and sort settings

        for_loop_effect = {
            start = 0

        }

    else_if = {

                }

    set_temp_variable = { temp_current_list_size = temp_final_countries_list^num }

    resize_array = { global.oceania_countries = 0 }
    resize_array = { global.middle_east_countries = 0 }

    for_loop_effect = {
        start = 0
        end = global.countries^num
        value = i

        # Get the country at this index
        set_temp_variable = { current_country = global.countries^i }

        # Check which continent this country belongs to using capital state
        var:current_country = {
            if = {
                limit = { any_core_state = { is_on_continent = europe } }
                if = {
                    limit = { capital_scope = { is_on_continent = europe }  }
                    add_to_array = { global.europe_countries = THIS }
                }
            }
            else_if = {
                limit = { any_core_state = { is_on_continent = asia } }
                if = {
                    limit = { capital_scope = { is_on_continent = asia }  }
                    add_to_array = { global.asia_countries = THIS }
                }

                limit = { any_core_state = { is_on_continent = africa } }
                if = {
                    limit = { capital_scope = { is_on_continent = africa }  }
                    add_to_array = { global.africa_countries = THIS }
                }
            }
            else_if = {
                limit = { any_core_state = { is_on_continent = north_america } }
                if = {
                    limit = { capital_scope = { is_on_continent = north_america }  }
                    add_to_array = { global.north_america_countries = THIS }

                    add_to_variable = { global.enabled_countries_count = 1 }
                }
            }
        }
        update_enabled_countries_list = yes

# Disable event system for selected tag

    if = {
        limit = { has_variable = settings_current_selected_country }

            clr_country_flag = events_activated

            # Remove from enabled countries list
            if = {
                limit = { is_in_array = { global.enabled_countries_list = THIS } }
                ROOT = {
                    remove_from_array = { global.enabled_countries_list = PREV }
                    subtract_from_variable = { global.enabled_countries_count = 1 }
                }
            }
        }

        update_enabled_countries_list = yes

        if = {
            limit = {
                has_global_flag = settings_disable_previous_tag
                check_variable = { settings_current_selected_country = THIS }
            }
            log = "CHAOS REDUX TAG MANAGEMENT: Disabled event system for current player tag as requested"
        }
    }
    else = {
        log = "CHAOS REDUX TAG MANAGEMENT: Cannot disable - no country selected in current filter"
    }

# Update actual chaos meter to match preview settings
update_actual_chaos_meter = {
    reset_settings_numeric_manual_entry = yes

    set_temp_variable = { old_chaos_value = 0 }
    if = {
        limit = { has_variable = global.chaos_meter_value }
        set_temp_variable = { old_chaos_value = global.chaos_meter_value }
    }

    set_temp_variable = { chaos_change = settings_chaos_meter_value }
    subtract_from_temp_variable = { chaos_change = old_chaos_value }
	set_temp_variable = { chaos_history_reason = constant:chaos_meter_history_reason.settings.manual_update }

	# Set the actual chaos meter value
	set_variable = { global.chaos_meter_value = settings_chaos_meter_value }
	if = {
		limit = {
			date < 1940.01.01
			check_variable = {
				var = settings_chaos_meter_value
				value = constant:chaos_meter_tier_range.tier_1.min
				compare = greater_than_or_equals
			}
		}
		set_global_flag = achievement_chaos_reached_gathering_storm_pre_1940
	}

	# Update actual chaos meter tier
	update_chaos_meter_tier = yes
    if = {
        limit = {
            OR = {
                check_variable = { chaos_change > 0 }
                check_variable = { chaos_change < 0 }
            }
        }
        record_chaos_meter_history_entry = yes
    }

    log = "CHAOS REDUX SETTINGS: Chaos meter updated to [?settings_chaos_meter_value] via settings panel"
}

# Toggle chaos meter disabled state
toggle_chaos_meter_disabled = {
    if = {
        limit = { has_global_flag = settings_chaos_meter_disabled }
        # Re-enabling chaos meter
        clr_global_flag = settings_chaos_meter_disabled

	        # Restore the chaos meter value from backup if it exists
	        if = {
	            limit = { has_variable = global.chaos_meter_backup_value }
	            set_temp_variable = { old_chaos_value = 0 }
	            if = {
	                limit = { has_variable = global.chaos_meter_value }
	                set_temp_variable = { old_chaos_value = global.chaos_meter_value }
	            }
	            set_variable = { global.chaos_meter_value = global.chaos_meter_backup_value }
	            if = {
	                limit = {
	                    date < 1940.01.01
	                    check_variable = {
	                        var = global.chaos_meter_value
	                        value = constant:chaos_meter_tier_range.tier_1.min
	                        compare = greater_than_or_equals
	                    }
	                }
	                set_global_flag = achievement_chaos_reached_gathering_storm_pre_1940
	            }
	            set_temp_variable = { chaos_change = global.chaos_meter_value }
	            subtract_from_temp_variable = { chaos_change = old_chaos_value }
	            set_temp_variable = { chaos_history_reason = constant:chaos_meter_history_reason.settings.reenabled }
	            clear_variable = global.chaos_meter_backup_value

            # Update chaos meter tier after restoring value
            update_chaos_meter_tier = yes
            if = {
                limit = {
                    OR = {
                        check_variable = { chaos_change > 0 }
                        check_variable = { chaos_change < 0 }
                    }
                }
                record_chaos_meter_history_entry = yes
            }
        }

        log = "CHAOS REDUX SETTINGS: Chaos meter ENABLED - value restored to [?global.chaos_meter_value]"
    }
    else = {
        # Disabling chaos meter
        set_global_flag = settings_chaos_meter_disabled

        # Backup current chaos meter value before setting to 0
        if = {
            limit = { has_variable = global.chaos_meter_value }
            set_variable = { global.chaos_meter_backup_value = global.chaos_meter_value }
        }
        else = {
            set_variable = { global.chaos_meter_backup_value = 0 }
        }

        # Set chaos meter to 0 temporarily
        set_temp_variable = { old_chaos_value = 0 }
        if = {
            limit = { has_variable = global.chaos_meter_value }
            set_temp_variable = { old_chaos_value = global.chaos_meter_value }
        }
        set_variable = { global.chaos_meter_value = 0 }
        set_temp_variable = { chaos_change = global.chaos_meter_value }
        subtract_from_temp_variable = { chaos_change = old_chaos_value }
        set_temp_variable = { chaos_history_reason = constant:chaos_meter_history_reason.settings.disabled }

        update_chaos_meter_tier = yes
        if = {
            limit = {
                OR = {
                    check_variable = { chaos_change > 0 }
                    check_variable = { chaos_change < 0 }
                }
            }
            record_chaos_meter_history_entry = yes
        }

        log = "CHAOS REDUX SETTINGS: Chaos meter DISABLED - backed up value [?global.chaos_meter_backup_value] and set to 0"
    }
}

# ============================================================================
# RANDOM EVENT SELECTION EFFECTS
# ============================================================================
# Shared random event selection and firing helpers for the live event system
# and the settings menu.
#
# Inputs:
# - `random_event_selection_filter` temp variable using `settings_event_filter`
#   constants. Falls back to `settings_event_id_filter`, then `all`.
#
# Outputs:
# - `random_event_selection_id` temp variable with the chosen event ID.
# ============================================================================

ensure_random_event_selection_filter = {
    if = {
        limit = {
            NOT = { has_variable = random_event_selection_filter }
            has_variable = settings_event_id_filter
        }
        set_temp_variable = { random_event_selection_filter = settings_event_id_filter }
    }

    if = {
        limit = { NOT = { has_variable = random_event_selection_filter } }
        set_temp_variable = { random_event_selection_filter = constant:settings_event_filter.all }
    }
}

evaluate_random_event_selection_candidate = {
    set_temp_variable = { random_event_selection_candidate_is_valid = 1 }
    clear_variable = random_event_selection_candidate_weight_scaled

    if = {
        limit = {
            OR = {
                is_in_array = { global.disabled_events = random_event_selection_candidate_id }
                AND = {
                    is_in_array = { global.fired_events = random_event_selection_candidate_id }
                    NOT = { is_in_array = { global.repeatable_events = random_event_selection_candidate_id } }
                }
            }
        }
        set_temp_variable = { random_event_selection_candidate_is_valid = 0 }
    }

    if = {
        limit = {
            check_variable = { random_event_selection_candidate_is_valid = 1 }
            NOT = {
                OR = {
                    check_variable = { random_event_selection_filter = constant:settings_event_filter.all }
                    AND = {
                        check_variable = { random_event_selection_filter = constant:settings_event_filter.major }
                        is_in_array = { global.major_events = random_event_selection_candidate_id }
                    }
                    AND = {
                        check_variable = { random_event_selection_filter = constant:settings_event_filter.repeatable }
                        is_in_array = { global.repeatable_events = random_event_selection_candidate_id }
                    }
                    AND = {
                        check_variable = { random_event_selection_filter = constant:settings_event_filter.fire_once }
                        is_in_array = { global.fire_once_events = random_event_selection_candidate_id }
                    }
                }
            }
        }
        set_temp_variable = { random_event_selection_candidate_is_valid = 0 }
    }

    if = {
        limit = { check_variable = { random_event_selection_candidate_is_valid = 1 } }
        set_temp_variable = { random_event_selection_candidate_weight_scaled = global.event_weights^random_event_selection_candidate_id }
        multiply_temp_variable = { random_event_selection_candidate_weight_scaled = 100 }
        round_temp_variable = random_event_selection_candidate_weight_scaled

        if = {
            limit = { check_variable = { random_event_selection_candidate_weight_scaled < 1 } }
            set_temp_variable = { random_event_selection_candidate_is_valid = 0 }
            clear_variable = random_event_selection_candidate_weight_scaled
        }
    }
}

select_weighted_random_event_id = {
    ensure_random_event_selection_filter = yes
    clear_variable = random_event_selection_id
    clear_variable = random_event_selection_roll_scaled
    set_temp_variable = { random_event_selection_total_weight_scaled = 0 }

    for_loop_effect = {
        start = 0
        end = global.all_events^num
        value = i

        set_temp_variable = { random_event_selection_candidate_id = global.all_events^i }
        evaluate_random_event_selection_candidate = yes

        if = {
            limit = { check_variable = { random_event_selection_candidate_is_valid = 1 } }
            add_to_temp_variable = { random_event_selection_total_weight_scaled = random_event_selection_candidate_weight_scaled }
        }
    }

    if = {
        limit = { check_variable = { random_event_selection_total_weight_scaled > 0 } }
        set_variable_to_random = {
            var = random_event_selection_roll_scaled
            min = 1
            max = random_event_selection_total_weight_scaled
            integer = yes
        }

        set_temp_variable = { random_event_selection_running_weight_scaled = 0 }

        for_loop_effect = {
            start = 0
            end = global.all_events^num
            value = i

            if = {
                limit = { NOT = { has_variable = random_event_selection_id } }
                set_temp_variable = { random_event_selection_candidate_id = global.all_events^i }
                evaluate_random_event_selection_candidate = yes

                if = {
                    limit = { check_variable = { random_event_selection_candidate_is_valid = 1 } }
                    add_to_temp_variable = { random_event_selection_running_weight_scaled = random_event_selection_candidate_weight_scaled }

                    if = {
                        limit = {
                            check_variable = {
                                var = random_event_selection_running_weight_scaled
                                value = random_event_selection_roll_scaled
                                compare = greater_than_or_equals
                            }
                        }
                        set_temp_variable = { random_event_selection_id = random_event_selection_candidate_id }
                    }
                }
            }
        }
    }
}

select_unweighted_random_event_id = {
    ensure_random_event_selection_filter = yes
    clear_variable = random_event_selection_id

    if = {
        limit = {
            check_variable = { random_event_selection_filter = constant:settings_event_filter.all }
            check_variable = { global.all_events^num > 0 }
        }
        set_temp_variable = { array_size = global.all_events^num }
        subtract_from_temp_variable = { array_size = 1 }
        set_variable_to_random = {
            var = temp_random_index
            min = 0
            max = array_size
            integer = yes
        }
        set_temp_variable = { random_event_selection_id = global.all_events^temp_random_index }
    }
    else_if = {
        limit = {
            check_variable = { random_event_selection_filter = constant:settings_event_filter.major }
            check_variable = { global.major_events^num > 0 }
        }
        set_temp_variable = { array_size = global.major_events^num }
        subtract_from_temp_variable = { array_size = 1 }
        set_variable_to_random = {
            var = temp_random_index
            min = 0
            max = array_size
            integer = yes
        }
        set_temp_variable = { random_event_selection_id = global.major_events^temp_random_index }
    }
    else_if = {
        limit = {
            check_variable = { random_event_selection_filter = constant:settings_event_filter.repeatable }
            check_variable = { global.repeatable_events^num > 0 }
        }
        set_temp_variable = { array_size = global.repeatable_events^num }
        subtract_from_temp_variable = { array_size = 1 }
        set_variable_to_random = {
            var = temp_random_index
            min = 0
            max = array_size
            integer = yes
        }
        set_temp_variable = { random_event_selection_id = global.repeatable_events^temp_random_index }
    }
    else_if = {
        limit = {
            check_variable = { random_event_selection_filter = constant:settings_event_filter.fire_once }
            check_variable = { global.fire_once_events^num > 0 }
        }
        set_temp_variable = { array_size = global.fire_once_events^num }
        subtract_from_temp_variable = { array_size = 1 }
        set_variable_to_random = {
            var = temp_random_index
            min = 0
            max = array_size
            integer = yes
        }
        set_temp_variable = { random_event_selection_id = global.fire_once_events^temp_random_index }
    }
}

fire_event_by_temp_id = {
    # Input: temp variable `event_id`
    get_event_type = yes

    if = {
        limit = { check_variable = { event_type = constant:event_system_event_type.major } }
        on_major_event_fired = yes
    }
    else_if = {
        limit = { check_variable = { event_type = constant:event_system_event_type.repeatable } }
        on_repeatable_event_fired = yes
    }
    else_if = {
        limit = { check_variable = { event_type = constant:event_system_event_type.fire_once } }
        on_fire_once_event_fired = yes
    }

    meta_effect = {
        text = {
            country_event = {
                id = chaosx.nr[EVENT_ID].1
            }
        }
        EVENT_ID = "[?event_id]"
    }

    update_last_fired_event_display = yes
}

# ============================================================================
# CHAOS REDUX - SUPER EVENT AUDIO EFFECTS
# ============================================================================
# Settings-aware super event playback helpers.
#
# Requirements:
# - Current scope must be the player country that should hear the audio.
# - `super_event_visible` must already be set to the current super event ID.
# - `global.current_super_event_audio_id` must mirror that ID.
#
# Settings used:
# - `settings_super_event_audio_mode`
# - `settings_super_event_audio_volume`
# ============================================================================

play_current_super_event_audio = {
    if = {
        limit = {
            OR = {
                NOT = { has_global_flag = super_event_visible }
                NOT = { has_variable = global.current_super_event_audio_id }
            }
        }
    }
    else_if = {
        limit = { check_variable = { settings_super_event_audio_mode = constant:settings_super_event_audio_mode.sound } }
        play_current_super_event_sound = yes
    }
    else = {
        play_current_super_event_music = yes
    }
}

play_current_super_event_music = {
    if = {
        limit = { check_variable = { settings_super_event_audio_volume = constant:settings_super_event_audio_volume.x0 } }

        play_dynamic_super_event_music = yes
    }
}
