germany_mengele_add_autonomy = {
	if = {
		limit = { NOT = { has_variable = mengele_autonomy } }
		set_variable = { mengele_autonomy = 0 }
	}
	add_to_variable = { mengele_autonomy = mengele_autonomy_delta }
	clamp_variable = {
		var = mengele_autonomy
		min = 0
		max = constant:germany_mengele_autonomy.autonomy_max
	}
}

germany_mengele_add_condemnation = {

		set_variable = { chem_warfare_condemnation = 0 }

germany_mengele_record_stage = {
	if = {
		limit = { NOT = { has_variable = germany_mengele_current_stage } }
		set_variable = { germany_mengele_current_stage = 0 }
	}
	if = {
		limit = { check_variable = { mengele_stage_value > germany_mengele_current_stage } }
		set_variable = { germany_mengele_current_stage = mengele_stage_value }
	}
}

germany_mengele_count_biowarfare_facilities = {
	set_temp_variable = { mengele_biowarfare_facility_count = 0 }
	every_controlled_state = {
		limit = { biowarfare_facility > 0 }
		add_to_temp_variable = { mengele_biowarfare_facility_count = 1 }
	}
}

germany_mengele_calculate_coup_pressure = {
	if = {
		limit = { has_variable = mengele_autonomy }
		set_temp_variable = { mengele_coup_pressure = mengele_autonomy }
	}
	else = {
		set_temp_variable = { mengele_coup_pressure = 0 }
	}
	germany_mengele_count_biowarfare_facilities = yes
	set_temp_variable = { mengele_facility_pressure = mengele_biowarfare_facility_count }
	multiply_temp_variable = { mengele_facility_pressure = constant:germany_mengele_autonomy.facility_count_factor }
	add_to_temp_variable = { mengele_coup_pressure = mengele_facility_pressure }
	if = {
		limit = { has_war_with = SOV }
		add_to_temp_variable = { mengele_coup_pressure = constant:germany_mengele_autonomy.war_with_soviets }
	}
	if = {
		limit = { germany_mengele_has_high_fascist_support = yes }
		add_to_temp_variable = { mengele_coup_pressure = constant:germany_mengele_autonomy.high_fascist_support }
	}
	if = {
		limit = {
			OR = {
				has_global_flag = { flag = chaos_tier value = 3 }
				has_global_flag = { flag = chaos_tier value = 4 }
				has_global_flag = { flag = chaos_tier value = 5 }
			}
		}
		add_to_temp_variable = { mengele_coup_pressure = constant:germany_mengele_autonomy.high_chaos }
	}
	if = {
		limit = { germany_mengele_is_desperate_in_war = yes }
		add_to_temp_variable = { mengele_coup_pressure = constant:germany_mengele_autonomy.desperate_war }
	}
	if = {
		limit = { has_variable = mengele_permission_level }
		set_temp_variable = { mengele_permission_pressure = mengele_permission_level }
		multiply_temp_variable = { mengele_permission_pressure = constant:germany_mengele_permission.coup_pressure_factor }
		add_to_temp_variable = { mengele_coup_pressure = mengele_permission_pressure }
	}
}

germany_mengele_register_experiment_deaths = {
	set_temp_variable = { mengele_experiment_deaths = constant:germany_mengele_deaths.base }
	if = {
		limit = { has_variable = mengele_autonomy }
		set_temp_variable = { mengele_autonomy_death_add = mengele_autonomy }
		multiply_temp_variable = { mengele_autonomy_death_add = mengele_autonomy }
		multiply_temp_variable = { mengele_autonomy_death_add = constant:germany_mengele_deaths.autonomy_square_factor }
		add_to_temp_variable = { mengele_experiment_deaths = mengele_autonomy_death_add }
	}
	germany_mengele_count_biowarfare_facilities = yes
	set_temp_variable = { mengele_facility_death_add = mengele_biowarfare_facility_count }
	multiply_temp_variable = { mengele_facility_death_add = constant:germany_mengele_deaths.facility_factor }
	add_to_temp_variable = { mengele_experiment_deaths = mengele_facility_death_add }
	if = {
		limit = { has_variable = mengele_permission_level }
		set_temp_variable = { mengele_permission_death_add = mengele_permission_level }
		multiply_temp_variable = { mengele_permission_death_add = constant:germany_mengele_permission.experiment_death_factor }
		add_to_temp_variable = { mengele_experiment_deaths = mengele_permission_death_add }
		set_temp_variable = { mengele_permission_hidden_atrocity = mengele_permission_level }
		multiply_temp_variable = { mengele_permission_hidden_atrocity = constant:germany_mengele_permission.experiment_hidden_atrocity_factor }
		add_to_variable = { hidden_atrocity_score = mengele_permission_hidden_atrocity }
	}
	if = {
		limit = { 88 = { has_state_flag = genocide_auschwitz_experiment_site } }
		add_to_temp_variable = { mengele_experiment_deaths = constant:germany_mengele_deaths.auschwitz_site_bonus }
	}
	clamp_temp_variable = {
		var = mengele_experiment_deaths
		min = 0
		max = constant:germany_mengele_deaths.max_report_deaths
	}
	round_temp_variable = mengele_experiment_deaths
	88 = {
		set_temp_variable = { chaos_deaths_change = mengele_experiment_deaths }
		set_temp_variable = { chaos_deaths_is_civilian = 1 }
		set_temp_variable = { chaos_deaths_apply_state_pop = 1 }
		set_temp_variable = { chaos_deaths_reason = constant:chaos_meter_deaths_reason.biowarfare_outbreak }
		set_temp_variable = { chaos_deaths_target_country = OWNER }
		set_temp_variable = { chaos_deaths_has_target_country = 1 }
		chaos_meter_register_deaths = yes
	}
}

germany_tibet_finish_expedition = {
	clr_country_flag = germany_tibet_expedition_active
	set_country_flag = germany_tibet_expedition_completed
	if = {
		limit = {
			OR = {
				has_country_flag = germany_tibet_route_final_silence_danger
				check_variable = { tibet_expedition_integrity < constant:germany_mengele_tibet.lost_threshold }
			}
		}
		set_country_flag = germany_tibet_expedition_lost
	}
	else_if = {
		limit = { check_variable = { tibet_expedition_exposure > constant:germany_mengele_tibet.scandal_threshold } }
		set_country_flag = germany_tibet_expedition_scandal
		set_temp_variable = { mengele_chaos_change = constant:germany_mengele_chaos.tibet_scandal }
		set_temp_variable = { mengele_chaos_reason = constant:chaos_meter_history_reason.special.tibet_scandal }
		germany_mengele_add_chaos = yes
		set_temp_variable = { mengele_condemnation_add = constant:germany_mengele_condemnation.tibet_scandal }
		germany_mengele_add_condemnation = yes
		every_other_country = {
			limit = {
				exists = yes
				is_major = yes
				NOT = { original_tag = GER }
			}
			add_opinion_modifier = {
				target = ROOT
				modifier = germany_tibet_expedition_rejected
			}
		}
	}
	else_if = {
		limit = {
			OR = {
				has_country_flag = germany_tibet_holy_realm_claim_guaranteed
				check_variable = { tibet_expedition_holy_contact > constant:germany_mengele_tibet.stage_outcome_threshold }
			}
		}
		set_country_flag = germany_tibet_expedition_holy_confirmed
		set_country_flag = germany_holy_realm_contact_positive
		if = {
			limit = { has_event_target = germany_tibet_holy_realm_contact_country }
			event_target:germany_tibet_holy_realm_contact_country = {
				set_country_flag = holy_realm_german_origin_claim_contact
				if = {
					limit = { has_variable = mandala_reach }
					add_to_variable = { mandala_reach = constant:germany_mengele_tibet.holy_realm_mandala_reach }
				}
			}
		}
		set_temp_variable = { mengele_chaos_change = constant:germany_mengele_chaos.tibet_holy_confirmed }
		set_temp_variable = { mengele_chaos_reason = constant:chaos_meter_history_reason.special.tibet_claim }
		germany_mengele_add_chaos = yes
		set_temp_variable = { mengele_condemnation_add = constant:germany_mengele_condemnation.tibet_holy_confirmed }
		germany_mengele_add_condemnation = yes
	}
	else_if = {
		limit = { has_country_flag = germany_tibet_fabricated_result }
		set_country_flag = germany_tibet_expedition_fabricated_positive
		set_temp_variable = { mengele_chaos_change = constant:germany_mengele_chaos.tibet_public_claim }
		set_temp_variable = { mengele_chaos_reason = constant:chaos_meter_history_reason.special.tibet_claim }
		germany_mengele_add_chaos = yes
		set_temp_variable = { mengele_condemnation_add = constant:germany_mengele_condemnation.tibet_public_claim }
		germany_mengele_add_condemnation = yes
	}
	else_if = {
		limit = { check_variable = { tibet_expedition_progress > constant:germany_mengele_tibet.stage_outcome_threshold } }
		set_country_flag = germany_tibet_expedition_ambiguous_positive
	}
	else = {
		set_country_flag = germany_tibet_expedition_no_result
	}
	set_temp_variable = { mengele_stage_value = constant:germany_mengele_stage.resolution }
	germany_mengele_record_stage = yes
	if = { limit = { has_event_target = germany_tibet_origin_country } clear_global_event_target = germany_tibet_origin_country }
	if = { limit = { has_event_target = germany_tibet_holy_realm_contact_country } clear_global_event_target = germany_tibet_holy_realm_contact_country }
}
