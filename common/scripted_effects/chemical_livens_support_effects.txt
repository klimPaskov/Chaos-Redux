chem_livens_support_calculate_guaranteed_support_in_combat = {
	set_temp_variable = { var = chem_livens_total_units value = num_units }
	clamp_temp_variable = { var = chem_livens_total_units min = 0 }
	set_temp_variable = { var = chem_livens_units_in_combat value = num_units_in_combat }
	clamp_temp_variable = { var = chem_livens_units_in_combat min = 0 }

	set_temp_variable = { var = chem_livens_non_support_units value = chem_livens_support_total }
	multiply_temp_variable = { var = chem_livens_non_support_units value = -1 }
	add_to_temp_variable = { var = chem_livens_non_support_units value = chem_livens_total_units }
	clamp_temp_variable = { var = chem_livens_non_support_units min = 0 }

	set_temp_variable = { var = chem_livens_guaranteed_support_combat value = chem_livens_non_support_units }
	multiply_temp_variable = { var = chem_livens_guaranteed_support_combat value = -1 }
	add_to_temp_variable = { var = chem_livens_guaranteed_support_combat value = chem_livens_units_in_combat }
	clamp_temp_variable = { var = chem_livens_guaranteed_support_combat min = 0 }
	if = {
		limit = { check_variable = { var = chem_livens_guaranteed_support_combat value = chem_livens_support_total compare = greater_than } }
		set_temp_variable = { var = chem_livens_guaranteed_support_combat value = chem_livens_support_total }
	}
}

	chem_set_livens_integration_values = yes

chem_livens_support_set_owner_support_count_in_state = {
	set_temp_variable = { var = chem_livens_state_count_low value = 0 }
	set_temp_variable = { var = chem_livens_state_count_high value = @CHEM_LIVENS_STATE_SUPPORT_COUNT_MAX }
	add_to_temp_variable = { var = chem_livens_state_count_high value = 1 }

	set_temp_variable = { var = chem_livens_state_count_gap value = chem_livens_state_count_high }
	set_temp_variable = { var = chem_livens_state_count_gap_sub value = chem_livens_state_count_low }
	multiply_temp_variable = { var = chem_livens_state_count_gap_sub value = -1 }
	add_to_temp_variable = { var = chem_livens_state_count_gap value = chem_livens_state_count_gap_sub }

	while_loop_effect = {
		limit = { check_variable = { var = chem_livens_state_count_gap value = 1 compare = greater_than } }

		set_temp_variable = { var = chem_livens_state_count_mid value = chem_livens_state_count_low }
		add_to_temp_variable = { var = chem_livens_state_count_mid value = chem_livens_state_count_high }
		divide_temp_variable = { var = chem_livens_state_count_mid value = 2 }
		round_temp_variable = chem_livens_state_count_mid

		set_temp_variable = { var = chem_livens_state_count_probe_minus_one value = chem_livens_state_count_mid }
		add_to_temp_variable = { var = chem_livens_state_count_probe_minus_one value = -1 }

		if = {
			limit = {
				event_target:chem_state_dispersal_owner_country = {
					meta_trigger = {
						text = {
							num_battalions_in_states = {
								count > [CHEM_LIVENS_COUNT_PROBE]
								states = { [CHEM_STATE_ID] }
								types = {
									livens_projector_support_chlorine
									livens_projector_support_phosgene
									livens_projector_support_mustard
									livens_projector_support_lewisite
									livens_projector_support_sarin
									livens_projector_support_soman
									livens_projector_support
								}
							}
						}
						CHEM_LIVENS_COUNT_PROBE = "[?chem_livens_state_count_probe_minus_one|0]"
						CHEM_STATE_ID = "[?chem_state_id|0]"
					}
				}
			}
			set_temp_variable = { var = chem_livens_state_count_low value = chem_livens_state_count_mid }
		}
		else = {
			set_temp_variable = { var = chem_livens_state_count_high value = chem_livens_state_count_mid }
		}

		set_temp_variable = { var = chem_livens_state_count_gap value = chem_livens_state_count_high }
		set_temp_variable = { var = chem_livens_state_count_gap_sub value = chem_livens_state_count_low }
		multiply_temp_variable = { var = chem_livens_state_count_gap_sub value = -1 }
		add_to_temp_variable = { var = chem_livens_state_count_gap value = chem_livens_state_count_gap_sub }
	}

	set_temp_variable = { var = chem_livens_support_in_enemy_state value = chem_livens_state_count_low }
}

chem_livens_support_apply_state_dispersal_from_controlled_states = {
	every_controlled_state = {
		every_neighbor_state = {
			limit = {
				CONTROLLER = { has_war_with = event_target:chem_state_dispersal_owner_country }
				NOT = { is_controlled_by = event_target:chem_state_dispersal_owner_country }
			}

			meta_effect = {
				text = {
					set_temp_variable = { var = chem_state_id value = [CHEM_STATE_ID] }
				}
				CHEM_STATE_ID = "[THIS.GetID]"
			}
			chem_livens_support_set_owner_support_count_in_state = yes

			if = {
				limit = { check_variable = { var = chem_livens_support_in_enemy_state value = 0 compare = greater_than } }
				add_to_temp_variable = { var = chem_state_scan_states_with_units value = 1 }
				set_temp_variable = { var = chem_dose_each value = chem_livens_support_in_enemy_state }
				multiply_temp_variable = { var = chem_dose_each value = chem_state_dose_per_div }
				add_to_temp_variable = { var = chem_state_scan_states_applied value = 1 }
				chem_apply_state_contamination = yes
			}
		}
	}
}
