holy_realm_ensure_counters = {
	if = {
		limit = { NOT = { has_variable = spiritual_legitimacy } }
		set_variable = { spiritual_legitimacy = constant:holy_realm_counter.spiritual_legitimacy_start }
	}
	if = {
		limit = { NOT = { has_variable = compassion_drift } }
		set_variable = { compassion_drift = constant:holy_realm_counter.compassion_drift_start }
	}
	if = {
		limit = { NOT = { has_variable = mandala_reach } }
		set_variable = { mandala_reach = constant:holy_realm_counter.mandala_reach_start }
	}
	if = {
		limit = { NOT = { has_variable = final_silence_pressure } }
		set_variable = { final_silence_pressure = constant:holy_realm_counter.final_silence_pressure_start }
	}
}

holy_realm_add_spiritual_legitimacy = {
	holy_realm_ensure_counters = yes
	set_temp_variable = { holy_realm_adjusted_delta = holy_realm_delta }
	if = {
		limit = {
			check_variable = { holy_realm_adjusted_delta > 0 }
			check_variable = { spiritual_legitimacy > constant:holy_realm_counter.soft_cap_start }
		}
		multiply_temp_variable = { holy_realm_adjusted_delta = constant:holy_realm_counter.soft_cap_multiplier }
	}
	if = {
		limit = {
			check_variable = { holy_realm_adjusted_delta > 0 }
			check_variable = { spiritual_legitimacy > constant:holy_realm_counter.soft_cap_late }
		}
		multiply_temp_variable = { holy_realm_adjusted_delta = constant:holy_realm_counter.soft_cap_multiplier }
	}
	add_to_variable = { spiritual_legitimacy = holy_realm_adjusted_delta }
	clamp_variable = { var = spiritual_legitimacy min = 0 max = 100 }
}

holy_realm_initialize_final_doctrine_balance = {
	if = {
		limit = {
			NOT = {
				has_power_balance = {
					id = holy_realm_final_doctrine_balance
				}
			}
		}
		set_power_balance = {
			id = holy_realm_final_doctrine_balance
			left_side = holy_realm_vow_against_annihilation_side
			right_side = holy_realm_final_silence_side
			set_value = -0.10
		}
	}
}

holy_realm_daily_doctrine_balance_pulse = {
	holy_realm_ensure_counters = yes
	if = {
		limit = {
			has_power_balance = {
				id = holy_realm_final_doctrine_balance
			}
			NOT = { has_country_flag = holy_realm_final_doctrine_renounced }
			is_power_balance_in_range = {
				id = holy_realm_final_doctrine_balance
				range = holy_realm_final_silence_dominant_range
			}
		}
		set_temp_variable = { holy_realm_delta = constant:holy_realm_counter.bop_daily_final_major }
		holy_realm_add_final_silence_pressure = yes
		set_temp_variable = { holy_realm_delta = constant:holy_realm_counter.bop_daily_drift_major }
		holy_realm_add_compassion_drift = yes
	}
	else_if = {
		limit = {
			has_power_balance = {
				id = holy_realm_final_doctrine_balance
			}
			NOT = { has_country_flag = holy_realm_final_doctrine_renounced }
			is_power_balance_in_range = {
				id = holy_realm_final_doctrine_balance
				range = holy_realm_final_silence_influence_range
			}
		}
		set_temp_variable = { holy_realm_delta = constant:holy_realm_counter.bop_daily_final_minor }
		holy_realm_add_final_silence_pressure = yes
		set_temp_variable = { holy_realm_delta = constant:holy_realm_counter.bop_daily_drift_minor }
		holy_realm_add_compassion_drift = yes
	}
	else_if = {
		limit = {
			has_power_balance = {
				id = holy_realm_final_doctrine_balance
			}
			is_power_balance_in_range = {
				id = holy_realm_final_doctrine_balance
				range = holy_realm_vow_dominant_range
			}
		}
		set_temp_variable = { holy_realm_delta = constant:holy_realm_counter.bop_daily_vow_major }
		holy_realm_add_final_silence_pressure = yes
		set_temp_variable = { holy_realm_delta = constant:holy_realm_counter.bop_daily_mercy_drift }
		holy_realm_add_compassion_drift = yes
	}
	else_if = {
		limit = {
			has_power_balance = {
				id = holy_realm_final_doctrine_balance
			}
			is_power_balance_in_range = {
				id = holy_realm_final_doctrine_balance
				range = holy_realm_vow_influence_range
			}
		}
		set_temp_variable = { holy_realm_delta = constant:holy_realm_counter.bop_daily_vow_minor }
		holy_realm_add_final_silence_pressure = yes
	}
}

holy_realm_add_chaos = {
	if = {
		limit = { has_variable = holy_realm_chaos_delta }
		if = {
			limit = {
				check_variable = { holy_realm_chaos_delta < 0 }
				has_country_flag = holy_realm_chaos_reduction_cooldown
			}
		}
		else = {
			if = {
				limit = { check_variable = { holy_realm_chaos_delta < 0 } }
				set_country_flag = {
					flag = holy_realm_chaos_reduction_cooldown
					days = @HOLY_REALM_CHAOS_REDUCTION_COOLDOWN_DAYS
				}
			}
			set_temp_variable = { chaos_change = holy_realm_chaos_delta }
			if = {
				limit = { has_variable = holy_realm_chaos_reason }
				set_temp_variable = { chaos_history_reason = holy_realm_chaos_reason }
			}
			else_if = {
				limit = { check_variable = { holy_realm_chaos_delta < 0 } }
				set_temp_variable = { chaos_history_reason = constant:chaos_meter_history_reason.special.holy_realm_refuge_relief }
			}
			else = {
				set_temp_variable = { chaos_history_reason = constant:chaos_meter_history_reason.special.holy_realm_doctrinal_control }
			}
			set_temp_variable = { chaos_history_reason_custom = 1 }
			save_event_target_as = chaos_history_actor
			add_chaos_meter_value = yes
		}
	}
}

holy_realm_advance_bodhisattva_bhumi = {
	holy_realm_ensure_counters = yes
	if = {
		limit = { NOT = { has_variable = bodhisattva_bhumi } }
		set_variable = { bodhisattva_bhumi = 0 }
	}
	if = {
		limit = { check_variable = { bodhisattva_bhumi < 10 } }
		add_to_variable = { bodhisattva_bhumi = 1 }
	}
	if = {
		limit = { check_variable = { bodhisattva_bhumi > 9 } }
		set_country_leader_portrait = {
			portrait = "GFX_portrait_THR_bodhisattva_dharmamegha"
		}
	}
	else_if = {
		limit = { check_variable = { bodhisattva_bhumi > 7 } }
		set_country_leader_portrait = {
			portrait = "GFX_portrait_THR_bodhisattva_acala"
		}
	}
	else_if = {
		limit = { check_variable = { bodhisattva_bhumi > 0 } }
		set_country_leader_portrait = {
			portrait = "GFX_portrait_THR_bodhisattva_pramudita"
		}
	}
	set_temp_variable = { holy_realm_delta = constant:holy_realm_counter.small_gain }
	holy_realm_add_spiritual_legitimacy = yes
	if = {
		limit = { check_variable = { bodhisattva_bhumi > 3 } }
		add_political_power = 50
	}
	if = {
		limit = { check_variable = { bodhisattva_bhumi > 6 } }
		add_political_power = 75
		add_command_power = 15
	}
	if = {
		limit = { check_variable = { bodhisattva_bhumi > 8 } }
		add_stability = 0.02
	}
	holy_realm_refresh_bodhisattva_leader_trait = yes
}

holy_realm_late_vow_against_annihilation_crusade = {
	if = {
		limit = {
			OR = {
				has_country_flag = holy_realm_final_doctrine_armed
				has_country_flag = holy_realm_stage_final_silence
				check_variable = { final_silence_pressure > 70 }
				date > 1944.12.31
			}
			NOT = { has_country_flag = holy_realm_late_vow_crusade_launched }
		}
		set_country_flag = holy_realm_late_vow_crusade_launched
		add_ideas = holy_realm_vow_against_annihilation_crusade
		if = {
			limit = {
				is_in_faction = yes
				is_faction_leader = no
			}
			leave_faction = yes
		}
		if = {
			limit = { is_in_faction = no }
			create_faction_from_template = {
				template = faction_template_holy_realm_mandala_of_nations
				name = holy_realm_vow_against_annihilation_faction_name
				icon = GFX_faction_logo_generic_democratic
			}
		}
		set_global_flag = holy_realm_late_vow_war_declared
		every_other_country = {
			limit = {
				exists = yes
				uses_normal_civilian_systems = yes
				NOT = { is_holy_realm_country = yes }
				OR = {
					has_government = communism
					has_government = fascism
				}
				NOT = { has_war_with = ROOT }
				ROOT = { can_declare_war_on = PREV }
			}
			save_event_target_as = holy_realm_annihilation_war_target
			ROOT = {
				declare_war_on = {
					target = event_target:holy_realm_annihilation_war_target
					type = annex_everything
				}
			}
		}
	}
}
