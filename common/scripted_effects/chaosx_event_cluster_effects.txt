# ============================================================================
# CHAOS REDUX - EVENT CLUSTER EFFECTS
# ============================================================================
# Event clusters sit above the individual random-event selection layer. When an
# automatically selected event belongs to a cluster, the cluster can fire first,
# record a cluster log row, and then queue valid member events through the same
# event handlers used by single events. Member events run with a cluster context
# guard so the cluster updates timer and major-event pacing once for the group.
#
# To add a future cluster, register it in initialize_event_cluster_definitions,
# map member events in event_belongs_to_cluster, and add ordered member rows in
# load_event_cluster_members.
# ============================================================================

get_current_chaos_tier_value = {
    set_temp_variable = { current_chaos_tier_value = constant:chaos_meter_tier_id.tier_0 }
    if = {
        limit = { has_global_flag = { flag = chaos_tier value = 5 } }
        set_temp_variable = { current_chaos_tier_value = constant:chaos_meter_tier_id.tier_final }
    }
    else_if = {
        limit = { has_global_flag = { flag = chaos_tier value = 4 } }
        set_temp_variable = { current_chaos_tier_value = constant:chaos_meter_tier_id.tier_4 }
    }

        limit = { has_global_flag = { flag = chaos_tier value = 3 } }

        start = 0

                    var = global.event_clusters^i

event_belongs_to_cluster = {
    set_temp_variable = { event_cluster_candidate_id = 0 }
	    if = {
	        limit = {
				OR = {
		            check_variable = {
		                var = event_id
		                value = 4
		                compare = equals
		            }
		            check_variable = {
		                var = event_id
		                value = 7
		                compare = equals
		            }
				}
	        }
	        set_temp_variable = { event_cluster_candidate_id = constant:event_cluster_id.wars }
	    }
	else_if = {
		limit = {
			OR = {
				check_variable = {
					var = event_id
					value = 5
					compare = equals
				}
				check_variable = {
					var = event_id
					value = 6
					compare = equals
				}
			}
		}
		set_temp_variable = { event_cluster_candidate_id = constant:event_cluster_id.liberations }
	}
	else_if = {
		limit = {
			check_variable = {
				var = event_id
				value = constant:tensions_rising_event_log.event_id
				compare = equals
			}
		}
		set_temp_variable = { event_cluster_candidate_id = constant:event_cluster_id.diplomatic_panic }
	}
}

event_cluster_member_check_availability = {
    set_temp_variable = { event_cluster_member_can_fire = 1 }
    set_temp_variable = { event_cluster_member_status = constant:event_cluster_member_status.fired }
    get_current_chaos_tier_value = yes

    if = {
        limit = { is_in_array = { global.disabled_events = event_cluster_member_event_id } }
        set_temp_variable = { event_cluster_member_can_fire = 0 }
        set_temp_variable = { event_cluster_member_status = constant:event_cluster_member_status.skipped_disabled }
    }
    else_if = {
        limit = {
            check_variable = {
                var = current_chaos_tier_value
                value = event_cluster_member_min_tier
                compare = less_than
            }
        }
        set_temp_variable = { event_cluster_member_can_fire = 0 }
        set_temp_variable = { event_cluster_member_status = constant:event_cluster_member_status.skipped_chaos_tier }
    }
    else = {
        set_temp_variable = { random_event_selection_candidate_id = event_cluster_member_event_id }
        set_temp_variable = { random_event_selection_filter = constant:settings_event_filter.all }
        evaluate_random_event_selection_candidate = yes
        if = {
            limit = {
                check_variable = {
                    var = random_event_selection_candidate_is_valid
                    value = 1
                    compare = less_than
                }
            }
            set_temp_variable = { event_cluster_member_can_fire = 0 }
            set_temp_variable = { event_cluster_member_status = constant:event_cluster_member_status.skipped_unavailable }
        }
    }
}

can_event_cluster_fire = {
    set_temp_variable = { event_cluster_can_fire = 0 }
    get_event_cluster_definition_index = yes
    if = {
        limit = {
            check_variable = {
                var = event_cluster_definition_index
                value = 0
                compare = greater_than_or_equals
            }
        }
        get_event_cluster_type = yes
        get_event_cluster_unlock_tier = yes
        get_current_chaos_tier_value = yes

        if = {
            limit = {
                check_variable = {
                    var = current_chaos_tier_value
                    value = event_cluster_unlock_tier
                    compare = greater_than_or_equals
                }
                NOT = {
                    AND = {
                        check_variable = {
                            var = event_cluster_type_value
                            value = constant:event_cluster_type.one_time
                            compare = equals
                        }
                        is_in_array = { global.fired_event_clusters = event_cluster_id }
                    }
                }
                NOT = {
                    AND = {
                        check_variable = {
                            var = event_cluster_id
                            value = constant:event_cluster_id.wars
                            compare = equals
                        }
                        has_global_flag = event_cluster_wars_cooldown
                    }
                }
				NOT = {
					AND = {
						check_variable = {
							var = event_cluster_id
							value = constant:event_cluster_id.liberations
							compare = equals
						}
						has_global_flag = event_cluster_liberations_cooldown
					}
				}
				NOT = {
					AND = {
						check_variable = {
							var = event_cluster_id
							value = constant:event_cluster_id.diplomatic_panic
							compare = equals
						}
						has_global_flag = event_cluster_diplomatic_panic_cooldown
					}
				}
                NOT = { is_in_array = { global.disabled_event_clusters = event_cluster_id } }
            }
            load_event_cluster_members = yes
            set_temp_variable = { event_cluster_can_fire_scan_index = 0 }
            while_loop_effect = {
                limit = {
                    check_variable = {
                        var = event_cluster_can_fire_scan_index
                        value = temp_event_cluster_member_event_id_entries^num
                        compare = less_than
                    }
                }
                set_temp_variable = { event_cluster_member_event_id = temp_event_cluster_member_event_id_entries^event_cluster_can_fire_scan_index }
                set_temp_variable = { event_cluster_member_min_tier = temp_event_cluster_member_min_tier_entries^event_cluster_can_fire_scan_index }
                event_cluster_member_check_availability = yes
                if = {
                    limit = {
                        check_variable = {
                            var = event_cluster_member_can_fire
                            value = 0
                            compare = greater_than
                        }
                    }
                    set_temp_variable = { event_cluster_can_fire = 1 }
                }
                add_to_temp_variable = { event_cluster_can_fire_scan_index = 1 }
            }
        }
    }
}

prepare_event_cluster_firing = {
    load_event_cluster_members = yes
    clear_array = temp_event_cluster_member_status_entries
    clear_array = temp_event_cluster_firing_order_event_id_entries
    set_temp_variable = { event_cluster_member_count = temp_event_cluster_member_event_id_entries^num }
    set_temp_variable = { event_cluster_fired_member_count = 0 }
    set_temp_variable = { event_cluster_skipped_member_count = 0 }

    for_loop_effect = {
        start = 0
        end = temp_event_cluster_member_event_id_entries^num
        value = i

        set_temp_variable = { event_cluster_member_event_id = temp_event_cluster_member_event_id_entries^i }
        set_temp_variable = { event_cluster_member_role = temp_event_cluster_member_role_entries^i }
        set_temp_variable = { event_cluster_member_chance = temp_event_cluster_member_chance_entries^i }
        set_temp_variable = { event_cluster_member_min_tier = temp_event_cluster_member_min_tier_entries^i }

        if = {
            limit = {
                has_variable = event_cluster_trigger_event_id
                check_variable = {
                    var = event_cluster_member_event_id
                    value = event_cluster_trigger_event_id
                    compare = equals
                }
            }
            set_temp_variable = { event_cluster_member_role = constant:event_cluster_member_role.required }
            set_temp_variable = { event_cluster_member_chance = constant:event_cluster_member_participation.required }
        }

        if = {
            limit = { check_variable = { event_cluster_force_fire > 0 } }
            set_temp_variable = { event_cluster_member_can_fire = 1 }
            set_temp_variable = { event_cluster_member_status = constant:event_cluster_member_status.fired }
        }
        else = {
            event_cluster_member_check_availability = yes
        }

        if = {
            limit = {
                check_variable = { event_cluster_force_fire < 1 }
                check_variable = {
                    var = event_cluster_member_can_fire
                    value = 0
                    compare = greater_than
                }
                check_variable = {
                    var = event_cluster_member_role
                    value = constant:event_cluster_member_role.optional
                    compare = equals
                }
            }
            set_variable_to_random = {
                var = event_cluster_member_roll
                min = constant:event_cluster_roll.minimum
                max = constant:event_cluster_roll.maximum
                integer = yes
            }
            if = {
                limit = {
                    check_variable = {
                        var = event_cluster_member_roll
                        value = event_cluster_member_chance
                        compare = greater_than
                    }
                }
                set_temp_variable = { event_cluster_member_can_fire = 0 }
                set_temp_variable = { event_cluster_member_status = constant:event_cluster_member_status.skipped_participation }
            }
        }

        if = {
            limit = {
                check_variable = {
                    var = event_cluster_member_can_fire
                    value = 0
                    compare = greater_than
                }
            }
            add_to_temp_variable = { event_cluster_fired_member_count = 1 }
        }
        else = {
            add_to_temp_variable = { event_cluster_skipped_member_count = 1 }
        }

        add_to_array = {
            array = temp_event_cluster_member_status_entries
            value = event_cluster_member_status
        }
    }
    build_event_cluster_firing_order = yes
}

    set_temp_variable = { event_cluster_order_source_count = temp_event_cluster_order_candidate_event_id_entries^num }
    set_temp_variable = { event_cluster_order_sorted_count = 0 }
    while_loop_effect = {
        limit = { check_variable = { event_cluster_order_sorted_count < event_cluster_order_source_count } }

        set_temp_variable = { event_cluster_order_best_index = -1 }
        set_temp_variable = { event_cluster_order_scan_index = 0 }
        while_loop_effect = {
            limit = { check_variable = { event_cluster_order_scan_index < event_cluster_order_source_count } }
            if = {
                limit = {
                    check_variable = {
                        var = temp_event_cluster_order_candidate_event_id_entries^event_cluster_order_scan_index
                        value = 0
                        compare = greater_than
                    }
                }
                set_temp_variable = { event_cluster_order_candidate_score = temp_event_cluster_order_candidate_score_entries^event_cluster_order_scan_index }
                set_temp_variable = { event_cluster_order_candidate_event_id = temp_event_cluster_order_candidate_event_id_entries^event_cluster_order_scan_index }

                if = {
                    limit = { check_variable = { event_cluster_order_best_index < 0 } }
                    set_temp_variable = { event_cluster_order_best_index = event_cluster_order_scan_index }
                    set_temp_variable = { event_cluster_order_best_score = event_cluster_order_candidate_score }
                    set_temp_variable = { event_cluster_order_best_event_id = event_cluster_order_candidate_event_id }
                }
                else_if = {
                    limit = { check_variable = { event_cluster_order_candidate_score < event_cluster_order_best_score } }
                    set_temp_variable = { event_cluster_order_best_index = event_cluster_order_scan_index }
                    set_temp_variable = { event_cluster_order_best_score = event_cluster_order_candidate_score }
                    set_temp_variable = { event_cluster_order_best_event_id = event_cluster_order_candidate_event_id }
                }
                else_if = {
                    limit = {
                        check_variable = { event_cluster_order_candidate_score = event_cluster_order_best_score }
                        check_variable = {
                            var = event_cluster_order_candidate_event_id
                            value = event_cluster_order_best_event_id
                            compare = less_than
                        }
                    }
                    set_temp_variable = { event_cluster_order_best_index = event_cluster_order_scan_index }
                    set_temp_variable = { event_cluster_order_best_score = event_cluster_order_candidate_score }
                    set_temp_variable = { event_cluster_order_best_event_id = event_cluster_order_candidate_event_id }
                }
            }
            add_to_temp_variable = { event_cluster_order_scan_index = 1 }
        }

        if = {
            limit = { check_variable = { event_cluster_order_best_index > -1 } }
            add_to_array = {
                array = temp_event_cluster_firing_order_event_id_entries
                value = temp_event_cluster_order_candidate_event_id_entries^event_cluster_order_best_index
            }
            set_variable = { temp_event_cluster_order_candidate_event_id_entries^event_cluster_order_best_index = -1 }
            add_to_temp_variable = { event_cluster_order_sorted_count = 1 }
        }
        else = {
            set_temp_variable = { event_cluster_order_sorted_count = event_cluster_order_source_count }
        }
    }
}

rebuild_events_log_cluster_view = {
    if = {
        limit = { NOT = { has_variable = settings_events_log_cluster_sort_mode } }
        set_variable = { settings_events_log_cluster_sort_mode = 0 }
    }
    if = {
        limit = { check_variable = { settings_events_log_cluster_sort_mode > 3 } }
        set_variable = { settings_events_log_cluster_sort_mode = 0 }
    }
    if = {
        limit = { check_variable = { settings_events_log_cluster_sort_mode < 0 } }
        set_variable = { settings_events_log_cluster_sort_mode = 0 }
    }
    if = {
        limit = { NOT = { has_variable = settings_events_log_cluster_filter_type } }
        set_variable = { settings_events_log_cluster_filter_type = 0 }
    }
    if = {
        limit = { check_variable = { settings_events_log_cluster_filter_type > 4 } }
        set_variable = { settings_events_log_cluster_filter_type = 0 }
    }
    if = {
        limit = { check_variable = { settings_events_log_cluster_filter_type < 0 } }
        set_variable = { settings_events_log_cluster_filter_type = 0 }
    }

    clear_array = global.events_log_cluster_view_sequence_entries
    clear_array = global.events_log_cluster_view_date_entries
    clear_array = global.events_log_cluster_view_id_entries
    clear_array = global.events_log_cluster_view_type_entries
    clear_array = global.events_log_cluster_view_tier_entries
    clear_array = global.events_log_cluster_view_trigger_event_id_entries
    clear_array = global.events_log_cluster_view_actor_entries
    clear_array = global.events_log_cluster_view_has_actor_entries
    clear_array = global.events_log_cluster_view_member_count_entries
    clear_array = global.events_log_cluster_view_fired_count_entries
    clear_array = global.events_log_cluster_view_skipped_count_entries
    clear_array = global.events_log_cluster_view_unlock_tier_entries
    clear_array = global.events_log_cluster_view_roll_chance_entries
    clear_array = global.events_log_cluster_view_available_entries
    clear_array = global.events_log_cluster_view_enabled_entries

    clear_array = temp_events_log_cluster_source_sequence_entries
    clear_array = temp_events_log_cluster_source_date_entries
    clear_array = temp_events_log_cluster_source_id_entries
    clear_array = temp_events_log_cluster_source_type_entries
    clear_array = temp_events_log_cluster_source_tier_entries
    clear_array = temp_events_log_cluster_source_trigger_event_id_entries
    clear_array = temp_events_log_cluster_source_actor_entries
    clear_array = temp_events_log_cluster_source_has_actor_entries
    clear_array = temp_events_log_cluster_source_member_count_entries
    clear_array = temp_events_log_cluster_source_fired_count_entries
    clear_array = temp_events_log_cluster_source_skipped_count_entries
    clear_array = temp_events_log_cluster_source_unlock_tier_entries
    clear_array = temp_events_log_cluster_source_roll_chance_entries
    clear_array = temp_events_log_cluster_source_available_entries
    clear_array = temp_events_log_cluster_source_enabled_entries

    for_loop_effect = {
        start = 0
        end = global.event_clusters^num
        value = i

        set_temp_variable = { event_cluster_id = global.event_clusters^i }
        get_event_cluster_type = yes
        get_event_cluster_unlock_tier = yes
        get_event_cluster_roll_chance = yes
        load_event_cluster_members = yes
        can_event_cluster_fire = yes
        get_event_cluster_enabled_state = yes
        count_event_cluster_history_entries = yes

        set_temp_variable = { events_log_cluster_include = 1 }
        if = {
            limit = {
                check_variable = { settings_events_log_cluster_filter_type = 1 }
                check_variable = {
                    var = event_cluster_can_fire
                    value = 0
                    compare = less_than_or_equals
                }
            }
            set_temp_variable = { events_log_cluster_include = 0 }
        }
        else_if = {
            limit = {
                check_variable = { settings_events_log_cluster_filter_type = 2 }
                check_variable = {
                    var = event_cluster_can_fire
                    value = 0
                    compare = greater_than
                }
            }
            set_temp_variable = { events_log_cluster_include = 0 }
        }
        else_if = {
            limit = {
                check_variable = { settings_events_log_cluster_filter_type = 3 }
                check_variable = { event_cluster_enabled_state < 1 }
            }
            set_temp_variable = { events_log_cluster_include = 0 }
        }
        else_if = {
            limit = {
                check_variable = { settings_events_log_cluster_filter_type = 4 }
                check_variable = { event_cluster_enabled_state > 0 }
            }
            set_temp_variable = { events_log_cluster_include = 0 }
        }

        if = {
            limit = { check_variable = { events_log_cluster_include > 0 } }
            add_to_array = { array = temp_events_log_cluster_source_sequence_entries value = 0 }
            add_to_array = { array = temp_events_log_cluster_source_date_entries value = 0 }
            add_to_array = { array = temp_events_log_cluster_source_id_entries value = event_cluster_id }
            add_to_array = { array = temp_events_log_cluster_source_type_entries value = event_cluster_type_value }
            add_to_array = { array = temp_events_log_cluster_source_tier_entries value = current_chaos_tier_value }
            add_to_array = { array = temp_events_log_cluster_source_trigger_event_id_entries value = 0 }
            add_to_array = { array = temp_events_log_cluster_source_actor_entries value = 0 }
            add_to_array = { array = temp_events_log_cluster_source_has_actor_entries value = 0 }
            add_to_array = { array = temp_events_log_cluster_source_member_count_entries value = temp_event_cluster_member_event_id_entries^num }
            add_to_array = { array = temp_events_log_cluster_source_fired_count_entries value = event_cluster_history_count }
            add_to_array = { array = temp_events_log_cluster_source_skipped_count_entries value = 0 }
            add_to_array = { array = temp_events_log_cluster_source_unlock_tier_entries value = event_cluster_unlock_tier }
            add_to_array = { array = temp_events_log_cluster_source_roll_chance_entries value = event_cluster_roll_chance }
            add_to_array = { array = temp_events_log_cluster_source_available_entries value = event_cluster_can_fire }
            add_to_array = { array = temp_events_log_cluster_source_enabled_entries value = event_cluster_enabled_state }
        }
    }

    set_temp_variable = { events_log_cluster_source_count = temp_events_log_cluster_source_id_entries^num }
    set_temp_variable = { events_log_cluster_sorted_count = 0 }
    while_loop_effect = {
        limit = { check_variable = { events_log_cluster_sorted_count < events_log_cluster_source_count } }

        set_temp_variable = { events_log_cluster_best_index = -1 }
        set_temp_variable = { events_log_cluster_scan_index = 0 }
        while_loop_effect = {
            limit = { check_variable = { events_log_cluster_scan_index < events_log_cluster_source_count } }
            if = {
                limit = {
                    check_variable = {
                        var = temp_events_log_cluster_source_id_entries^events_log_cluster_scan_index
                        value = 0
                        compare = greater_than
                    }
                }

                set_temp_variable = { events_log_cluster_candidate_metric = temp_events_log_cluster_source_id_entries^events_log_cluster_scan_index }
                if = {
                    limit = { check_variable = { settings_events_log_cluster_sort_mode = 1 } }
                    set_temp_variable = { events_log_cluster_candidate_metric = temp_events_log_cluster_source_type_entries^events_log_cluster_scan_index }
                }
                else_if = {
                    limit = { check_variable = { settings_events_log_cluster_sort_mode = 2 } }
                    set_temp_variable = { events_log_cluster_candidate_metric = temp_events_log_cluster_source_roll_chance_entries^events_log_cluster_scan_index }
                }
                else_if = {
                    limit = { check_variable = { settings_events_log_cluster_sort_mode = 3 } }
                    set_temp_variable = { events_log_cluster_candidate_metric = temp_events_log_cluster_source_fired_count_entries^events_log_cluster_scan_index }
                }

                if = {
                    limit = { check_variable = { events_log_cluster_best_index < 0 } }
                    set_temp_variable = { events_log_cluster_best_index = events_log_cluster_scan_index }
                    set_temp_variable = { events_log_cluster_best_metric = events_log_cluster_candidate_metric }
                    set_temp_variable = { events_log_cluster_best_id = temp_events_log_cluster_source_id_entries^events_log_cluster_scan_index }
                }
                else_if = {
                    limit = {
                        has_country_flag = events_log_cluster_sort_descending
                        check_variable = { events_log_cluster_candidate_metric > events_log_cluster_best_metric }
                    }
                    set_temp_variable = { events_log_cluster_best_index = events_log_cluster_scan_index }
                    set_temp_variable = { events_log_cluster_best_metric = events_log_cluster_candidate_metric }
                    set_temp_variable = { events_log_cluster_best_id = temp_events_log_cluster_source_id_entries^events_log_cluster_scan_index }
                }
                else_if = {
                    limit = {
                        NOT = { has_country_flag = events_log_cluster_sort_descending }
                        check_variable = { events_log_cluster_candidate_metric < events_log_cluster_best_metric }
                    }
                    set_temp_variable = { events_log_cluster_best_index = events_log_cluster_scan_index }
                    set_temp_variable = { events_log_cluster_best_metric = events_log_cluster_candidate_metric }
                    set_temp_variable = { events_log_cluster_best_id = temp_events_log_cluster_source_id_entries^events_log_cluster_scan_index }
                }
                else_if = {
                    limit = { check_variable = { events_log_cluster_candidate_metric = events_log_cluster_best_metric } }
                    if = {
                        limit = {
                            has_country_flag = events_log_cluster_sort_descending
                            check_variable = {
                                var = temp_events_log_cluster_source_id_entries^events_log_cluster_scan_index
                                value = events_log_cluster_best_id
                                compare = greater_than
                            }
                        }
                        set_temp_variable = { events_log_cluster_best_index = events_log_cluster_scan_index }
                        set_temp_variable = { events_log_cluster_best_metric = events_log_cluster_candidate_metric }
                        set_temp_variable = { events_log_cluster_best_id = temp_events_log_cluster_source_id_entries^events_log_cluster_scan_index }
                    }
                    else_if = {
                        limit = {
                            NOT = { has_country_flag = events_log_cluster_sort_descending }
                            check_variable = {
                                var = temp_events_log_cluster_source_id_entries^events_log_cluster_scan_index
                                value = events_log_cluster_best_id
                                compare = less_than
                            }
                        }
                        set_temp_variable = { events_log_cluster_best_index = events_log_cluster_scan_index }
                        set_temp_variable = { events_log_cluster_best_metric = events_log_cluster_candidate_metric }
                        set_temp_variable = { events_log_cluster_best_id = temp_events_log_cluster_source_id_entries^events_log_cluster_scan_index }
                    }
                }
            }
            add_to_temp_variable = { events_log_cluster_scan_index = 1 }
        }

        if = {
            limit = { check_variable = { events_log_cluster_best_index > -1 } }
            events_log_cluster_append_view_row_from_source = yes
            set_variable = { temp_events_log_cluster_source_id_entries^events_log_cluster_best_index = -1 }
            add_to_temp_variable = { events_log_cluster_sorted_count = 1 }
        }
        else = {
            set_temp_variable = { events_log_cluster_sorted_count = events_log_cluster_source_count }
        }
    }
}

        force_fire_event_cluster_by_temp_id = yes
    }
    events_log_refresh_selected_cluster_details_state = yes
    events_log_rebuild_selected_cluster_details_view = yes
    rebuild_events_log_cluster_view = yes
}

events_log_cluster_sort_detail_members_by_danger = {
    clear_array = global.events_log_cluster_detail_member_event_id_entries
    clear_array = global.events_log_cluster_detail_member_status_entries
    clear_array = global.events_log_cluster_detail_member_danger_entries

    set_temp_variable = { events_log_cluster_detail_source_count = temp_events_log_cluster_detail_source_event_id_entries^num }
    set_temp_variable = { events_log_cluster_detail_sorted_count = 0 }
    while_loop_effect = {
        limit = { check_variable = { events_log_cluster_detail_sorted_count < events_log_cluster_detail_source_count } }

        set_temp_variable = { events_log_cluster_detail_best_index = -1 }
        set_temp_variable = { events_log_cluster_detail_scan_index = 0 }
        while_loop_effect = {
            limit = { check_variable = { events_log_cluster_detail_scan_index < events_log_cluster_detail_source_count } }
            if = {
                limit = {
                    check_variable = {
                        var = temp_events_log_cluster_detail_source_event_id_entries^events_log_cluster_detail_scan_index
                        value = 0
                        compare = greater_than
                    }
                }
                set_temp_variable = { events_log_cluster_detail_candidate_danger = temp_events_log_cluster_detail_source_danger_entries^events_log_cluster_detail_scan_index }
                set_temp_variable = { events_log_cluster_detail_candidate_event_id = temp_events_log_cluster_detail_source_event_id_entries^events_log_cluster_detail_scan_index }

                if = {
                    limit = { check_variable = { events_log_cluster_detail_best_index < 0 } }
                    set_temp_variable = { events_log_cluster_detail_best_index = events_log_cluster_detail_scan_index }
                    set_temp_variable = { events_log_cluster_detail_best_danger = events_log_cluster_detail_candidate_danger }
                    set_temp_variable = { events_log_cluster_detail_best_event_id = events_log_cluster_detail_candidate_event_id }
                }
                else_if = {
                    limit = { check_variable = { events_log_cluster_detail_candidate_danger < events_log_cluster_detail_best_danger } }
                    set_temp_variable = { events_log_cluster_detail_best_index = events_log_cluster_detail_scan_index }
                    set_temp_variable = { events_log_cluster_detail_best_danger = events_log_cluster_detail_candidate_danger }
                    set_temp_variable = { events_log_cluster_detail_best_event_id = events_log_cluster_detail_candidate_event_id }
                }
                else_if = {
                    limit = {
                        check_variable = { events_log_cluster_detail_candidate_danger = events_log_cluster_detail_best_danger }
                        check_variable = {
                            var = events_log_cluster_detail_candidate_event_id
                            value = events_log_cluster_detail_best_event_id
                            compare = less_than
                        }
                    }
                    set_temp_variable = { events_log_cluster_detail_best_index = events_log_cluster_detail_scan_index }
                    set_temp_variable = { events_log_cluster_detail_best_danger = events_log_cluster_detail_candidate_danger }
                    set_temp_variable = { events_log_cluster_detail_best_event_id = events_log_cluster_detail_candidate_event_id }
                }
            }
            add_to_temp_variable = { events_log_cluster_detail_scan_index = 1 }
        }

        if = {
            limit = { check_variable = { events_log_cluster_detail_best_index > -1 } }
            events_log_cluster_append_detail_member_from_source = yes
            set_variable = { temp_events_log_cluster_detail_source_event_id_entries^events_log_cluster_detail_best_index = -1 }
            add_to_temp_variable = { events_log_cluster_detail_sorted_count = 1 }
        }
        else = {
            set_temp_variable = { events_log_cluster_detail_sorted_count = events_log_cluster_detail_source_count }
        }
    }
}

events_log_rebuild_selected_cluster_details_view = {
    clear_array = temp_events_log_cluster_detail_source_event_id_entries
    clear_array = temp_events_log_cluster_detail_source_status_entries
    clear_array = temp_events_log_cluster_detail_source_danger_entries

    if = {
        limit = { check_variable = { events_log_selected_cluster_sequence > 0 } }
        for_loop_effect = {
            start = 0
            end = global.events_log_cluster_member_cluster_sequence_entries^num
            value = i

            if = {
                limit = {
                    check_variable = {
                        var = global.events_log_cluster_member_cluster_sequence_entries^i
                        value = events_log_selected_cluster_sequence
                        compare = equals
                    }
                }
                set_temp_variable = { events_log_cluster_detail_source_event_id = global.events_log_cluster_member_event_id_entries^i }
                set_temp_variable = { events_log_cluster_detail_source_status = global.events_log_cluster_member_status_entries^i }
                set_temp_variable = { events_log_cluster_detail_source_danger = global.events_log_cluster_member_danger_entries^i }
                events_log_cluster_append_detail_member_source = yes
            }
        }
    }
    else_if = {
        limit = { check_variable = { events_log_selected_cluster_id > 0 } }
        set_temp_variable = { event_cluster_id = events_log_selected_cluster_id }
        load_event_cluster_members = yes
        for_loop_effect = {
            start = 0
            end = temp_event_cluster_member_event_id_entries^num
            value = i

            set_temp_variable = { event_cluster_member_event_id = temp_event_cluster_member_event_id_entries^i }
            set_temp_variable = { event_cluster_member_min_tier = temp_event_cluster_member_min_tier_entries^i }
            event_cluster_member_check_availability = yes
            if = {
                limit = {
                    check_variable = {
                        var = event_cluster_member_can_fire
                        value = 0
                        compare = greater_than
                    }
                }
                set_temp_variable = { event_cluster_member_status = constant:event_cluster_member_status.available }
            }
            set_temp_variable = { events_log_cluster_detail_source_event_id = event_cluster_member_event_id }
            set_temp_variable = { events_log_cluster_detail_source_status = event_cluster_member_status }
            set_temp_variable = { events_log_cluster_detail_source_danger = temp_event_cluster_member_danger_entries^i }
            events_log_cluster_append_detail_member_source = yes
        }
    }
    events_log_cluster_sort_detail_members_by_danger = yes
}
