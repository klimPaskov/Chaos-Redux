on_actions = {

    # Initialization and activation of the event system
    on_startup = {
        effect = {
            # Global initialization
            random_country = {
                initialize_event_system = yes
                initialize_chaos_meter_system = yes
                initialize_global_settings_system = yes
            }

            # Player(s) initialization
            every_country = {
                limit = { is_ai = no }

                set_country_flag = events_activated # Event system activation

                # Initialize tag switch detection variable for current human players
                set_country_flag = was_human_yesterday
            }

            # All countries initialization
            every_possible_country = {
                limit = { always = yes }

	                set_variable_to_random = {
	                    var = event_timer_days
	                    min = constant:event_system_timer_random_start.min
	                    max = constant:event_system_timer_random_start.max
	                    integer = yes
	                }
                if = {
                    limit = { is_ai = no }
                    log = "CHAOS REDUX EVENT SYSTEM: First event timer set to [?event_timer_days] days for [THIS.GetName]"
                }

                initialize_dynamic_timer = yes
                initialize_settings_system = yes
            }

            random_country = {
                limit = { is_ai = no }
                set_country_flag = is_global_host
            }

            every_country = {
                limit = { is_ai = no }
                set_country_flag = chaosx_welcome_flag # Welcome event
            }
        }
    }

    # Daily check (for every country)
    on_daily = {
        effect = {
            # Tag switch detection system
            if = {
                limit = {
                    OR = {
                        has_global_flag = settings_auto_enable_on_switch
                        has_global_flag = settings_disable_previous_tag
                    }
                }

                # Initialize tracking flag if not set
                if = {
                    limit = { NOT = { has_country_flag = was_human_yesterday } }
                    clr_country_flag = was_human_yesterday
                }

                # Check for tag switch TO this country (now human, wasn't yesterday)
                if = {
                    limit = {
                        is_ai = no
                        NOT = { has_country_flag = was_human_yesterday }
                        has_global_flag = settings_auto_enable_on_switch
                    }

                    log = "CHAOS REDUX TAG MANAGEMENT: New human player detected for [This.GetTag] ([This.GetName])"

                    # Auto-enable events for new human player
                    if = {
                        limit = { NOT = { has_country_flag = events_activated } }

                        set_country_flag = events_activated

                        # Add to enabled countries list
                        if = {
                            limit = { NOT = { is_in_array = { array = global.enabled_countries_list value = THIS } } }
                            add_to_array = { array = global.enabled_countries_list value = THIS }
                            add_to_variable = { global.enabled_countries_count = 1 }
                            if = {
                                limit = { check_variable = { global.enabled_countries_count > 7 } }
                                set_temp_variable = { i = global.enabled_countries_count }
                                subtract_from_temp_variable = { i = 7 }
                            }
                        }

                        log = "CHAOS REDUX TAG MANAGEMENT: Auto-enabled event system for new human player [This.GetTag] ([This.GetName])"
                    }
                }

                # Check for tag switch FROM this country (now AI, was human yesterday)
                if = {
                    limit = {
                        is_ai = yes
                        has_country_flag = was_human_yesterday
                        has_global_flag = settings_disable_previous_tag
                    }

                    log = "CHAOS REDUX TAG MANAGEMENT: Former human player detected for [This.GetTag] ([This.GetName])"

                    # Auto-disable events for former human player
                    if = {
                        limit = { has_country_flag = events_activated }

                        clr_country_flag = events_activated

                        # Remove from enabled countries list
                        if = {
                            limit = { is_in_array = { array = global.enabled_countries_list value = THIS } }
                            remove_from_array = { array = global.enabled_countries_list value = THIS }
                            subtract_from_variable = { global.enabled_countries_count = 1 }
                        }

                        log = "CHAOS REDUX TAG MANAGEMENT: Auto-disabled event system for former human player [This.GetTag] ([This.GetName])"
                    }
                }

                # Update tracking flag for next day
                if = {
                    limit = { is_ai = no }
                    set_country_flag = was_human_yesterday
                }
                else = {
                    clr_country_flag = was_human_yesterday
                }
            }

            # Handle event timers for countries with events activated
            if = {
                limit = {
                    has_country_flag = events_activated
                    NOT = { has_global_flag = world_end }
                }

                # Handle dynamic event timer
                check_event_timer = yes

                # If timer reached 0, fire events
                if = {
                    limit = { check_variable = { event_timer_days = 0 } }

					set_temp_variable = { random_event_selection_filter = constant:settings_event_filter.all }
					select_weighted_random_event_id = yes

					if = {
						limit = { has_variable = random_event_selection_id }
						set_temp_variable = { event_id = random_event_selection_id }
						fire_event_by_temp_id = yes
					}

                    # Log debug information after events are processed
                    if = {
                        limit = { has_global_flag = settings_logic_log_lines_enabled }
                        log_event_system_debug = yes
                    }
                }
            }
        }
    }

}
