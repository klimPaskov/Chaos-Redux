defined_text = {
	name = ChemWindStatus
	text = {
		trigger = {
			OR = {
				check_variable = { var = chem_wind_state value = constant:chem_wind_state.strong_favor compare = greater_than_or_equals }
				OWNER = { check_variable = { var = chem_wind_state value = constant:chem_wind_state.strong_favor compare = greater_than_or_equals } }
			}
		}
		localization_key = CHEM_ABILITY_WIND_STRONG_FAVORABLE
	}
	text = {
		trigger = {
			OR = {
				check_variable = { var = chem_wind_state value = constant:chem_wind_state.favor compare = greater_than_or_equals }
				OWNER = { check_variable = { var = chem_wind_state value = constant:chem_wind_state.favor compare = greater_than_or_equals } }
			}
		}
		localization_key = CHEM_ABILITY_WIND_FAVORABLE
	}
	text = {
		trigger = {
			OR = {
				check_variable = { var = chem_wind_state value = constant:chem_wind_state.against compare = less_than_or_equals }
				OWNER = { check_variable = { var = chem_wind_state value = constant:chem_wind_state.against compare = less_than_or_equals } }
			}
		}
		localization_key = CHEM_ABILITY_WIND_AGAINST
	}
	text = {
		trigger = { always = yes }
		localization_key = CHEM_ABILITY_WIND_NEUTRAL
	}
}
