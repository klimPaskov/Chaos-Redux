# Chemical tank-shell scripted triggers:
# - Shared country-template detection for per-agent chemical tank support companies.
# - Uses country-scope template checks so it is valid in army-leader on_actions.

chem_has_chlorine_tank_support_in_template = {
	FROM = {
		OR = {
			has_template_containing_unit = light_chemical_tank_support_chlorine
			has_template_containing_unit = medium_chemical_tank_support_chlorine
			has_template_containing_unit = heavy_chemical_tank_support_chlorine
		}
	}
}
