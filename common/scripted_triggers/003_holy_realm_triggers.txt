holy_realm_peace_letter_target_valid = {
	is_holy_realm_country = yes
	FROM = {
		exists = yes
		NOT = { is_holy_realm_country = yes }
		NOT = { is_subject_of = ROOT }
		is_subject = no
		NOT = { has_war_with = ROOT }
		NOT = { has_country_flag = holy_realm_peace_letter_pending }
		NOT = { has_country_flag = holy_realm_peace_letter_accepted }
		NOT = { has_country_flag = holy_realm_peace_letter_blocked }
		NOT = { has_country_flag = holy_realm_mandala_protectorate }
	}
	OR = {
		FROM = { is_neighbor_of = ROOT }
		any_subject_country = {
			FROM = { is_neighbor_of = PREV }
		}
		AND = {
			has_country_flag = holy_realm_buddha_expansion_authority
			FROM = {
				any_owned_state = {
					OR = {
						holy_realm_himalayan_core_state = yes
						holy_realm_northern_indian_core_state = yes
						holy_realm_western_chinese_core_state = yes
						holy_realm_eastern_chinese_core_state = yes
					}
				}
			}
		}
	}
	OR = {
		FROM = {
			is_major = no
			is_in_faction = no
		}
		has_country_flag = holy_realm_world_asked_to_kneel_unlocked
	}
}
