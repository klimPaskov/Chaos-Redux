is_fury_actor = {
	has_country_flag = fury_actor
}

fury_is_player_linked_country = {
	OR = {
		is_ai = no
		any_country = {
			is_ai = no
			PREV = { is_subject_of = PREV }
		}
		any_allied_country = { is_ai = no }
	}
}

fury_can_be_selected = {
	exists = yes
	is_ai = yes
	is_major = no
	is_subject = no
	uses_normal_civilian_systems = yes
	NOT = { has_capitulated = yes }
	NOT = { any_enemy_country = { is_ai = no } }
	NOT = { has_country_flag = fury_actor }
	NOT = { has_country_flag = fury_recent_actor }
	NOT = { fury_is_player_linked_country = yes }
	capital_scope = { is_island_state = no }
	any_neighbor_country = {
		count = 2
		fury_is_valid_target = yes
	}
}

fury_can_be_selected_one_state = {
	fury_can_be_selected = yes
	num_of_controlled_states < constant:fury_target_score.one_state_max
}

fury_can_be_selected_two_state = {
	fury_can_be_selected = yes

}

fury_can_be_selected_three_state = {
	fury_can_be_selected = yes
	num_of_controlled_states < constant:fury_target_score.three_state_max
}

fury_actor_can_continue = {
	has_country_flag = fury_actor
	exists = yes
	NOT = { has_capitulated = yes }
	num_of_controlled_states > 0
}

fury_world_end_basic_readiness = {
	is_fury_actor = yes
	has_country_flag = fury_world_end_candidate
	has_global_flag = fury_world_end_candidate
	NOT = { has_global_flag = world_end }
	NOT = { has_global_flag = world_end_disabled }
	check_variable = {
		var = global.chaos_meter_value
		value = constant:chaos_meter_tier_range.tier_final.min
		compare = greater_than_or_equals
	}
	OR = {
		has_country_flag = fury_no_neutral_map_unlocked
		is_major = yes
		num_of_controlled_states > constant:fury_balance.world_end_actor_state_threshold
		NOT = { any_neighbor_country = { fury_is_valid_target = yes } }
	}
	OR = {
		NOT = { fury_occupation_pressure_extreme = yes }
		has_country_flag = fury_no_neutral_map_unlocked
	}
}

fury_world_end_safe_seed_candidate = {
	fury_can_be_selected = yes
	NOT = { has_global_flag = fury_triggerable_scenario_launch_active }
}

fury_is_valid_target = {
	exists = yes
	is_subject = no
	uses_normal_civilian_systems = yes
	NOT = { has_capitulated = yes }
	NOT = { has_country_flag = fury_actor }
	NOT = { is_subject_of = ROOT }
	NOT = { is_in_faction_with = ROOT }
	NOT = { has_war_with = ROOT }
	NOT = {
		AND = {
			is_major = yes
			ROOT = {
				is_major = no
				NOT = { has_global_flag = fury_world_end_active }
			}
		}
	}
	ROOT = {
		OR = {
			has_country_flag = fury_evolution_iii_applied
			has_global_flag = fury_world_end_active
			NOT = { fury_occupation_pressure_extreme = yes }
		}
	}
	ROOT = { can_declare_war_on = PREV }
}
