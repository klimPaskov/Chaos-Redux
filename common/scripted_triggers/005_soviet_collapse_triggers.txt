is_soviet_collapse_active = {
	has_global_flag = soviet_collapse_active
}

is_soviet_collapse_breakaway_country = {
	OR = {
		has_country_flag = soviet_collapse_breakaway
		tag = UKR
		tag = BLR
		tag = KAZ
		tag = LIT
		tag = LAT
		tag = EST
		tag = GEO
		tag = ARM
		tag = AZR
		tag = MOL
		tag = UZB
		tag = KYR
		tag = TAJ
		tag = TMS
	}
}

is_soviet_collapse_foreign_patron_candidate = {
	NOT = { tag = SOV }
	is_major = yes
	OR = {
		has_war_with = SOV
		has_opinion = { target = SOV value < 0 }
	}
}

can_pay_soviet_collapse_restore_party_control_cost = {
	has_equipment = { support_equipment > constant:soviet_collapse_decision_cost.restore_party_control_support_equipment }
	has_manpower > constant:soviet_collapse_decision_cost.restore_party_control_manpower
}

can_pay_soviet_collapse_send_loyalist_officers_cost = {
	has_army_experience > constant:soviet_collapse_decision_cost.send_loyalist_officers_army_xp
	has_equipment = { infantry_equipment > constant:soviet_collapse_decision_cost.send_loyalist_officers_equipment }
}

can_pay_soviet_collapse_cut_rebel_supply_routes_cost = {
	has_fuel > constant:soviet_collapse_decision_cost.cut_rebel_supply_routes_fuel
	has_equipment = { train_equipment > constant:soviet_collapse_decision_cost.cut_rebel_supply_routes_trains }
}

	FROM = {
		exists = yes
		has_country_flag = soviet_collapse_breakaway
		is_soviet_collapse_breakaway_country = yes
		NOT = { tag = SOV }
		NOT = { tag = ROOT }
	}
	NOT = { tag = SOV }
}

is_soviet_collapse_league_pressure_ready = {
	OR = {
		has_war_with = SOV
		check_variable = { var = soviet_collapse_recognition_progress value = constant:soviet_collapse_republic_focus.large_recognition_gain compare = greater_than }
		SOV = {
			OR = {
				check_variable = { var = soviet_collapse_total_collapse_threat value = constant:soviet_collapse_republic_focus.league_threat_gate compare = greater_than }
				check_variable = { var = soviet_collapse_republic_confidence value = constant:soviet_collapse_republic_focus.league_threat_gate compare = greater_than }
			}
		}
	}
}

is_soviet_collapse_southern_breakaway_active = {
	OR = {
		UZB = { exists = yes has_country_flag = soviet_collapse_breakaway }
		KYR = { exists = yes has_country_flag = soviet_collapse_breakaway }
		TAJ = { exists = yes has_country_flag = soviet_collapse_breakaway }
		TMS = { exists = yes has_country_flag = soviet_collapse_breakaway }
	}
}

can_soviet_collapse_open_kazakhstan_first_wave = {
	OR = {
		is_soviet_collapse_southern_breakaway_active = yes
		AND = {
			has_global_flag = { flag = chaos_tier value = 3 }
			OR = {
				has_war = yes
				has_stability < constant:soviet_collapse_soviet_objective.min_stability
				has_war_support < constant:soviet_collapse_soviet_objective.min_war_support
			}
		}
		has_global_flag = { flag = chaos_tier value = 4 }
		has_global_flag = { flag = chaos_tier value = 5 }
	}
}
