	88 = { is_controlled_by = ROOT }
}

	exists = yes
	original_tag = GER
	NOT = { has_country_flag = germany_mengele_faction }
}

germany_mengele_can_start_chain = {
	germany_mengele_is_germany_scope = yes
	has_government = fascism
	date > 1940.06.13
	germany_controls_auschwitz_area = yes
	NOT = { has_country_flag = germany_mengele_program_rejected }
	NOT = { has_country_flag = germany_mengele_program_closed }
	NOT = { has_country_flag = germany_mengele_program_authorized }
	NOT = { has_country_flag = germany_mengele_program_restricted }
}

	has_war = yes

		surrender_progress > constant:germany_mengele_ai.desperate_surrender_threshold

		}
	}
}

	OR = {
		AND = {
			has_country_flag = mengele_clone_decision_streamlined_deployment_unlocked
			command_power > constant:mengele_clone_decision_threshold.raise_clone_wave_streamlined_command_power
		}
		AND = {
			has_country_flag = mengele_clone_decision_discounted_deployment_unlocked
			NOT = { has_country_flag = mengele_clone_decision_streamlined_deployment_unlocked }
			command_power > constant:mengele_clone_decision_threshold.raise_clone_wave_discount_command_power
		}
		AND = {
			NOT = { has_country_flag = mengele_clone_decision_discounted_deployment_unlocked }
			NOT = { has_country_flag = mengele_clone_decision_streamlined_deployment_unlocked }
			command_power > constant:mengele_clone_decision_threshold.raise_clone_wave_command_power
		}
	}
}
