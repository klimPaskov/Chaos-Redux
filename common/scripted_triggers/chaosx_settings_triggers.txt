check_event_availability = {
    meta_trigger = {
        text = {
            has_global_flag = { flag = event_disabled value = [EVENT_ID] }
        }
        EVENT_ID = event_id
    }
}

has_disabled_current_evolution = {
    meta_trigger = {
        text = {
            has_global_flag = {
                flag = events_log_disabled_evolution_[EVOLUTION_EVENT_ID]_[EVOLUTION_TYPE]_[EVOLUTION_STAGE]
            }
        }
        EVOLUTION_EVENT_ID = "[?events_log_evolution_event_id|0]"
        EVOLUTION_TYPE = "[?events_log_evolution_type|0]"
        EVOLUTION_STAGE = "[?events_log_evolution_stage|0]"
    }
}

is_current_evolution_enabled = {
    NOT = { has_disabled_current_evolution = yes }
}

has_events_log_selected_fury_evolution = {
	check_variable = {
		var = events_log_selected_evolution_event_id
		value = constant:fury_event_log.event_id
		compare = equals
	}
	check_variable = {
		var = events_log_selected_evolution_type
		value = constant:fury_event_log.evolution_type
		compare = equals
	}
	OR = {
		check_variable = {
			var = events_log_selected_evolution_stage
			value = constant:fury_event_log.hardened_stage
			compare = equals
		}
		check_variable = {
			var = events_log_selected_evolution_stage
			value = constant:fury_event_log.two_fires_stage
			compare = equals
		}
		check_variable = {
			var = events_log_selected_evolution_stage
			value = constant:fury_event_log.all_borders_stage
			compare = equals
		}
	}
}

has_events_log_selected_soviet_collapse_evolution = {
	check_variable = {
		var = events_log_selected_evolution_event_id
		value = constant:soviet_collapse_event_log.event_id
		compare = equals
	}
	OR = {
		AND = {
			check_variable = {
				var = events_log_selected_evolution_type
				value = constant:soviet_collapse_event_log.secession_evolution_type
				compare = equals
			}
			check_variable = {
				var = events_log_selected_evolution_stage
				value = 0
				compare = greater_than
			}
			check_variable = {
				var = events_log_selected_evolution_stage
				value = constant:soviet_collapse_event_log.world_collapse_stage
				compare = less_than_or_equals
			}
		}
		AND = {
			check_variable = {
				var = events_log_selected_evolution_type
				value = constant:soviet_collapse_high_chaos_event_log.evolution_type
				compare = equals
			}
			check_variable = {
				var = events_log_selected_evolution_stage
				value = 0
				compare = greater_than
			}
		}
	}
}
