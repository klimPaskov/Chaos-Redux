    OR = {
        original_tag = ZZZ
        has_country_flag = zombie_outbreak_dynamic_country
    }
}

is_in_anti_zombie_league_faction = {
	OR = {
		AND = {
			has_country_flag = azl_leader
			is_faction_leader = yes
			is_in_faction = yes
		}
		AND = {
			is_in_faction = yes
			has_event_target = azl_faction_leader
			event_target:azl_faction_leader = {
				is_faction_leader = yes
			}
			is_in_faction_with = event_target:azl_faction_leader
		}
	}
}

is_anti_zombie_league_member = {
	AND = {
		has_anti_zombie_league_membership_idea = yes
		is_in_anti_zombie_league_faction = yes
	}
}

can_lead_anti_zombie_league = {
    is_major = yes
    NOT = { original_tag = ZZZ }
    NOT = { has_country_flag = zombie_outbreak_dynamic_country }
    NOT = { has_country_flag = weaponized_zombie_outbreak_country }
    is_subject = no
}

can_join_anti_zombie_league = {
    NOT = { original_tag = ZZZ }
    NOT = { has_country_flag = zombie_outbreak_dynamic_country }
    NOT = { has_country_flag = weaponized_zombie_outbreak_country }
    NOT = { is_anti_zombie_league_member = yes }
    exists = yes
    is_subject = no
}

zombie_threat_confined_to_americas = {
	OR = {
		any_country = {
			exists = yes
			OR = {
				is_zombie_outbreak_country = yes
				is_weaponized_zombie_outbreak_country = yes
			}
			any_controlled_state = {
				is_island_state = no
				OR = {
					is_on_continent = north_america
					is_on_continent = south_america
				}
			}
		}
		any_country = {
			exists = yes
			OR = {
				is_zombie_outbreak_country = yes
				is_weaponized_zombie_outbreak_country = yes
			}
			capital_scope = {
				OR = {
					is_on_continent = north_america
					is_on_continent = south_america
				}
			}
		}
	}
	NOT = {
		any_country = {
			exists = yes
			OR = {
				is_zombie_outbreak_country = yes
				is_weaponized_zombie_outbreak_country = yes
			}
			any_controlled_state = {
				is_island_state = no
				NOT = {
					OR = {
						is_on_continent = north_america
						is_on_continent = south_america
					}
				}
			}
		}
	}
}

is_outside_americas_country = {

		OR = {

		}
	}
}

can_request_zombie_evacuation = {
    uses_normal_civilian_systems = yes
    NOT = { has_capitulated = yes }
    NOT = { has_country_flag = zombie_evacuation_request_pending }
    NOT = { has_country_flag = zombie_evacuation_cooldown }
    is_at_war_with_any_zombie_country = yes
    surrender_progress > constant:zombie_evacuation.source_surrender_threshold
    num_of_controlled_states > 0
    check_variable = {
        var = num_equipment@convoy
        value = 0
        compare = greater_than
    }
    any_controlled_state = {
        state_population_k > 0
    }
}

is_zombie_evacuation_host_candidate = {
    uses_normal_civilian_systems = yes
    exists = yes
    NOT = { has_capitulated = yes }
    surrender_progress < constant:zombie_evacuation.host_surrender_threshold

        is_island_state = yes
        is_controlled_by = PREV

can_receive_zombie_evacuees = {

is_medium_risk_rear_zombie_outbreak_state = {
    is_rear_zombie_outbreak_state = yes
    state_population_k > constant:zombie_outbreak.state_population_medium_risk_threshold
    OR = {
        has_state_category = city
        has_state_category = large_city
        has_state_category = metropolis
        has_state_category = megalopolis
        infrastructure < constant:zombie_outbreak.low_infrastructure_threshold
        AND = {
            industrial_complex < constant:zombie_outbreak.low_industry_threshold
            arms_factory < constant:zombie_outbreak.low_industry_threshold
        }
    }
}

is_high_risk_rear_zombie_outbreak_state = {
    is_medium_risk_rear_zombie_outbreak_state = yes
    state_population_k > constant:zombie_outbreak.state_population_high_risk_threshold
    OR = {
        AND = {
            OR = {
                has_state_category = city
                has_state_category = large_city
                has_state_category = metropolis
                has_state_category = megalopolis
            }
            infrastructure < constant:zombie_outbreak.low_infrastructure_threshold
        }
        AND = {
            OR = {
                has_state_category = city
                has_state_category = large_city
                has_state_category = metropolis
                has_state_category = megalopolis
            }
            industrial_complex < constant:zombie_outbreak.low_industry_threshold
            arms_factory < constant:zombie_outbreak.low_industry_threshold
        }
        AND = {
            infrastructure < constant:zombie_outbreak.low_infrastructure_threshold
            industrial_complex < constant:zombie_outbreak.low_industry_threshold
            arms_factory < constant:zombie_outbreak.low_industry_threshold
        }
        AND = {
            OR = {
                has_state_category = metropolis
                has_state_category = megalopolis
            }
            infrastructure < constant:zombie_outbreak.high_risk_urban_infrastructure_threshold
        }
    }
}

is_extreme_risk_rear_zombie_outbreak_state = {
    is_high_risk_rear_zombie_outbreak_state = yes
    state_population_k > constant:zombie_outbreak.state_population_extreme_risk_threshold
    OR = {
        AND = {
            OR = {
                has_state_category = metropolis
                has_state_category = megalopolis

            }
            industrial_complex < constant:zombie_outbreak.very_low_industry_threshold

        }

            infrastructure < constant:zombie_outbreak.very_low_infrastructure_threshold
            industrial_complex < constant:zombie_outbreak.very_low_industry_threshold
            arms_factory < constant:zombie_outbreak.very_low_industry_threshold
        }
    }
}
