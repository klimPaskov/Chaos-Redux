    OR = {
        has_country_flag = used_unconventional_weapon
        has_country_flag = used_bioweapon
        has_country_flag = used_anthrax_weapon
        has_country_flag = used_plague_weapon
        has_country_flag = used_tularemia_weapon
        has_country_flag = used_smallpox_weapon

            has_variable = chem_warfare_condemnation
            check_variable = { chem_warfare_condemnation > 0 }
        }
    }
}
