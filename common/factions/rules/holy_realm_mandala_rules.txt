# Overview:
# - Faction rules for the Mandala of Nations.
# - The rules make the peaceful Holy Realm faction visibly defensive and bind
#   members to acts of kindness rather than offensive wars.

@thr_mandala_rule_stability = 0.04
@thr_mandala_rule_pp = 0.05
@thr_mandala_rule_defensive_stability = 0.10
@thr_mandala_rule_research_sharing = 0.15

thr_joining_rule_mandala_peace_only = {
	type = joining_rules

	visible = {
		has_faction_template = faction_template_holy_realm_mandala_of_nations
	}

	available = {
		has_country_flag = holy_realm_mandala_of_nations_leader
	}

	can_remove = {
		always = no
	}

	trigger = {
		holy_realm_sangha_compact_candidate_valid = yes
	}

	ai_will_do = {
		base = 0.1
		modifier = {
			factor = 0
			NOT = { holy_realm_sangha_compact_candidate_valid = yes }
		}
		modifier = {
			factor = 3
			any_enemy_country = { is_special_chaos_country = yes }
		}
		modifier = {
			factor = 2
			any_neighbor_country = { is_special_chaos_country = yes }
		}
		modifier = {
			factor = 2
			has_country_flag = holy_realm_bodhisattva_recent_visit
		}
		modifier = {
			factor = 1.5
			has_country_flag = holy_realm_mandala_observer_recent
		}
		modifier = {
			factor = 1.5
			has_country_flag = holy_realm_mandala_development_recent_country
		}
		modifier = {
			factor = 1.5
			FROM = { has_country_flag = holy_realm_governance_pilgrim_assembly }
			has_stability < 0.45
		}
	}
}

thr_member_rule_vow_against_offensive_war = {
	type = war_declaration_rule

	visible = {
		has_faction_template = faction_template_holy_realm_mandala_of_nations
	}

	available = {
		has_country_flag = holy_realm_mandala_of_nations_leader
	}

	can_remove = {
		always = no
	}

	trigger = {
		always = no
	}

	ai_will_do = {
		base = 0
	}
}

thr_member_rule_acts_of_kindness = {
	type = member_rules

	visible = {
		has_faction_template = faction_template_holy_realm_mandala_of_nations
	}

	available = {
		has_country_flag = holy_realm_mandala_of_nations_leader
	}

	can_remove = {
		always = no
	}

	trigger = {
		always = yes
	}

	modifier = {
		stability_factor = @thr_mandala_rule_stability
		political_power_gain = @thr_mandala_rule_pp
		defensive_war_stability_factor = @thr_mandala_rule_defensive_stability
		research_sharing_per_country_bonus_factor = @thr_mandala_rule_research_sharing
	}

	ai_will_do = {
		base = 0
	}
}
