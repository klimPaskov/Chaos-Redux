# Overview:
# - Positive-only faction rules for the Anti-Zombie League.
# - Exposes the coalition's coordinated command and shared research apparatus
#   in the vanilla faction UI instead of keeping all benefits hidden.

@azl_rule_joint_planning_speed = 0.03
@azl_rule_joint_org_regain = 0.03
@azl_rule_joint_defence_factor = 0.03
@azl_rule_shared_labs_special_project_speed = 0.05
@azl_rule_shared_labs_research_sharing = 0.25
@azl_rule_shared_labs_research_speed = 0.02

azl_joining_rule_human_survivors_only = {
	type = joining_rules

	visible = {
		has_faction_template = faction_template_anti_zombie_league
	}

	available = {
		has_global_flag = has_zombie_outbreak
	}

	can_remove = {
		always = no
	}

	trigger = {
		has_country_flag = azl_can_join_later
		should_actively_join_anti_zombie_league = yes
	}

	ai_will_do = {
		base = 0.1

		modifier = {
			factor = 8
			has_regional_zombie_threat = yes
		}
		modifier = {
			factor = 4
			has_same_continent_zombie_presence = yes
			NOT = { zombie_threat_below_serious_join_threshold = yes }
		}
		modifier = {
			factor = 6
			zombie_threat_is_global_emergency = yes
		}
	}
}
