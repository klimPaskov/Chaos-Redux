# Overview:
# - Shows the completed weaponized zombie profile only to the country that created it.
# - Uses one internal category per resolved zombie type so the category picture can match the final outbreak portrait.

chaosx_weaponized_zombie_creator_infected_category = {
	visible = {
		has_country_flag = weaponized_zombies_completed
		has_country_flag = weaponized_zombie_type_infected
	}
	visible_when_empty = yes
	icon = GFX_decision_category_generic_crisis
	picture = GFX_portrait_ZZZ_weaponized_infected
	priority = 101
}
