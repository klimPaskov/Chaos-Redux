chaosx_communism_fight_category = {

	communism_national_counter_agitation = {
		icon = GFX_decision_generic_political_discourse
		days_remove = constant:communism_spread_timing.national_counter_agitation_days
		days_re_enable = constant:communism_spread_timing.national_counter_agitation_reenable_days
		custom_cost_text = communism_national_counter_agitation_cost
		ai_hint_pp_cost = 35

		custom_cost_trigger = {
			command_power > communism_counter_agitation_command_power_cost
			has_political_power > communism_counter_agitation_political_power_cost
		}

		available = {
			hidden_trigger = { NOT = { is_special_chaos_country = yes } }
			NOT = { has_government = communism }
			communism > 0.001
		}

		visible = {
			has_global_flag = communism_spread
			hidden_trigger = { NOT = { is_special_chaos_country = yes } }
			NOT = { has_government = communism }
			has_idea = communism_spread_idea
		}

		ai_will_do = {
			base = 1
			modifier = {
				add = 2
				communism > constant:communism_spread_state_control.min_support_for_state_spread
			}
			modifier = {
				add = 2
				has_communist_controlled_state = yes
			}
		}

		complete_effect = {
			add_political_power = communism_counter_agitation_political_power_cost_negative
			add_command_power = communism_counter_agitation_command_power_cost_negative
			add_stability = constant:communism_spread_countermeasures.national_stability_cost
		}

		remove_effect = {
			add_popularity = {
				ideology = communism
				popularity = constant:communism_spread_countermeasures.national_support_reduction
			}
			hidden_effect = {
				reduce_communism_support_per_controlled_state = yes
				refresh_communism_spread_pressure = yes
				try_defeat_country_communism_spread_system = yes
			}
		}
	}

	communism_local_military_intervention = {
		icon = {
			key = GFX_decision_NOR_prepare_state_comm_1
			trigger = { FROM = { has_state_flag = communist_control_level_1 } }
		}
		icon = {
			key = GFX_decision_NOR_prepare_state_comm_2
			trigger = { FROM = { has_state_flag = communist_control_level_2 } }
		}
		icon = {
			key = GFX_decision_NOR_prepare_state_comm_3
			trigger = { FROM = { has_state_flag = communist_control_level_3 } }
		}
		icon = {
			key = GFX_decision_NOR_prepare_state_comm_0
			trigger = {
				FROM = {
					NOT = { has_state_flag = communist_control_level_1 }
					NOT = { has_state_flag = communist_control_level_2 }
					NOT = { has_state_flag = communist_control_level_3 }
				}
			}
		}
		state_target = any_owned_state
		on_map_mode = map_and_decisions_view
		days_remove = constant:communism_spread_timing.local_intervention_days
		days_re_enable = constant:communism_spread_timing.local_intervention_reenable_days
		custom_cost_text = communism_local_military_intervention_cost
		ai_hint_pp_cost = 50

		highlight_states = {
			highlight_states_trigger = {
				is_owned_by = ROOT
				OR = {
					has_state_flag = communist_control_level_1
					has_state_flag = communist_control_level_2
					has_state_flag = communist_control_level_3
				}
			}
			highlight_color_before_active = 2
			highlight_color_while_active = 2
		}

		target_root_trigger = {
			has_global_flag = communism_spread
			hidden_trigger = { NOT = { is_special_chaos_country = yes } }
			NOT = { has_government = communism }
			OR = {
				has_idea = communism_spread_idea
				has_communist_controlled_state = yes
			}
		}

		target_trigger = {
			FROM = {
				is_owned_by = ROOT
				has_state_flag = communist_controlled_state
			}
		}

		custom_cost_trigger = {
			command_power > FROM.communist_intervention_command_power_cost
			has_manpower > FROM.communist_intervention_manpower_cost
			has_equipment = {
				infantry_equipment > FROM.communist_intervention_equipment_cost
			}
		}

		available = {
			command_power > FROM.communist_intervention_command_power_cost
			has_manpower > FROM.communist_intervention_manpower_cost
			has_equipment = {
				infantry_equipment > FROM.communist_intervention_equipment_cost
			}
		}

		visible = {
			has_global_flag = communism_spread
			hidden_trigger = { NOT = { is_special_chaos_country = yes } }
			NOT = { has_government = communism }
			OR = {
				has_idea = communism_spread_idea
				has_communist_controlled_state = yes
			}
		}

		ai_will_do = {
			base = 1
			modifier = {
				add = 3
				FROM = { has_state_flag = communist_control_level_2 }
			}
			modifier = {
				add = 4
				FROM = { has_state_flag = communist_control_level_3 }
			}
			modifier = {
				factor = 0.4
				has_war = yes
				enemies_strength_ratio > 0.7
			}
			modifier = {
				factor = 0.5
				has_stability < 0.35
			}
		}

		complete_effect = {
			hidden_effect = {
				apply_communism_local_intervention_payment = yes
				if = {
					limit = { FROM = { has_state_flag = communist_worker_ritual_state } }
					add_command_power = -10
					add_stability = -0.005
				}
				if = {
					limit = { FROM = { has_state_flag = communist_world_revolution_whisper_state } }
					add_war_support = -0.005
				}
			}
		}

		remove_effect = {
			hidden_effect = {
				FROM = { save_event_target_as = communist_target_state }
				if = {
					limit = { FROM = { has_state_flag = communist_control_level_1 } }
					random_list = {
						constant:communism_spread_intervention.level_1_success = {
							FROM = {
								clear_communism_state_control = yes
								add_dynamic_modifier = {
									modifier = communism_post_crackdown_scars
									days = constant:communism_spread_timing.crackdown_scars_days
								}
							}
							country_event = { id = chaosx.nr1.6 }
						}
						constant:communism_spread_intervention.level_1_partial = {
							FROM = {
								add_communism_intervention_progress_partial = yes
								try_finish_communism_level_1_intervention = yes
							}
							add_stability = -0.005
							country_event = { id = chaosx.nr1.6 }
						}
						constant:communism_spread_intervention.level_1_bad_failure = {
							FROM = { set_communism_state_control_level_2 = yes }
							add_popularity = {
								ideology = communism
								popularity = constant:communism_spread_intervention.failed_support_gain
							}
							country_event = { id = chaosx.nr1.7 }
						}
					}
				}
				else_if = {
					limit = { FROM = { has_state_flag = communist_control_level_2 } }
					random_list = {
						constant:communism_spread_intervention.level_2_success = {
							FROM = {
								add_communism_intervention_progress_success = yes
								try_finish_communism_level_2_intervention = yes
							}
							country_event = { id = chaosx.nr1.6 }
						}
						constant:communism_spread_intervention.level_2_partial = {
							FROM = {
								add_communism_intervention_progress_partial = yes
								try_finish_communism_level_2_intervention = yes
							}
							add_stability = -0.01
							country_event = { id = chaosx.nr1.6 }
						}
						constant:communism_spread_intervention.level_2_bad_failure = {
							FROM = { set_communism_state_control_level_3 = yes }
							add_popularity = {
								ideology = communism
								popularity = constant:communism_spread_intervention.failed_support_gain
							}
							country_event = { id = chaosx.nr1.7 }
						}
					}
				}
				else_if = {
					limit = { FROM = { has_state_flag = communist_control_level_3 } }
					random_list = {
						constant:communism_spread_intervention.level_3_suppression = {
							FROM = { apply_communism_local_suppression = yes }
							add_stability = -0.015
							country_event = { id = chaosx.nr1.6 }
						}
						constant:communism_spread_intervention.level_3_bad_failure = {
							FROM = { advance_communism_state_control_duration = yes }
							add_popularity = {
								ideology = communism
								popularity = constant:communism_spread_intervention.failed_support_gain
							}
							add_stability = -0.02
							country_event = { id = chaosx.nr1.7 }
						}
					}
				}
				refresh_communism_spread_pressure = yes
			}
		}
	}

	communism_emergency_intervention = {
		icon = GFX_decision_generic_protection
		days_remove = constant:communism_spread_timing.emergency_intervention_days
		days_re_enable = constant:communism_spread_timing.emergency_intervention_reenable_days
		custom_cost_text = communism_emergency_intervention_cost
		ai_hint_pp_cost = 100

		custom_cost_trigger = {
			command_power > communism_emergency_command_power_cost
			has_manpower > communism_emergency_manpower_cost
			has_equipment = { infantry_equipment > communism_emergency_equipment_cost }
			has_political_power > communism_emergency_political_power_cost
		}

		available = {
			hidden_trigger = { NOT = { is_special_chaos_country = yes } }
			can_launch_communist_emergency_intervention = yes
		}

		visible = {
			has_global_flag = communism_spread
			hidden_trigger = { NOT = { is_special_chaos_country = yes } }
			NOT = { has_government = communism }
			has_idea = communism_spread_idea
			has_communist_controlled_state = yes
		}

		ai_will_do = {
			base = 0
			modifier = {
				add = 4
				check_variable = { communism_control_level_3_states > 0 }
			}
			modifier = {
				add = 3
				check_variable = { communism_controlled_states > communism_emergency_control_threshold }
			}
			modifier = {
				factor = 0.25
				has_war = yes
				enemies_strength_ratio > 0.7
			}
			modifier = {
				factor = 0.5
				has_stability < 0.35
			}
		}

		complete_effect = {
			add_political_power = communism_emergency_political_power_cost_negative
			add_command_power = communism_emergency_command_power_cost_negative
			add_manpower = communism_emergency_manpower_cost_negative
			add_equipment_to_stockpile = {
				type = infantry_equipment
				amount = communism_emergency_equipment_cost_negative
			}
			add_stability = constant:communism_spread_emergency.stability_loss
			add_war_support = constant:communism_spread_emergency.war_support_loss
			add_dynamic_modifier = {
				modifier = communism_emergency_disruption
				days = constant:communism_spread_timing.post_emergency_disruption_days
			}
		}

		remove_effect = {
			hidden_effect = {
				prepare_communist_emergency_uprising_roll = yes
				random_list = {
					var:communist_uprising_chance = {
						create_communist_emergency_uprising = yes
						country_event = { id = chaosx.nr1.10 }
					}
					var:communist_no_uprising_chance = {
						resolve_communist_emergency_without_uprising = yes
						country_event = { id = chaosx.nr1.9 }
					}
				}
			}
		}
	}
