#Low Scientist XP gain
sp_biowarfare_generic_reward_scientist_xp_1 = {

    fire_only_once = yes

    threshold = {
        min = 0
        max = 100
    }

    # ROOT is the project
    # FROM is the Country
    # var:facility_state is the State where the facility is (ensured to be set in this case)
    # var:facility_province_id is the province ID where the facility is (ensured to be set in this case)
    # var:scientist is the Scientist (ensured to be set in this case)
    weight = {
        base = constant:bio_prototype_reward_weight.base
        modifier = {
            factor = constant:bio_prototype_reward_weight.theme_repeat_factor
            has_project_flag = sp_biowarfare_generic_reward_scientist_xp_flag
        }
        modifier = {
            factor = 0
            var:scientist = {
                has_scientist_level = {
                    level > 4
                    specialization = specialization_biowarfare
                }
            }
        }
    }

    option = {
        token = sp_biowarfare_generic_reward_option_scientist_xp

        iteration_output = {
            scientist_effects = {
                if = {
                    limit = {

                    }
                    add_scientist_xp = {
                        experience = constant:sp_scientist_xp_gain.low
                        specialization = specialization_biowarfare
                    }
                }
                FROM = { set_project_flag = sp_biowarfare_generic_reward_scientist_xp_flag }
            }
        }
    }
}

    }

    # ROOT is the project
    # FROM is the Country
    # var:facility_state is the State where the facility is (ensured to be set in this case)
    # var:facility_province_id is the province ID where the facility is (ensured to be set in this case)
    # var:scientist is the Scientist (ensured to be set in this case)

        base = 0.25
    }

    option = {
        token = sp_biowarfare_generic_reward_option_critical_failure

        iteration_output = {
            country_effects = {
                FROM = { add_project_progress_ratio = constant:sp_progress.loss.medium }
            }

            scientist_effects = {
                add_scientist_xp = {
                    experience = constant:sp_scientist_xp_gain.low
                    specialization = specialization_biowarfare
                }
                injure_scientist_for_days = 30
            }

            facility_state_effects = {
                damage_building = {
                    tags = facility
                    damage = 0.25
                    province = var:facility_province_id
                }
            }
        }
    }
}
