# Holy Realm AI strategy package.
# The Holy Realm stays defensive early, can open a separate expansion tree,
# and foreign majors become more willing to contain the active carrier.

holy_realm_refuge_defence = {
	allowed = { always = yes }
	enable = {
		is_holy_realm_country = yes
		NOT = { holy_realm_has_buddha_mandate = yes }
	}
	abort_when_not_enabled = yes
	ai_strategy = {
		type = avoid_starting_wars
		value = 150
	}
	ai_strategy = {
		type = force_defend_ally_borders
		target = asia
		value = 100
	}
}

holy_realm_himalayan_consolidation = {
	allowed = { always = yes }
	enable = {
		is_holy_realm_country = yes
		holy_realm_himalayan_mandala_incomplete = yes
		NOT = { has_war = yes }
		OR = {
			holy_realm_has_bodhisattva_stage = yes
			check_variable = { spiritual_legitimacy > 30 }
		}
	}
	abort_when_not_enabled = yes
	ai_strategy = {
		type = avoid_starting_wars
		value = -20
	}
}

holy_realm_expansionist_mandala = {
	allowed = { always = yes }
	enable = {
		is_holy_realm_country = yes
		has_country_flag = holy_realm_expansion_tree_opened
		NOT = { has_country_flag = holy_realm_final_doctrine_renounced }
		OR = {
			check_variable = { mandala_reach > 45 }
			holy_realm_chaos_above_intervention_floor = yes
			has_country_flag = holy_realm_memory_peace_letter_refused
		}
	}
	abort_when_not_enabled = yes
	ai_strategy = {
		type = build_army
		value = 80
	}
	ai_strategy = {
		type = avoid_starting_wars
		value = 30
	}
}

holy_realm_expansionist_forced_integration = {
	allowed = { always = yes }

		has_country_flag = holy_realm_expansion_path_hardened

	}
	abort_when_not_enabled = yes
	ai_strategy = {
		type = avoid_starting_wars
		value = -35
	}
}
