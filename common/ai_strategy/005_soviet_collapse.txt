soviet_collapse_soviet_low_threat_restraint = {
	allowed = { original_tag = SOV }
	enable = {
		is_soviet_collapse_active = yes
		check_variable = {
			var = soviet_collapse_total_collapse_threat
			value = constant:soviet_collapse_progressive_release.moderate_threat
			compare = less_than
		}
		check_variable = {
			var = soviet_collapse_moscow_authority
			value = constant:soviet_collapse_soviet_objective.contested_authority
			compare = greater_than
		}
	}
	abort_when_not_enabled = yes
	ai_strategy = {
		type = avoid_starting_wars
		value = @SOV_COLLAPSE_AI_SOV_LOW_THREAT_RESTRAINT
	}
}

soviet_collapse_soviet_high_threat_recovery = {
	allowed = { original_tag = SOV }
	enable = {
		is_soviet_collapse_active = yes
		OR = {
			check_variable = {
				var = soviet_collapse_total_collapse_threat
				value = constant:soviet_collapse_progressive_release.high_threat
				compare = greater_than_or_equals
			}
			check_variable = {
				var = soviet_collapse_breakaway_count
				value = constant:soviet_collapse_threat_guard.early_breakaway_gate
				compare = greater_than_or_equals
			}
			check_variable = {
				var = soviet_collapse_military_obedience
				value = constant:soviet_collapse_soviet_objective.contested_authority
				compare = less_than
			}
		}
	}
	abort_when_not_enabled = yes
	ai_strategy = {
		type = build_army
		value = @SOV_COLLAPSE_AI_SOV_HIGH_THREAT_ARMY
	}
	ai_strategy = {
		type = avoid_starting_wars
		value = @SOV_COLLAPSE_AI_SOV_DEFENSE
	}
}
