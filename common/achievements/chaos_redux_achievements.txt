# Chaos Redux custom achievements.
#
# Structure:
# - Internal ID prefixes group achievements by intended rarity in the in-game list.
# - All achievements are available as any country.
# - The mod achievement UI resolves icon art from gfx/achievements/<achievement_id>*.dds.

00_calm_before_the_storm_achievement = {
	possible = {
		custom_override_tooltip = {
			tooltip = achievement_available_as_any_tag_tooltip
			always = yes
		}
	}

	happened = {
		custom_override_tooltip = {
			tooltip = calm_before_the_storm_achievement_tooltip
			date > 1939.12.31
			check_variable = {
				var = global.chaos_meter_value
				value = constant:chaos_meter_tier_range.tier_1.min
				compare = less_than
			}
			NOT = { has_global_flag = achievement_chaos_reached_gathering_storm_pre_1940 }
		}
	}
}
